# TradeAgent factor_service 因子计算服务实现方案

---

# 🧠 一、实现目标

```
阶段目标：实现 factor_service 因子计算服务

核心职责：
✅ 接收特征数据（Kafka订阅 feature_service）
✅ 计算各类因子（Trend、Flow、Sentiment、Macro、Behavioral、Historical）
✅ 因子标准化（z-score/tanh/min-max，统一到 [-1, +1]）
✅ 因子融合（加权合成 composite_score）
✅ 输出到下游（Risk Engine / Decision Engine / 前端）

输入：feature_service 的特征数据
输出：因子分数（factor_scores）+ 综合分数（composite_score）
```

# 🔧 四、因子基类 `factors/base.py`

```python
from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import Dict, Optional
import time
import logging

logger = logging.getLogger(__name__)

@dataclass
class FactorResult:
    name: str
    value: float
    confidence: float
    metadata: Dict = None
    timestamp: int = None

    def __post_init__(self):
        if self.timestamp is None:
            self.timestamp = int(time.time())
        if self.metadata is None:
            self.metadata = {}

class BaseFactor(ABC):
    def __init__(self, name: str, weight: float, enabled: bool = True):
        self.name = name
        self.weight = weight
        self.enabled = enabled
        self.history: list = []
        self.max_history = 100

    @abstractmethod
    def calculate(self, features: Dict, events: List[Dict] = None) -> FactorResult:
        pass

    def normalize(self, value: float, method: str = "tanh") -> float:
        if method == "tanh":
            return float(max(-1, min(1, value)))
        elif method == "zscore":
            if len(self.history) < 10:
                return float(max(-1, min(1, value)))
            mean = sum(self.history) / len(self.history)
            std = (sum((x - mean) ** 2 for x in self.history) / len(self.history)) ** 0.5
            if std == 0:
                return 0.0
            zscore = (value - mean) / std
            return float(max(-1, min(1, zscore)))
        elif method == "minmax":
            if len(self.history) < 10:
                return 0.0
            min_val = min(self.history)
            max_val = max(self.history)
            if max_val == min_val:
                return 0.0
            normalized = 2 * (value - min_val) / (max_val - min_val) - 1
            return float(max(-1, min(1, normalized)))
        return 0.0

    def update_history(self, value: float):
        self.history.append(value)
        if len(self.history) > self.max_history:
            self.history.pop(0)

    def get_confidence(self, features: Dict) -> float:
        return 1.0
```

```python
from typing import Dict, List
from .base import BaseFactor, FactorResult
import logging

logger = logging.getLogger(__name__)

class TrendFactor(BaseFactor):
    def __init__(self, weight: float = 0.30):
        super().__init__("trend", weight)

    def calculate(self, features: Dict, events: List[Dict] = None) -> FactorResult:
        price_features = features.get("price_features", {})
        btc_features = price_features.get("BTC", {})

        if not btc_features:
            return FactorResult(
                name=self.name,
                value=0.0,
                confidence=0.0,
                metadata={"reason": "no_data"}
            )

        momentum = btc_features.get("momentum", 0)
        ma_alignment = btc_features.get("ma_alignment", 0)
        trend_strength = btc_features.get("trend_strength", 0)
        ma_slope = btc_features.get("ma_slope", 0)

        momentum_score = momentum * 10
        alignment_score = ma_alignment * 0.5
        strength_score = min(trend_strength * 10, 1.0) if trend_strength > 0 else max(trend_strength * 10, -1.0)
        slope_score = min(ma_slope * 100, 1.0) if ma_slope > 0 else max(ma_slope * 100, -1.0)

        raw_value = (
            0.4 * momentum_score +
            0.3 * alignment_score +
            0.2 * strength_score +
            0.1 * slope_score
        )

        normalized = self.normalize(raw_value, method="tanh")

        self.update_history(normalized)
        confidence = self.get_confidence(features)

        return FactorResult(
            name=self.name,
            value=normalized,
            confidence=confidence,
            metadata={
                "momentum_score": momentum_score,
                "alignment_score": alignment_score,
                "strength_score": strength_score,
                "slope_score": slope_score,
                "raw_value": raw_value
            }
        )

    def get_confidence(self, features: Dict) -> float:
        price_features = features.get("price_features", {})
        if len(price_features) >= 2:
            return 1.0
        return 0.5
```

```python
logger = logging.getLogger(__name__)

class FlowFactor(BaseFactor):
    def __init__(self, weight: float = 0.25):
        super().__init__("flow", weight)

    def calculate(self, features: Dict, events: List[Dict] = None) -> FactorResult:
        flow_features = features.get("flow_features", {})

        btc_net_flow = flow_features.get("btc_net_flow", 0)
        btc_flow_direction = flow_features.get("btc_flow_direction", 0)
        eth_net_flow = flow_features.get("eth_net_flow", 0)
        total_net_flow = flow_features.get("total_net_flow", 0)

        flow_scale = 1e8

        btc_flow_score = (btc_net_flow / flow_scale) * btc_flow_direction
        eth_flow_score = (eth_net_flow / flow_scale)
        total_flow_score = total_net_flow / flow_scale

        acceleration = flow_features.get("btc_flow_acceleration", 0)
        acceleration_score = min(acceleration, 1.0) if acceleration > 0 else max(acceleration, -1.0)

        raw_value = (
            0.5 * btc_flow_score +
            0.25 * eth_flow_score +
            0.15 * total_flow_score +
            0.1 * acceleration_score
        )

        normalized = self.normalize(raw_value, method="tanh")

        self.update_history(normalized)

        return FactorResult(
            name=self.name,
            value=normalized,
            confidence=0.8 if btc_net_flow != 0 else 0.3,
            metadata={
                "btc_flow_score": btc_flow_score,
                "eth_flow_score": eth_flow_score,
                "total_flow_score": total_flow_score,
                "acceleration_score": acceleration_score,
                "raw_value": raw_value
            }
```

# 🔧 七、情绪因子 `factors/sentiment_factor.py`

```python
from typing import Dict, List
from .base import BaseFactor, FactorResult
import logging
import time

logger = logging.getLogger(__name__)

class SentimentFactor(BaseFactor):
    def __init__(self, weight: float = 0.20):
        super().__init__("sentiment", weight)
        self.event_history: List[Dict] = []

    def calculate(self, features: Dict, events: List[Dict] = None) -> FactorResult:
        if not events:
            return FactorResult(
                name=self.name,
                value=0.0,
                confidence=0.0,
                metadata={"reason": "no_events"}
            )

        self.event_history.extend(events)
        if len(self.event_history) > 100:
            self.event_history = self.event_history[-100:]

        total_sentiment = 0.0
        total_weight = 0.0
        event_count = 0

        current_time = time.time()

        for event in self.event_history:
            sentiment_score = event.get("sentiment_score", 0)
            event_time = event.get("timestamp", current_time)
            time_diff = current_time - event_time

            decay = self._time_decay(time_diff)
            importance = self._importance_weight(event)

            weight = decay * importance
            total_sentiment += sentiment_score * weight
            total_weight += weight
            event_count += 1

        if total_weight == 0:
            return FactorResult(
                name=self.name,
                value=0.0,
                confidence=0.0,
                metadata={"reason": "no_valid_events"}
            )

        raw_value = total_sentiment / total_weight

        normalized = self.normalize(raw_value, method="tanh")

        self.update_history(normalized)

        confidence = min(event_count / 10, 1.0)

        return FactorResult(
            name=self.name,
            value=normalized,
            confidence=confidence,
            metadata={
                "event_count": event_count,
                "raw_value": raw_value,
                "avg_sentiment": raw_value
            }
        )

    def _time_decay(self, time_diff: float, half_life: float = 3600) -> float:
        import math
        return math.exp(-0.693 * time_diff / half_life)

    def _importance_weight(self, event: Dict) -> float:
        impact_map = {
            "very_high": 1.0,
            "high": 0.8,
            "medium": 0.5,
            "low": 0.3,
            "very_low": 0.1
        }
        impact = event.get("impact", "medium")
        return impact_map.get(impact, 0.5)
```

```python
from typing import Dict, List
from .base import BaseFactor, FactorResult
import logging

logger = logging.getLogger(__name__)

class BehavioralFactor(BaseFactor):
    def __init__(self, weight: float = 0.10):
        super().__init__("behavioral", weight)
        self.kol_sentiment_history = []
        self.kol_position_history = []

    def calculate(self, features: Dict, events: List[Dict] = None) -> FactorResult:
        flow_features = features.get("flow_features", {})

        btc_flow = flow_features.get("btc_net_flow", 0)
        eth_flow = flow_features.get("eth_net_flow", 0)

        social_events = [e for e in (events or []) if e.get("type") == "social"]

        kol_bullish = 0
        kol_bearish = 0
        total_kol_weight = 0

        for event in social_events:
            sentiment = event.get("sentiment_score", 0)
            influence = event.get("metadata", {}).get("influence_score", 5) / 10

            if sentiment > 0.2:
                kol_bullish += influence
            elif sentiment < -0.2:
                kol_bearish += influence

            total_kol_weight += influence

        flow_score = 0.5
        if btc_flow > 1e8:
            flow_score += 0.3
        elif btc_flow < -1e8:
            flow_score -= 0.3

        kol_score = 0.0
        if total_kol_weight > 0:
            kol_score = (kol_bullish - kol_bearish) / total_kol_weight

        raw_value = 0.6 * flow_score + 0.4 * kol_score

        normalized = self.normalize(raw_value, method="tanh")

        self.update_history(normalized)

        confidence = min(total_kol_weight / 5, 1.0) if total_kol_weight > 0 else 0.5

        return FactorResult(
            name=self.name,
            value=normalized,
            confidence=confidence,
            metadata={
                "flow_score": flow_score,
                "kol_score": kol_score,
                "kol_bullish": kol_bullish,
                "kol_bearish": kol_bearish,
                "total_kol_weight": total_kol_weight,
                "raw_value": raw_value
            }
        )
```

# 🔧 十、历史相似度因子 `factors/historical_factor.py`

```python
from typing import Dict, List, Optional
from .base import BaseFactor, FactorResult
import logging
import time

logger = logging.getLogger(__name__)

class HistoricalFactor(BaseFactor):
    def __init__(self, weight: float = 0.05, vector_db=None):
        super().__init__("historical", weight)
        self.vector_db = vector_db
        self.max_weight = 0.10

    def calculate(self, features: Dict, events: List[Dict] = None) -> FactorResult:
        if not self.vector_db:
            return FactorResult(
                name=self.name,
                value=0.0,
                confidence=0.0,
                metadata={"reason": "no_vector_db"}
            )

        current_features = self._extract_feature_vector(features)

        if not current_features:
            return FactorResult(
                name=self.name,
                value=0.0,
                confidence=0.0,
                metadata={"reason": "no_features"}
            )

        try:
            similar = self.vector_db.search(
                collection="market_features",
                query_vector=current_features,
                limit=5
            )

            if not similar:
                return FactorResult(
                    name=self.name,
                    value=0.0,
                    confidence=0.0,
                    metadata={"reason": "no_similar_found"}
                )

            avg_similarity = sum(s["score"] for s in similar) / len(similar)

            price_change = sum(s.get("price_change_after", 0) for s in similar) / len(similar)

            historical_score = avg_similarity * price_change * 10

            normalized = self.normalize(historical_score, method="tanh")

            confidence = min(avg_similarity * len(similar) / 3, 1.0)

            return FactorResult(
                name=self.name,
                value=normalized,
                confidence=confidence,
                metadata={
                    "avg_similarity": avg_similarity,
                    "price_change_after": price_change,
                    "similar_count": len(similar),
                    "scenario": similar[0].get("scenario", "unknown") if similar else None
                }
            )

        except Exception as e:
            logger.error(f"历史相似度计算失败: {e}")
            return FactorResult(
                name=self.name,
                value=0.0,
                confidence=0.0,
                metadata={"reason": "error", "error": str(e)}
            )

    def _extract_feature_vector(self, features: Dict) -> List[float]:
        price_features = features.get("price_features", {})
        btc = price_features.get("BTC", {})

        if not btc:
            return []

        return [
            btc.get("returns_1h", 0),
            btc.get("volatility", 0),
            btc.get("momentum", 0),
            btc.get("ma_alignment", 0),
            features.get("flow_features", {}).get("total_net_flow", 0) / 1e8,
            features.get("macro_features", {}).get("macro_signal", 0)
        ]
```

# 🔧 十一、因子融合 `fusion/scorer.py`

```python
from typing import Dict, List
from dataclasses import dataclass
from ..factors.base import FactorResult
import logging

logger = logging.getLogger(__name__)

@dataclass
class CompositeScore:
    symbol: str
    factor_scores: Dict[str, float]
    composite_score: float
    regime: str
    confidence: float
    timestamp: int

class FactorScorer:
    def __init__(self, weights: Dict[str, float]):
        self.weights = weights

    def compute_composite(self, factor_results: List[FactorResult], symbol: str = "BTC") -> CompositeScore:
        factor_scores = {}
        total_weight = 0.0
        weighted_sum = 0.0
        min_confidence = 1.0

        for result in factor_results:
            if result.value == 0.0 and result.confidence == 0.0:
                continue

            factor_scores[result.name] = result.value

            weight = self.weights.get(result.name, 0)
            weighted_sum += result.value * weight
            total_weight += weight

            min_confidence = min(min_confidence, result.confidence)

        if total_weight == 0:
            composite = 0.0
        else:
            composite = weighted_sum / total_weight

        regime = self._detect_regime(factor_scores)

        return CompositeScore(
            symbol=symbol,
            factor_scores=factor_scores,
            composite_score=composite,
            regime=regime,
            confidence=min_confidence,
            timestamp=int(__import__("time").time())
        )

    def _detect_regime(self, factor_scores: Dict[str, float]) -> str:
        trend = factor_scores.get("trend", 0)
        flow = factor_scores.get("flow", 0)
        sentiment = factor_scores.get("sentiment", 0)
        macro = factor_scores.get("macro", 0)

        if trend > 0.3 and flow > 0.2 and sentiment > 0.2:
            return "RISK_ON"
        elif sentiment < -0.4 and macro < -0.3:
            return "RISK_OFF"
        elif abs(trend) < 0.2 and abs(flow) < 0.2:
            return "NEUTRAL"
        else:
            return "TRANSITIONAL"
```

# 🔧 十二、因子管理器 `factors/manager.py`

```python
from typing import Dict, List
from .trend_factor import TrendFactor
from .flow_factor import FlowFactor
from .sentiment_factor import SentimentFactor
from .macro_factor import MacroFactor
from .behavioral_factor import BehavioralFactor
from .historical_factor import HistoricalFactor
from .base import FactorResult
from ..fusion.scorer import FactorScorer, CompositeScore
from ..config import settings
import logging

logger = logging.getLogger(__name__)

class FactorManager:
    def __init__(self, vector_db=None):
        self.weights = settings.FACTOR_WEIGHTS

        self.factors = {
            "trend": TrendFactor(self.weights.get("trend", 0.30)),
            "flow": FlowFactor(self.weights.get("flow", 0.25)),
            "sentiment": SentimentFactor(self.weights.get("sentiment", 0.20)),
            "macro": MacroFactor(self.weights.get("macro", 0.15)),
            "behavioral": BehavioralFactor(self.weights.get("behavioral", 0.10)),
            "historical": HistoricalFactor(self.weights.get("historical", 0.05), vector_db)
        }

        self.scorer = FactorScorer(self.weights)

    def calculate_all(self, features: Dict, events: List[Dict] = None) -> CompositeScore:
        results = []

        for name, factor in self.factors.items():
            if not factor.enabled:
                continue

            try:
                if name in ["sentiment", "behavioral"]:
                    result = factor.calculate(features, events)
                else:
                    result = factor.calculate(features)

                results.append(result)
                logger.debug(f"{name} factor: {result.value:.4f} (conf: {result.confidence:.2f})")

            except Exception as e:
                logger.error(f"{name} factor calculation failed: {e}")

        return self.scorer.compute_composite(results)

    def update_weights(self, new_weights: Dict[str, float]):
        self.weights.update(new_weights)
        for name, weight in new_weights.items():
            if name in self.factors:
                self.factors[name].weight = weight

    def get_factor_status(self) -> Dict:
        return {
            name: {
                "enabled": factor.enabled,
                "weight": factor.weight,
                "history_len": len(factor.history)
            }
            for name, factor in self.factors.items()
        }
```

```python
        self.consumer = None
        self.factor_manager = FactorManager(vector_db)

    async def start(self):
        self.consumer = AIOKafkaConsumer(
            settings.KAFKA_TOPIC_FEATURES,
            bootstrap_servers=settings.KAFKA_BOOTSTRAP_SERVERS,
            group_id="factor_service",
            value_deserializer=lambda m: json.loads(m.decode('utf-8')),
            auto_offset_reset='latest'
        )
        await self.consumer.start()
        logger.info("Factor Consumer 已启动")

    async def stop(self):
        if self.consumer:
            await self.consumer.stop()

    async def consume(self):
        async for message in self.consumer:
            try:
                payload = message.value

                features = payload.get("price_features", {})
                flow_features = payload.get("flow_features", {})
                macro_features = payload.get("macro_features", {})

                events = payload.get("events", [])

                combined_features = {
                    "price_features": features,
                    "flow_features": flow_features,
                    "macro_features": macro_features
                }

                composite = self.factor_manager.calculate_all(combined_features, events)

                output = {
                    "symbol": composite.symbol,
                    "factor_scores": composite.factor_scores,
                    "composite_score": composite.composite_score,
                    "regime": composite.regime,
                    "confidence": composite.confidence,
                    "timestamp": composite.timestamp
                }

                await factor_producer.send_factors(output)

            except Exception as e:
                logger.error(f"因子计算失败: {e}")
```

```python
from fastapi import FastAPI, WebSocket
from contextlib import asynccontextmanager
import asyncio
import logging

from .queue.consumer import FactorConsumer
from .queue.producer import factor_producer
from .websocket.manager import ws_manager
from shared.state import state_client
from shared.config import settings

logger = logging.getLogger(__name__)

consumer = None

@asynccontextmanager
async def lifespan(app: FastAPI):
    global consumer
    await factor_producer.start()

    vector_db = None

    consumer = FactorConsumer(vector_db)
    asyncio.create_task(consumer.consume())

    state_client.update_data_status("RUNNING")
    yield

    await consumer.stop()
    await factor_producer.stop()
    state_client.update_data_status("STOPPED")

app = FastAPI(
    title="TradeAgent Factor Service",
    version="1.0.0",
    lifespan=lifespan
)

@app.websocket("/ws/factors")
async def websocket_endpoint(websocket: WebSocket):
    await ws_manager.connect(websocket)
    try:
        while True:
            await websocket.receive_text()
    except Exception:
        ws_manager.disconnect(websocket)

@app.get("/health")
async def health():
    return {
        "status": "ok",
        "service": settings.SERVICE_NAME
    }

@app.get("/api/v1/factors/status")

    if consumer:
        return consumer.factor_manager.get_factor_status()
    return {}

@app.get("/api/v1/factors/weights")
async def get_weights():
    if consumer:

    return {}
```
