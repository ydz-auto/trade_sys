# TradeAgent regime_service 市场状态服务实现方案

---

# 🧠 一、实现目标

```
阶段目标：实现 regime_service 市场状态服务

核心职责：
✅ 接收因子数据（Kafka订阅 factor_service）
✅ 判断市场状态（RISK_ON / RISK_OFF / NEUTRAL / TRENDING / RANGE / PANIC / EUPHORIA）
✅ 多时间框架融合（1h / 1d / 5m）
✅ 惯性机制（防止频繁切换）
✅ 计算置信度和风险指数
✅ 输出驱动因素（drivers）

输入：factor_service 的因子分数
输出：市场状态（regime）+ 置信度 + 风险等级 + 驱动因素
```

# 🔧 四、Regime状态定义 `regime/classifier.py`

```python
from enum import Enum
from dataclasses import dataclass
from typing import Dict, List, Optional
import logging

logger = logging.getLogger(__name__)

class RegimeType(Enum):
    TRENDING = "TRENDING"
    RANGE = "RANGE"
    PANIC = "PANIC"
    EUPHORIA = "EUPHORIA"
    RISK_OFF = "RISK_OFF"
    UNCERTAIN = "UNCERTAIN"

@dataclass
class RegimeOutput:
    regime: str
    confidence: float
    risk_level: int
    drivers: List[str]
    regime_scores: Dict[str, float]
    timeframe: str
    duration: int
    timestamp: int
    metadata: Dict = None

    def __post_init__(self):
        if self.metadata is None:
            self.metadata = {}

class RegimeClassifier:
    REGIME_RULES = {
        "PANIC": {
            "condition": "vol_score > 0.8 AND trend_score < -0.5",
            "priority": 1,
            "action": "禁止开多/减仓"
        },
        "EUPHORIA": {
            "condition": "sentiment_score > 0.7 AND trend_score > 0.5",
            "priority": 2,
            "action": "可能反转/轻仓做空"
        },
        "RISK_OFF": {
            "condition": "macro_score < -0.6",
            "priority": 3,
            "action": "降低风险暴露"
        },
        "TRENDING": {
            "condition": "abs(trend_score) > 0.6 AND vol_score < 0.6",
            "priority": 4,
            "action": "趋势跟踪"
        },
        "RANGE": {
            "condition": "abs(trend_score) < 0.3 AND vol_score < 0.5",
            "priority": 5,
            "action": "震荡/均值回归"
        }
    }

    def classify(self, factor_scores: Dict[str, float]) -> tuple[str, Dict[str, float]]:
        scores = self._compute_subscores(factor_scores)

        regime_scores = {}

        regime_scores["PANIC"] = 1.0 if scores["vol"] > 0.8 and scores["trend"] < -0.5 else 0.0
        regime_scores["EUPHORIA"] = 1.0 if scores["sentiment"] > 0.7 and scores["trend"] > 0.5 else 0.0
        regime_scores["RISK_OFF"] = 1.0 if scores["macro"] < -0.6 else 0.0
        regime_scores["TRENDING"] = 1.0 if abs(scores["trend"]) > 0.6 and scores["vol"] < 0.6 else 0.0
        regime_scores["RANGE"] = 1.0 if abs(scores["trend"]) < 0.3 and scores["vol"] < 0.5 else 0.0

        detected_regime = self._select_regime_by_priority(regime_scores)

        return detected_regime, regime_scores

    def _compute_subscores(self, factor_scores: Dict[str, float]) -> Dict[str, float]:
        trend = factor_scores.get("trend", 0)
        sentiment = factor_scores.get("sentiment", 0)
        macro = factor_scores.get("macro", 0)
        flow = factor_scores.get("flow", 0)
        historical = factor_scores.get("historical", 0)

        vol_score = abs(flow) * 0.5 + abs(historical) * 0.3 + (1 - abs(trend)) * 0.2
        vol_score = max(-1, min(1, vol_score))

        return {
            "trend": trend,
            "sentiment": sentiment,
            "macro": macro,
            "vol": vol_score,
            "flow": flow
        }

    def _select_regime_by_priority(self, regime_scores: Dict[str, float]) -> str:
        priorities = {
            "PANIC": 1,
            "EUPHORIA": 2,
            "RISK_OFF": 3,
            "TRENDING": 4,
            "RANGE": 5
        }

        active_regimes = [(r, s) for r, s in regime_scores.items() if s > 0.5]

        if not active_regimes:
            return "NEUTRAL"

        active_regimes.sort(key=lambda x: priorities.get(x[0], 99))

        return active_regimes[0][0]
```

# 🔧 五、置信度计算 `regime/scorer.py`

```python
from typing import Dict
import logging

logger = logging.getLogger(__name__)

class ConfidenceScorer:
    def __init__(self, min_samples: int = 10):
        self.min_samples = min_samples
        self.sample_count = 0

    def compute(self, regime_scores: Dict[str, float], data_complete: bool = True) -> float:
        self.sample_count += 1

        sorted_scores = sorted(regime_scores.values(), reverse=True)

        if len(sorted_scores) < 2:
            return 0.5

        top_score = sorted_scores[0]
        second_score = sorted_scores[1] if len(sorted_scores) > 1 else 0.0

        gap = top_score - second_score

        confidence = 0.5 + 0.5 * min(gap, 1.0)

        if not data_complete:
            confidence *= 0.8

        if self.sample_count < self.min_samples:
            confidence *= 0.9

        return max(0.0, min(1.0, confidence))

    def get_confidence_level(self, confidence: float) -> str:
        if confidence >= 0.9:
            return "VERY_HIGH"
        elif confidence >= 0.7:
            return "HIGH"
        elif confidence >= 0.6:
            return "MEDIUM"
        elif confidence >= 0.5:
            return "LOW"
        else:
            return "VERY_LOW"

    def get_action_by_confidence(self, confidence: float, base_action: str) -> str:
        level = self.get_confidence_level(confidence)

        action_modifiers = {
            "VERY_HIGH": "",
            "HIGH": "",
            "MEDIUM": "降低仓位50%",
            "LOW": "观望",
            "VERY_LOW": "不执行"
        }

        modifier = action_modifiers.get(level, "")
        return f"{base_action} + {modifier}" if modifier else base_action
```

# 🔧 六、惯性机制 `regime/hysteresis.py`

```python
from typing import Dict, Optional
from .classifier import RegimeType
from ..config import settings
import logging

logger = logging.getLogger(__name__)

class HysteresisManager:
    def __init__(self):
        self.current_regime: Optional[str] = None
        self.pending_regime: Optional[str] = None
        self.regime_duration: int = 0
        self.min_duration = settings.MIN_DURATION
        self.min_duration_urgent = settings.MIN_DURATION_URGENT
        self.impossible_transitions = settings.IMPOSSIBLE_TRANSITIONS

    def update(self, raw_regime: str) -> tuple[bool, Optional[str]]:
        if raw_regime == self.current_regime:
            self.regime_duration += 1
            self.pending_regime = None
            return False, None

        if raw_regime == self.pending_regime:
            self.regime_duration += 1
        else:
            self.pending_regime = raw_regime
            self.regime_duration = 1
            return False, None

        min_dur = self.min_duration_urgent if raw_regime == "PANIC" else self.min_duration

        if self.regime_duration >= min_dur:
            if self._can_transition(raw_regime):
                self.current_regime = raw_regime
                self.regime_duration = 0
                self.pending_regime = None
                return True, raw_regime

        return False, None

    def _can_transition(self, to_regime: str) -> bool:
        if self.current_regime is None:
            return True

        impossible = self.impossible_transitions.get(self.current_regime, [])
        if to_regime in impossible:
            logger.warning(f"不可能的转换: {self.current_regime} -> {to_regime}")
            return False

        return True

    def force_transition(self, new_regime: str):
        self.current_regime = new_regime
        self.pending_regime = None
        self.regime_duration = 0
        logger.info(f"强制转换到: {new_regime}")

    def get_current_regime(self) -> Optional[str]:
        return self.current_regime

    def reset(self):
        self.current_regime = None
        self.pending_regime = None
        self.regime_duration = 0
```

# 🔧 七、风险指数计算 `risk/index.py`

```python
from typing import Dict
import logging

logger = logging.getLogger(__name__)

class RiskIndexCalculator:
    WEIGHTS = {
        "volatility": 0.4,
        "drawdown": 0.3,
        "flow": 0.2,
        "macro": 0.1
    }

    def calculate(self, factor_scores: Dict[str, float], metadata: Dict = None) -> tuple[int, str]:
        volatility = abs(factor_scores.get("flow", 0))

        drawdown = 0.0
        if metadata and "max_drawdown" in metadata:
            drawdown = min(abs(metadata["max_drawdown"]) / 20, 1.0)

        flow = abs(factor_scores.get("flow", 0))

        macro = abs(factor_scores.get("macro", 0))

        risk_index = (
            self.WEIGHTS["volatility"] * volatility +
            self.WEIGHTS["drawdown"] * drawdown +
            self.WEIGHTS["flow"] * flow +
            self.WEIGHTS["macro"] * macro
        )

        risk_level = int(min(risk_index * 100, 100))

        level_str = self._get_level(risk_level)

        return risk_level, level_str

    def _get_level(self, risk_level: int) -> str:
        if risk_level < 30:
            return "LOW"
        elif risk_level < 60:
            return "MEDIUM"
        elif risk_level < 80:
            return "HIGH"
        else:
            return "EXTREME"
```

# 🔧 八、驱动因素提取 `drivers/extractor.py`

```python
from typing import Dict, List
import logging

logger = logging.getLogger(__name__)

class DriverExtractor:
    DRIVER_THRESHOLDS = {
        "volatility_spike": 0.7,
        "ETF_outflow": -0.5,
        "ETF_inflow": 0.5,
        "negative_news": -0.6,
        "positive_news": 0.6,
        "gold_surge": -0.5,
        "oil_surge": -0.5,
        "fear_extreme": -0.6,
        "greed_extreme": 0.6,
        "volume_anomaly": 0.7,
        "drawdown_high": 0.15
    }

    def extract(self, factor_scores: Dict[str, float], metadata: Dict = None) -> List[str]:
        drivers = []

        flow = factor_scores.get("flow", 0)
        sentiment = factor_scores.get("sentiment", 0)
        macro = factor_scores.get("macro", 0)
        historical = factor_scores.get("historical", 0)

        vol_score = abs(flow) * 0.5 + abs(historical) * 0.5
        if vol_score > self.DRIVER_THRESHOLDS["volatility_spike"]:
            drivers.append("volatility_spike")

        if flow < self.DRIVER_THRESHOLDS["ETF_outflow"]:
            drivers.append("ETF_outflow")
        elif flow > self.DRIVER_THRESHOLDS["ETF_inflow"]:
            drivers.append("ETF_inflow")

        if sentiment < self.DRIVER_THRESHOLDS["negative_news"]:
            drivers.append("negative_news")
        elif sentiment > self.DRIVER_THRESHOLDS["positive_news"]:
            drivers.append("positive_news")

        if macro < self.DRIVER_THRESHOLDS["gold_surge"]:
            drivers.append("gold_surge")

        if metadata:
            if metadata.get("volume_anomaly", 0) > self.DRIVER_THRESHOLDS["volume_anomaly"]:
                drivers.append("volume_anomaly")

            drawdown = metadata.get("max_drawdown", 0)
            if abs(drawdown) > self.DRIVER_THRESHOLDS["drawdown_high"]:
                drivers.append("drawdown_high")

        return drivers
```

# 🔧 九、多时间框架融合 `timeframe/fusion.py`

```python
from typing import Dict, Tuple, Optional
import logging

logger = logging.getLogger(__name__)

class TimeFrameFusion:
    WEIGHTS = {
        "1h": 0.5,
        "1d": 0.3,
        "5m": 0.2
    }

    def __init__(self):
        self.timeframe_data: Dict[str, Dict] = {
            "5m": {},
            "1h": {},
            "1d": {}
        }

    def update(self, timeframe: str, factor_scores: Dict, regime: str):
        self.timeframe_data[timeframe] = {
            "factor_scores": factor_scores,
            "regime": regime,
            "timestamp": int(__import__("time").time())
        }

    def fuse(self) -> Tuple[Optional[str], float]:
        tf_1h = self.timeframe_data.get("1h", {}).get("regime")
        tf_1d = self.timeframe_data.get("1d", {}).get("regime")
        tf_5m = self.timeframe_data.get("5m", {}).get("regime")

        if not tf_1h or not tf_1d:
            return None, 0.0

        if tf_1h != tf_1d:
            logger.warning(f"时间框架不一致: 1h={tf_1h}, 1d={tf_1d}")
            return "UNCERTAIN", 0.5

        final_regime = tf_1h

        confidence = self.WEIGHTS["1h"] + self.WEIGHTS["1d"]

        if tf_5m == tf_1h:
            confidence += self.WEIGHTS["5m"]
        else:
            confidence *= 0.7

        return final_regime, min(confidence, 1.0)

    def get_regime_for_timeframe(self, timeframe: str) -> Optional[str]:
        return self.timeframe_data.get(timeframe, {}).get("regime")
```

# 🔧 十、主检测器 `regime/detector.py`

```python
from typing import Dict, Optional
from .classifier import RegimeClassifier, RegimeOutput
from .scorer import ConfidenceScorer
from .hysteresis import HysteresisManager
from .risk.index import RiskIndexCalculator
from .drivers.extractor import DriverExtractor
from .timeframe.fusion import TimeFrameFusion
import logging
import time

logger = logging.getLogger(__name__)

class RegimeDetector:
    def __init__(self):
        self.classifier = RegimeClassifier()
        self.confidence_scorer = ConfidenceScorer()
        self.hysteresis = HysteresisManager()
        self.risk_calculator = RiskIndexCalculator()
        self.driver_extractor = DriverExtractor()
        self.timeframe_fusion = TimeFrameFusion()
        self.current_output: Optional[RegimeOutput] = None

    def detect(self, factor_scores: Dict[str, float], timeframe: str = "1h", metadata: Dict = None) -> RegimeOutput:
        raw_regime, regime_scores = self.classifier.classify(factor_scores)

        confidence = self.confidence_scorer.compute(regime_scores)

        transitioned, new_regime = self.hysteresis.update(raw_regime)

        if transitioned:
            final_regime = new_regime
            logger.info(f"Regime切换: {self.hysteresis.current_regime} -> {final_regime}")
        else:
            final_regime = self.hysteresis.get_current_regime() or raw_regime

        self.timeframe_fusion.update(timeframe, factor_scores, final_regime)

        fused_regime, fuse_confidence = self.timeframe_fusion.fuse()
        if fused_regime:
            final_regime = fused_regime
            confidence *= fuse_confidence

        risk_level, risk_str = self.risk_calculator.calculate(factor_scores, metadata)

        drivers = self.driver_extractor.extract(factor_scores, metadata)

        self.current_output = RegimeOutput(
            regime=final_regime,
            confidence=confidence,
            risk_level=risk_level,
            drivers=drivers,
            regime_scores=regime_scores,
            timeframe=timeframe,
            duration=self.hysteresis.regime_duration,
            timestamp=int(time.time()),
            metadata=metadata or {}
        )

        return self.current_output

    def force_regime(self, regime: str):
        self.hysteresis.force_transition(regime)
        logger.info(f"强制Regime: {regime}")

    def get_current(self) -> Optional[RegimeOutput]:
        return self.current_output
```

```python
import json
import asyncio
from aiokafka import AIOKafkaConsumer
from typing import Dict
import logging

from ..config import settings
from ..regime.detector import RegimeDetector
from .producer import regime_producer

logger = logging.getLogger(__name__)

class RegimeConsumer:
    def __init__(self):
        self.consumer = None
        self.detector = RegimeDetector()

    async def start(self):
        self.consumer = AIOKafkaConsumer(
            settings.KAFKA_TOPIC_FACTORS,
            bootstrap_servers=settings.KAFKA_BOOTSTRAP_SERVERS,
            group_id="regime_service",
            value_deserializer=lambda m: json.loads(m.decode('utf-8')),
            auto_offset_reset='latest'
        )
        await self.consumer.start()
        logger.info("Regime Consumer 已启动")

    async def stop(self):
        if self.consumer:
            await self.consumer.stop()

    async def consume(self):
        async for message in self.consumer:
            try:
                payload = message.value

                factor_scores = payload.get("factor_scores", {})
                symbol = payload.get("symbol", "BTC")
                timestamp = payload.get("timestamp", 0)

                metadata = {
                    "symbol": symbol,
                    "original_timestamp": timestamp
                }

                regime_output = self.detector.detect(factor_scores, metadata=metadata)

                output = {
                    "symbol": symbol,
                    "regime": regime_output.regime,
                    "confidence": regime_output.confidence,
                    "risk_level": regime_output.risk_level,
                    "drivers": regime_output.drivers,
                    "regime_scores": regime_output.regime_scores,
                    "timeframe": regime_output.timeframe,
                    "duration": regime_output.duration,
                    "timestamp": regime_output.timestamp
                }

                await regime_producer.send_regime(output)

                logger.debug(f"Regime检测: {symbol} = {regime_output.regime} (conf: {regime_output.confidence:.2f})")

            except Exception as e:
                logger.error(f"Regime检测失败: {e}")

    return {
        "status": "ok",
        "service": settings.SERVICE_NAME
    }

@app.get("/api/v1/regime/current")
async def get_current_regime():
    if consumer and consumer.detector.current_output:
        output = consumer.detector.current_output
        return {
            "regime": output.regime,
            "confidence": output.confidence,
            "risk_level": output.risk_level,
            "drivers": output.drivers,
            "timestamp": output.timestamp
        }
    return {"regime": "UNKNOWN"}

@app.get("/api/v1/regime/history")
async def get_regime_history(symbol: str = "BTC"):
    return redis_store.get_regime_history(symbol)

@app.post("/api/v1/regime/force")
async def force_regime(regime: str):
    if consumer:
        consumer.detector.force_regime(regime)
        return {"status": "ok", "regime": regime}
    return {"status": "error"}
```
