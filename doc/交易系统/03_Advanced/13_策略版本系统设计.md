# 🧠 一、模块定位

## 1.1 核心问题

你现在的系统是：
```
一个策略 → 跑到底
```

专业系统是：
```
多个策略版本 → 持续迭代 → 优胜劣汰
```

## 1.2 作用

策略版本系统让你能够：
- **灰度发布**：新策略先小资金验证
- **A/B测试**：对比不同策略表现
- **快速回滚**：出问题立即切回旧版本
- **持续优化**：不断改进策略而不影响实盘

## 1.3 在系统中的位置

```
Strategy Registry
       ↓
Strategy Selector → Decision Engine → Position → Execution
       ↑
  Portfolio Risk / Signal Quality
```

## 5.2 评估计算

```python
class StrategyEvaluator:
    def __init__(self, strategy):
        self.strategy = strategy
        self.equity_curve = []

    def compute_sharpe(self, returns):
        """年化Sharpe Ratio"""
        if len(returns) < 2:
            return None

        excess_returns = returns - RISK_FREE_RATE
        if np.std(excess_returns) == 0:
            return None

        return np.mean(excess_returns) / np.std(excess_returns) * np.sqrt(252)

    def compute_win_rate(self, trades):
        winning = sum(1 for t in trades if t["pnl"] > 0)
        return winning / len(trades) if trades else 0

    def compute_max_drawdown(self, equity_curve):
        """最大回撤"""
        peak = equity_curve[0]
        max_dd = 0

        for value in equity_curve:
            if value > peak:
                peak = value
            dd = (peak - value) / peak
            max_dd = max(max_dd, dd)

        return max_dd

    def compute_profit_factor(self, trades):
        gross_profit = sum(t["pnl"] for t in trades if t["pnl"] > 0)
        gross_loss = abs(sum(t["pnl"] for t in trades if t["pnl"] < 0))

        return gross_profit / gross_loss if gross_loss > 0 else None
```

## 5.3 升级评估

```python
def evaluate_upgrade(candidate_strategy, current_strategy):
    """
    评估是否应该升级到新版本
    """
    report = {
        "should_upgrade": False,
        "reasons": [],
        "warnings": [],
    }

    # 1. Sharpe 必须提升
    if candidate_strategy.sharpe <= current_strategy.sharpe:
        report["warnings"].append("Sharpe未提升")
        if candidate_strategy.sharpe < current_strategy.sharpe * 0.95:
            report["should_upgrade"] = False
            report["reasons"].append("Sharpe下降，拒绝升级")

    # 2. Max Drawdown 必须改善
    if candidate_strategy.max_drawdown >= current_strategy.max_drawdown:
        report["warnings"].append("Drawdown未改善")
    else:
        report["reasons"].append(f"Drawdown改善: {current_strategy.max_drawdown:.2%} → {candidate_strategy.max_drawdown:.2%}")

    # 3. 样本量要求
    if candidate_strategy.total_trades < 100:
        report["warnings"].append("交易样本不足")

    # 4. 一致性要求
    if candidate_strategy.sharpe > current_strategy.sharpe * 1.2:
        report["reasons"].append("Sharpe显著提升（>20%）")
        report["should_upgrade"] = True

    return report
```

# 💰 六、资金分配

## 6.1 分配策略（金丝雀 + 渐进式）

```python
ALLOCATION_STRATEGY = {
    "initial_testing": 0.10,      # 初始测试资金10%
    "max_testing": 0.20,          # 测试资金最多20%
    "min_production_to_keep": 0.30, # 旧版本降级时最少保留30%
    "promotion_threshold": 1.5,   # Sharpe比旧版本高50%才可晋升
}

def compute_allocation(active_strategies, market_regime):
    """
    根据策略表现和市场状态分配资金
    """
    total_capital = 1.0
    allocations = {}

    # 按表现排序
    sorted_strategies = sorted(
        active_strategies,
        key=lambda s: s.get("live_production", {}).get("sharpe", 0),
        reverse=True
    )

    # Regime 影响
    regime_multipliers = {
        "TRENDING": {"trend_following": 1.2, "mean_reversion": 0.8},
        "RANGE": {"trend_following": 0.7, "mean_reversion": 1.3},
        "PANIC": {"trend_following": 0.5, "emergency": 1.0},
    }

    # 分配逻辑
    remaining = total_capital
    for i, strategy in enumerate(sorted_strategies):
        base_weight = 1.0 / (i + 1)  # 排名越高权重越高

        # Regime调整
        regime_type = strategy.get("type", "trend_following")
        regime_mult = regime_multipliers.get(market_regime, {}).get(regime_type, 1.0)

        weight = base_weight * regime_mult

        # 测试版本上限
        if strategy["status"] == "testing":
            weight = min(weight, ALLOCATION_STRATEGY["max_testing"])

        allocations[strategy["id"]] = remaining * weight
        remaining -= allocations[strategy["id"]]

    return allocations
```

## 6.2 动态再平衡

```python
def rebalance_strategies(current_allocations, performance_metrics):
    """
    根据近期表现重新平衡策略资金
    """
    changes = {}

    for strategy_id, current_pct in current_allocations.items():
        metrics = performance_metrics[strategy_id]
        sharpe = metrics.get("sharpe", 0)
        dd = metrics.get("max_drawdown", 1)

        # 表现好的增加资金
        if sharpe > 1.5:
            new_pct = current_pct * 1.2
        # 表现差的减少资金
        elif sharpe < 0.8:
            new_pct = current_pct * 0.7
        # 回撤超标的紧急降级
        elif dd > 0.08:
            new_pct = current_pct * 0.5
            changes[strategy_id] = {
                "action": "EMERGENCY_REDUCE",
                "reason": "drawdown_breach"
            }
        else:
            continue

        changes[strategy_id] = {
            "from": current_pct,
            "to": min(new_pct, ALLOCATION_STRATEGY["max_testing"]),
        }

    return changes
```

## 7.2 回滚流程

```python
def rollback_strategy(strategy_id, reason):
    """
    回滚到上一个稳定版本
    """
    current = get_strategy(strategy_id)
    history = get_strategy_history(strategy_id)

    # 找到最近的生产版本
    stable_version = None
    for version in reversed(history):
        if version["status"] == "production":
            stable_version = version
            break

    if not stable_version:
        logger.error(f"No stable version to rollback to for {strategy_id}")
        return False

    # 执行回滚
    logger.warning(f"Rolling back {strategy_id}: {reason}")
    logger.warning(f"  From: {current['version']}")
    logger.warning(f"  To: {stable_version['version']}")

    # 1. 减少当前策略资金
    adjust_allocation(strategy_id, reduce_to=0.20)

    # 2. 增加稳定版本资金
    adjust_allocation(stable_version["id"], increase_to=0.70)

    # 3. 记录事件
    record_event({
        "type": "ROLLBACK",
        "strategy": strategy_id,
        "from_version": current["version"],
        "to_version": stable_version["version"],
        "reason": reason,
        "timestamp": time.time(),
    })

    return True
```

# 🧪 八、A/B 测试框架

## 8.1 测试设计

```python
class ABTest:
    def __init__(self, name, strategy_a, strategy_b, traffic_split=0.5):
        self.name = name
        self.strategy_a = strategy_a
        self.strategy_b = strategy_b
        self.traffic_split = traffic_split

        self.results_a = []
        self.results_b = []

    def assign_strategy(self, signal):
        """
        随机分配策略
        """
        if random.random() < self.traffic_split:
            return self.strategy_a
        else:
            return self.strategy_b

    def record_result(self, strategy_id, outcome):
        if strategy_id == self.strategy_a.id:
            self.results_a.append(outcome)
        else:
            self.results_b.append(outcome)

    def compute_significance(self):
        """
        计算统计显著性
        """
        if len(self.results_a) < 30 or len(self.results_b) < 30:
            return None, "insufficient_samples"

        sharpe_a = compute_sharpe([r["returns"] for r in self.results_a])
        sharpe_b = compute_sharpe([r["returns"] for r in self.results_b])

        # t-test
        t_stat, p_value = scipy.stats.ttest_ind(
            [r["returns"] for r in self.results_a],
            [r["returns"] for r in self.results_b]
        )

        is_significant = p_value < 0.05

        return {
            "sharpe_a": sharpe_a,
            "sharpe_b": sharpe_b,
            "improvement": (sharpe_b - sharpe_a) / sharpe_a,
            "p_value": p_value,
            "is_significant": is_significant,
            "winner": "B" if sharpe_b > sharpe_a and is_significant else "A" if sharpe_a > sharpe_b else None
        }
```

def decide_ab_upgrade(ab_test):
    result = ab_test.compute_significance()

    if result is None:
        return {"action": "CONTINUE", "reason": "收集更多数据"}

    if result["winner"] == "B" and result["improvement"] > 0.2:
        return {
            "action": "PROMOTE_B",
            "confidence": result["p_value"],
            "reason": f"B显著优于A，提升{result['improvement']:.1%}"
        }
    elif result["winner"] == "A":
        return {
            "action": "DEMOTE_B",
            "confidence": result["p_value"],
            "reason": "A优于B，放弃B"
        }
    else:
        return {
            "action": "CONTINUE",
            "reason": "差异不显著"

## 9.2 存储策略

```python
class StrategyRegistry:
    def __init__(self, storage_path):
        self.storage_path = storage_path
        self.strategies = {}
        self.load_all()

    def save_strategy(self, strategy):
        path = f"{self.storage_path}/{strategy.id}.json"
        with open(path, "w") as f:
            json.dump(strategy.to_dict(), f, indent=2)
        self.strategies[strategy.id] = strategy

    def get_strategy(self, strategy_id):
        return self.strategies.get(strategy_id)

    def get_active_strategies(self):
        return [s for s in self.strategies.values()
                if s.status in ["production", "testing", "shadow"]]

    def get_latest_production(self):
        production = [s for s in self.strategies.values()
                      if s.status == "production"]
        return max(production, key=lambda s: s.created_at) if production else None
```

# ⚠️ 十一、常见错误

| 错误 | 后果 | 正确做法 |
|------|------|----------|
| 直接切新策略 | 未知风险全压 | 先影子运行 |
| 不保留旧版本 | 回滚困难 | 归档保留 |
| 测试期太短 | 样本不足 | 最少4周 |
| 不监控信号一致性 | 策略可能已失效 | 影子对比 |
| 升级太激进 | 资金全押新版本 | 金丝雀分配 |
