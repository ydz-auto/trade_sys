# TradeAgent Feedback Loop（反馈闭环系统）设计文档

---

# 🧠 一、模块定位

## 1.1 核心问题

```
现在的系统：

Feature → Factor → Regime → Risk → Decision → Position → Execution
                                                              ↓
                                                          PnL追踪
                                                              ↓
                                                            结束

问题：PnL 结果没有反向影响决策
```

## 1.2 反馈闭环的本质

```
Execution → PnL → 分析 → 调整 → 再决策
                ↑
              Risk
```

## 1.3 在系统中的位置

```
Factor → Regime → Risk → Decision → Position → Execution
                                          ↑              ↓
                                      调整          PnL追踪
                                          ↑              ↓
                                        Risk ←─── 反馈分析
```

## 3.2 盈亏反馈规则

```python
class PnLFeedback:
    def __init__(self):
        self.consecutive_losses = 0
        self.consecutive_wins = 0
        self.total_pnl = 0
        self.pnl_history = []

    def record_trade(self, pnl):
        self.pnl_history.append(pnl)
        self.total_pnl += pnl

        if pnl > 0:
            self.consecutive_wins += 1
            self.consecutive_losses = 0
        else:
            self.consecutive_losses += 1
            self.consecutive_wins = 0

    def should_reduce_exposure(self):
        """连续亏损后应该降低暴露"""
        if self.consecutive_losses >= 3:
            return True, "consecutive_losses_3"
        if self.consecutive_losses >= 5:
            return True, "consecutive_losses_5"
        return False, None

    def get_exposure_multiplier(self):
        """根据近期表现调整暴露倍数"""
        if self.consecutive_losses >= 5:
            return 0.3  # 降到30%
        elif self.consecutive_losses >= 3:
            return 0.5  # 降到50%
        elif self.consecutive_losses >= 2:
            return 0.7  # 降到70%

        recent_pnl = sum(self.pnl_history[-5:]) if len(self.pnl_history) >= 5 else sum(self.pnl_history)
        if recent_pnl > 0.05:
            return 1.2  # 表现好可以加仓

        return 1.0
```

## 3.3 动态仓位调整

```python
def adjust_position_for_pnl(position, pnl_feedback):
    base_position = position

    multiplier = pnl_feedback.get_exposure_multiplier()

    if pnl_feedback.should_reduce_exposure()[0]:
        reason = pnl_feedback.should_reduce_exposure()[1]
        logger.warning(f"PnL feedback: reducing position to {multiplier}, reason: {reason}")

    return position * multiplier
```

# ⚠️ 四、Risk Feedback（风险反馈）

## 4.1 风险反馈机制

```python
class RiskFeedback:
    def __init__(self):
        self.recent_drawdowns = []
        self.risk_adjusted_returns = []

    def update_after_trade(self, trade_pnl, trade_risk):
        """
        记录交易的风险调整后收益
        """
        risk_adjusted = trade_pnl / trade_risk if trade_risk > 0 else 0
        self.risk_adjusted_returns.append(risk_adjusted)

        if trade_pnl < 0:
            self.recent_drawdowns.append(abs(trade_pnl))

    def is_risk_being_rewarded(self):
        """
        检查"高风险是否有高回报"
        如果低风险交易反而赚钱多，需要调整
        """
        if len(self.risk_adjusted_returns) < 10:
            return True

        recent = self.risk_adjusted_returns[-10:]
        high_risk_trades = [r for r in recent if r > 0.1]
        low_risk_trades = [r for r in recent if r <= 0.1]

        if not high_risk_trades or not low_risk_trades:
            return True

        avg_high = sum(high_risk_trades) / len(high_risk_trades)
        avg_low = sum(low_risk_trades) / len(low_risk_trades)

        # 如果低风险交易收益更高，说明风险模型有问题
        if avg_low > avg_high:
            return False

        return True

    def get_risk_adjustment(self):
        """
        返回风险调整系数
        """
        if not self.is_risk_being_rewarded():
            return 0.8, "risk_not_being_rewarded"

        recent_dd = sum(self.recent_drawdowns[-3:]) if self.recent_drawdowns else 0
        if recent_dd > 0.10:
            return 0.7, "high_recent_drawdown"

        return 1.0, "normal"
```

## 4.2 风险与决策联动

```python
def adjust_risk_for_feedback(base_risk_index, risk_feedback):
    adjustment, reason = risk_feedback.get_risk_adjustment()

    adjusted_risk = base_risk_index * (2 - adjustment)

    if reason != "normal":
        logger.warning(f"Risk adjusted: {base_risk_index} -> {adjusted_risk}, reason: {reason}")

    return min(adjusted_risk, 100)
```

## 5.2 信号反馈计算

```python
class SignalFeedback:
    def __init__(self):
        self.signal_history = []

    def record_signal_outcome(self, signal, quality_score, outcome):
        """
        记录信号的结果
        outcome: "profit" / "loss" / "breakeven"
        """
        self.signal_history.append({
            "signal": signal,
            "quality": quality_score,
            "outcome": outcome,
            "timestamp": time.time()
        })

    def get_quality_accuracy(self):
        """
        计算每个质量等级的准确性
        """
        quality_results = {}

        for record in self.signal_history:
            q = record["quality"]
            grade = self.score_to_grade(q)

            if grade not in quality_results:
                quality_results[grade] = {"correct": 0, "total": 0}

            quality_results[grade]["total"] += 1
            if record["outcome"] in ["profit", "breakeven"]:
                quality_results[grade]["correct"] += 1

        for grade, data in quality_results.items():
            data["accuracy"] = data["correct"] / data["total"] if data["total"] > 0 else 0

        return quality_results

    def should_adjust_threshold(self):
        """
        如果B级信号准确性低于40%，提高执行门槛
        """
        quality_acc = self.get_quality_accuracy()

        if "B" in quality_acc and quality_acc["B"]["accuracy"] < 0.40:
            return True, "B_accuracy_too_low"

        if "B+" in quality_acc and quality_acc["B+"]["accuracy"] < 0.50:
            return True, "B+_accuracy_too_low"

        return False, None
```

## 5.3 动态质量阈值

```python
class DynamicQualityThreshold:
    def __init__(self):
        self.base_threshold = 0.65
        self.current_threshold = 0.65
        self.signal_feedback = SignalFeedback()

    def update_threshold(self):
        """
        根据信号反馈动态调整阈值
        """
        should_adjust, reason = self.signal_feedback.should_adjust_threshold()

        if should_adjust:
            old = self.current_threshold
            self.current_threshold = min(self.current_threshold + 0.05, 0.80)
            logger.warning(f"Quality threshold raised: {old:.2f} -> {self.current_threshold:.2f}, reason: {reason}")

    def get_threshold(self):
        return self.current_threshold
```

# 🔄 六、Regime Feedback（状态反馈）

## 6.1 Regime 准确性追踪

```python
class RegimeFeedback:
    def __init__(self):
        self.regime_history = []

    def record_regime_outcome(self, regime, price_change_1h, price_change_4h):
        """
        记录Regime判定后的实际价格变化
        """
        self.regime_history.append({
            "regime": regime,
            "price_change_1h": price_change_1h,
            "price_change_4h": price_change_4h,
            "timestamp": time.time()
        })

    def get_regime_accuracy(self):
        """
        评估每个Regime的预测准确性
        """
        regime_results = {}

        for record in self.regime_history:
            r = record["regime"]

            if r not in regime_results:
                regime_results[r] = {"correct": 0, "total": 0}

            # 简化的准确性判断
            # 趋势市场应该有一定方向性
            if r == "TRENDING":
                if abs(record["price_change_4h"]) > 0.02:
                    regime_results[r]["correct"] += 1
            elif r == "RANGE":
                if abs(record["price_change_4h"]) < 0.02:
                    regime_results[r]["correct"] += 1
            elif r == "PANIC":
                if record["price_change_4h"] < -0.02:
                    regime_results[r]["correct"] += 1

            regime_results[r]["total"] += 1

        return regime_results
```

## 6.2 Regime 权重调整

```python
def adjust_regime_weights(regime_feedback):
    """
    根据历史准确性调整不同Regime下的策略权重
    """
    accuracy = regime_feedback.get_regime_accuracy()

    adjustments = {}

    for regime, data in accuracy.items():
        if data["total"] < 10:
            adjustments[regime] = 1.0
            continue

        acc = data["correct"] / data["total"]

        if acc < 0.4:
            adjustments[regime] = 0.5  # 准确性太低，减少依赖
        elif acc > 0.7:
            adjustments[regime] = 1.2  # 准确性高，增加依赖

    return adjustments
```

# 🔗 七、完整反馈闭环流程

## 7.1 交易后反馈流程

```python
def post_trade_feedback(trade, position_engine):
    """
    交易执行后的完整反馈流程
    """
    results = {
        "pnl": trade["pnl"],
        "risk": trade["risk"],
        "signal": trade["signal"],
        "regime": trade["regime"]
    }

    # 1. PnL反馈
    pnl_feedback.record_trade(results["pnl"])
    exposure_mult = pnl_feedback.get_exposure_multiplier()

    # 2. Risk反馈
    risk_feedback.update_after_trade(results["pnl"], results["risk"])
    risk_adjustment, risk_reason = risk_feedback.get_risk_adjustment()

    # 3. Signal反馈
    signal_feedback.record_signal_outcome(
        results["signal"],
        results["signal"]["quality"],
        classify_outcome(results["pnl"])
    )

    # 4. Regime反馈
    regime_feedback.record_regime_outcome(
        results["regime"],
        results["price_change_1h"],
        results["price_change_4h"]
    )

    # 生成综合反馈报告
    feedback_report = {
        "exposure_multiplier": exposure_mult,
        "risk_adjustment": risk_adjustment,
        "risk_reason": risk_reason,
        "should_alert": pnl_feedback.consecutive_losses >= 3,
        "alert_message": generate_alert_message(pnl_feedback, risk_feedback)
    }

    return feedback_report
```

## 7.2 反馈应用到下次决策

```python
def apply_feedback_to_decision(base_decision, feedback_report):
    """
    将反馈闭环的结果应用到下次决策
    """
    decision = base_decision.copy()

    # 1. 调整仓位
    decision["position_multiplier"] = feedback_report["exposure_multiplier"]

    # 2. 调整风险承受
    decision["risk_adjustment"] = feedback_report["risk_adjustment"]

    # 3. 如果连续亏损，降低仓位
    if feedback_report["should_alert"]:
        decision["action"] = "REDUCE"
        decision["reduce_ratio"] = 0.5
        decision["alert"] = feedback_report["alert_message"]

    return decision
```

```python
def handle_alert(alert_type, alert_data):
    """
    处理不同级别的告警
    """
    alert_config = ALERT_CONDITIONS.get(alert_type)

    if not alert_config:
        return

    severity = alert_config["severity"]
    action = alert_config["action"]

    # 发送通知
    notify(f"[{severity}] {alert_type}: {action}")

    # 执行相应动作
    if "reduce_exposure" in action:
        ratio = float(action.split("_")[-1].replace("%", "")) / 100
        position_engine.reduce_exposure(ratio)

    if action == "stop_trading":
        control_service.pause_trading(reason=alert_type)

    if action == "emergency_stop":
        execution_service.close_all_positions()
        control_service.emergency_stop(reason=alert_type)
```

# 🔄 十、反馈周期

| 反馈类型 | 更新频率 | 影响 |
|----------|----------|------|
| PnL反馈 | 每次交易后 | 立即影响仓位 |
| Risk反馈 | 每日 | 调整风险参数 |
| Signal反馈 | 每日/每周 | 调整质量阈值 |
| Regime反馈 | 每周 | 调整策略权重 |
