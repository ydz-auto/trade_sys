## 3.3 价格数据采集（主备切换）

```python
from dataclasses import dataclass
from typing import Dict, List, Optional, Tuple
from datetime import datetime

@dataclass
class ExchangePrice:
    """单个交易所的价格数据"""
    exchange: str              # "binance", "okx", "coinbase"
    symbol: str               # "BTC/USDT", "ETH/USDT"
    price: float              # 最新成交价
    bid: float                # 买一价
    ask: float                # 卖一价
    spread: float             # 买卖价差 (ask - bid)
    spread_percent: float    # 价差百分比
    volume_24h: float         # 24小时成交量
    timestamp: datetime       # 数据时间戳

@dataclass
class MultiExchangePrices:
    """多交易所价格（不融合！分别保存）"""
    symbol: str
    prices: Dict[str, ExchangePrice]  # 各交易所原始价格

    @property
    def max_price(self) -> Tuple[str, float]:
        """最高价交易所"""
        return max(self.prices.items(), key=lambda x: x[1].price)

    @property
    def min_price(self) -> Tuple[str, float]:
        """最低价交易所"""
        return min(self.prices.items(), key=lambda x: x[1].price)

    @property
    def arbitrage_opportunity(self) -> Optional[Dict]:
        """套利机会检测"""
        min_exchange, min_data = self.min_price
        max_exchange, max_data = self.max_price
        diff = max_data.price - min_data.price
        diff_percent = diff / min_data.price * 100

        if diff_percent > 0.01:  # 超过0.01%视为套利机会
            return {
                "buy_exchange": min_exchange,
                "sell_exchange": max_exchange,
                "buy_price": min_data.price,
                "sell_price": max_data.price,
                "diff": diff,
                "diff_percent": diff_percent
            }
        return None


class PriceCollector:
    """价格采集器（主备切换，不融合）"""

    def __init__(self, config: Dict):
        self.config = config
        self.ccxt = ccxt.pro.binance()  # 主交易所
        self.fallback_exchanges = {
            "okx": ccxt.pro.okx(),
            "hyperliquid": ccxt.pro.hyperliquid(),
            "coinbase": ccxt.pro.coinbase(),
        }
        self.consecutive_failures = {name: 0 for name in self.fallback_exchanges}
        self.current_active = "binance"

    async def collect(self, symbol: str) -> MultiExchangePrices:
        """并行采集所有交易所价格（不融合！）"""
        tasks = [
            self._fetch_price("binance", symbol),
            self._fetch_price("okx", symbol),
            self._fetch_price("hyperliquid", symbol),
            self._fetch_price("coinbase", symbol),
        ]

        results = await asyncio.gather(*tasks, return_exceptions=True)

        prices = {}
        for exchange, result in zip(["binance", "okx", "coinbase"], results):
            if isinstance(result, ExchangePrice):
                prices[exchange] = result
                self.consecutive_failures[exchange] = 0
            else:
                logger.warning(f"{exchange}采集失败: {result}")
                self.consecutive_failures[exchange] += 1

                # 检查是否需要切换
                if self.consecutive_failures[exchange] >= 3:
                    await self._switch_to_fallback(exchange)

        return MultiExchangePrices(symbol=symbol, prices=prices)

    async def _fetch_price(self, exchange: str, symbol: str) -> ExchangePrice:
        """从单个交易所获取价格"""
        if exchange == "binance":
            ticker = await self.ccxt.fetch_ticker(symbol)
        elif exchange == "okx":
            ticker = await self.fallback_exchanges["okx"].fetch_ticker(symbol)
        elif exchange == "hyperliquid":
            ticker = await self.fallback_exchanges["hyperliquid"].fetch_ticker(symbol)
        elif exchange == "coinbase":
            ticker = await self.fallback_exchanges["coinbase"].fetch_ticker(symbol)

        return ExchangePrice(
            exchange=exchange,
            symbol=symbol,
            price=ticker["last"],
            bid=ticker["bid"],
            ask=ticker["ask"],
            spread=ticker["ask"] - ticker["bid"],
            spread_percent=(ticker["ask"] - ticker["bid"]) / ticker["ask"] * 100,
            volume_24h=ticker["baseVolume"],
            timestamp=datetime.now()
        )

    async def _switch_to_fallback(self, failed_exchange: str):
        """切换到备用交易所"""
        logger.warning(f"{failed_exchange}连续失败，切换到备用交易所")

        for name, exchange in self.fallback_exchanges.items():
            if name != failed_exchange and self.consecutive_failures[name] < 3:
                self.current_active = name
                logger.info(f"已切换到 {name}")
                return

        raise AllExchangesFailedError("所有交易所均不可用")

    async def collect_for_trading(self, symbol: str, target_exchange: str) -> ExchangePrice:
        """采集特定交易所价格用于交易"""
        prices = await self.collect(symbol)

        if target_exchange in prices.prices:
            return prices.prices[target_exchange]

        # 目标交易所失败，切换到可用的
        for name, price in prices.prices.items():
            if self.consecutive_failures[name] < 3:
                logger.warning(f"{target_exchange}不可用，切换到{name}")
                return price

        raise AllExchangesFailedError(f"无法获取{symbol}价格")
```

## 3.4 数据融合代码示例（非价格数据）

```python
from dataclasses import dataclass, field
from typing import List, Dict, Any, Optional
from enum import Enum
import asyncio

class DataSourceType(Enum):
    REST_API = "rest_api"
    LLM_SCRAPER = "llm_scraper"
    NPM_PACKAGE = "npm_package"
    RSS = "rss"
    WEBSOCKET = "websocket"
    WEB_SCRAPER = "web_scraper"

@dataclass
class SourceResult:
    source_name: str
    data: Any
    timestamp: float
    confidence: float
    error: Optional[str] = None

@dataclass
class FusedResult:
    data: Any
    sources_used: List[str]
    confidence: float
    total_weight: float
    validation_notes: List[str] = field(default_factory=list)

class MultiSourceCollector:
    """多数据源采集器（非价格数据专用）"""

    def __init__(self, config: Dict):
        self.config = config
        self.collectors = self._init_collectors()

    def _init_collectors(self):
        return {
            "rest_api": RESTCollector(),
            "llm_scraper": FireCrawlCollector(),
            "npm_package": NpmPackageCollector(),
            "rss": RSSCollector(),
            "websocket": WebSocketCollector(),
            "web_scraper": BeautifulSoupCollector(),
        }

    async def collect(self, data_type: str) -> FusedResult:
        """多数据源采集融合（ETF/新闻/宏观数据用）"""
        config = self.config[data_type]
        sources = config["sources"]

        # 1. 并行采集所有数据源
        tasks = []
        for source in sources:
            collector = self.collectors[source["type"]]
            tasks.append(self._collect_from_source(collector, source))

        results = await asyncio.gather(*tasks, return_exceptions=True)

        # 2. 过滤成功的源
        valid_results = [
            SourceResult(
                source_name=r["source"],
                data=r["data"],
                timestamp=r["timestamp"],
                confidence=source["weight"]
            )
            for r, source in zip(results, sources)
            if isinstance(r, dict) and r.get("data") is not None
        ]

        if not valid_results:
            raise Exception(f"所有{data_type}数据源均失败")

        # 3. 交叉验证
        validated = self._cross_validate(valid_results, config)

        # 4. 加权融合
        fused = self._fuse(validated, config)

        return fused

    async def _collect_from_source(self, collector, source_config):
        """从单个数据源采集"""
        try:
            data = await collector.fetch(source_config)
            return {
                "source": source_config["name"],
                "data": data,
                "timestamp": time.time(),
                "error": None
            }
        except Exception as e:
            return {
                "source": source_config["name"],
                "data": None,
                "timestamp": time.time(),
                "error": str(e)
            }

    def _cross_validate(self, results: List[SourceResult], config: Dict) -> List[SourceResult]:
        """交叉验证多源数据一致性"""
        fusion_config = config["fusion"]
        threshold = fusion_config.get("diff_threshold_percent", 5) / 100

        if len(results) < 2:
            return results

        # 计算平均值
        values = [r.data for r in results if isinstance(r.data, (int, float))]
        if not values:
            return results

        avg = sum(values) / len(values)

        # 检查偏差
        validated = []
        for r in results:
            if isinstance(r.data, (int, float)):
                deviation = abs(r.data - avg) / avg
                if deviation > threshold:
                    r.confidence *= 0.5
                    r.validation_notes.append(f"偏差{int(deviation*100)}%")

            if r.confidence < 0.3:
                r.validation_notes.append("置信度过低")
            else:
                validated.append(r)

        return validated

    def _fuse(self, results: List[SourceResult], config: Dict) -> FusedResult:
        """加权融合多源数据"""
        fusion_method = config["fusion"]["method"]

        if fusion_method == "weighted_average":
            total_weight = sum(r.confidence for r in results)

            numeric_results = [r for r in results if isinstance(r.data, (int, float))]
            if numeric_results:
                fused_data = sum(r.data * r.confidence for r in numeric_results) / total_weight
            else:
                fused_data = max(results, key=lambda x: x.confidence).data

            return FusedResult(
                data=fused_data,
                sources_used=[r.source_name for r in results],
                confidence=total_weight / len(results),
                total_weight=total_weight,
                validation_notes=[note for r in results for note in r.validation_notes]
            )

        elif fusion_method == "confidence_weighted":
            best = max(results, key=lambda x: x.confidence)
            return FusedResult(
                data=best.data,
                sources_used=[best.source_name],
                confidence=best.confidence,
                total_weight=best.confidence
            )
```

```python
class RESTCollector:

    async def fetch(self, config: Dict) -> Any:
        url = config.get("url")
        headers = config.get("headers", {})
        
        async with aiohttp.ClientSession() as session:

    def parse(self, data: Dict, config: Dict) -> Any:

        parse_path = config.get("parse_path", "")
        if parse_path:
            for key in parse_path.split("."):
                data = data[key]
        return data
```

## 4.2 LLM爬虫

```python
class FireCrawlCollector:
    """FireCrawl LLM爬虫采集器"""
    
    def __init__(self, api_key: str = None):
        self.client = FireCrawl(api_key=api_key or os.getenv("FIRECRAWL_API_KEY"))
    
    async def fetch(self, config: Dict) -> Any:
        url = config.get("url")
        formats = config.get("formats", ["markdown"])
        
        result = self.client.scrape_url(url, formats=formats)
        
        # 使用LLM提取结构化数据
        if config.get("llm_extract"):
            return await self._llm_extract(result.markdown, config)
        
        return result.markdown
    
    async def _llm_extract(self, content: str, config: Dict) -> Dict:
        """使用LLM从内容中提取结构化数据"""
        prompt = config.get("llm_prompt", "提取JSON数据")
        
        response = openai.chat.completions.create(
            model="gpt-4o-mini",
            messages=[
                {"role": "system", "content": "你是一个数据提取专家"},
                {"role": "user", "content": f"{prompt}\n\n内容：\n{content[:8000]}"}
            ],
            response_format={"type": "json_object"}
        )
        
        return json.loads(response.choices[0].message.content)
```

```python
class NpmPackageCollector:

        result = await asyncio.create_subprocess_exec(
            "node", "-e", cmd,
            stdout=asyncio.subprocess.PIPE,
            stderr=asyncio.subprocess.PIPE
        )
        
        stdout, _ = await result.communicate()
        return json.loads(stdout.decode())
```

```python
class RSSCollector:
    """RSS订阅采集器"""
    
    async def fetch(self, config: Dict) -> List[Dict]:
        import feedparser
        
        feeds = config.get("feeds", [config.get("url")])
        results = []
        
        for feed_url in feeds:
            feed = await asyncio.to_thread(feedparser.parse, feed_url)
            
            for entry in feed.entries[:config.get("max_items", 20)]:
                results.append({
                    "title": entry.get("title"),
                    "link": entry.get("link"),
                    "published": entry.get("published"),
                    "summary": entry.get("summary", "")[:500],
                    "source": feed_url
                })
        
        return results
```

## 4.5 网页爬虫

```python
class BeautifulSoupCollector:
    """传统网页爬虫采集器"""
    
    async def fetch(self, config: Dict) -> Any:
        import aiohttp
        from bs4 import BeautifulSoup
        
        url = config.get("url")
        
        async with aiohttp.ClientSession() as session:
            async with session.get(url) as resp:
                html = await resp.text()
        
        soup = BeautifulSoup(html, "html.parser")
        
        # 根据选择器提取数据
        selector = config.get("selector")
        if selector:
            elements = soup.select(selector)
            return [elem.get_text(strip=True) for elem in elements]
        
        return soup
```
