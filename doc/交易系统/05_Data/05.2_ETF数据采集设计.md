# 📊 为什么需要多源融合

| ETF数据特点 | 说明 |
|------------|------|
| **来源单一** | 不同源可能覆盖不同ETF |
| **更新延迟** | 各源更新时间可能不同 |
| **格式差异** | 需要统一解析 |
| **准确性重要** | ETF资金流影响交易决策 |

# 🏗️ 架构设计

## 1. 数据模型

```python
from dataclasses import dataclass
from datetime import datetime
from typing import Dict, List, Optional

@dataclass
class ETFFlow:
    """单只ETF资金流数据"""
    ticker: str              # "IBIT", "FBTC", "ARKB"
    flow: float             # 净流入（百万美元）
    flow_type: str         # "inflow" / "outflow"
    cumulative_flow: float  # 累计净流入
    aum: float            # 资产管理规模
    volume: float         # 成交量
    timestamp: datetime    # 数据时间戳
    source: str          # 数据来源

@dataclass
class ETFFlowData:
    """融合后的ETF数据"""
    date: str
    total_flow: float           # 总净流入
    total_flow_percent: float    # 总净流入百分比
    etf_flows: List[ETFFlow]    # 各ETF资金流
    sources_used: List[str]      # 使用的数据源
    confidence: float            # 置信度
    validation_notes: List[str] # 验证备注
    timestamp: datetime         # 融合时间

@dataclass
class SourceResult:
    """单数据源采集结果"""
    source: str
    data: ETFFlowData
    weight: float
    timestamp: float
    error: Optional[str] = None
```

## 2. 采集器设计

```python
class ETFCollector:
    """ETF数据采集器（多源融合）"""

    def __init__(self):
        self.sources = {
            "farside": FarsideLLMScraper(),
            "sosovalue": SoSoValueNpmScraper(),
            "coinglass": CoinGlassAPIScraper(),
        }
        self.weights = {
            "farside": 0.45,
            "sosovalue": 0.30,
            "coinglass": 0.25,
        }

    async def collect(self) -> ETFFlowData:
        """多源并行采集 + 融合"""
        # 1. 并行采集所有数据源
        tasks = [
            self._collect_source(name, scraper)
            for name, scraper in self.sources.items()
        ]
        results = await asyncio.gather(*tasks, return_exceptions=True)

        # 2. 过滤成功的源
        valid_results = [
            SourceResult(
                source=name,
                data=r,
                weight=self.weights[name],
                timestamp=time.time()
            )
            for name, r in zip(self.sources.keys(), results)
            if isinstance(r, ETFFlowData)
        ]

        if not valid_results:
            raise AllSourcesFailedError("所有ETF数据源均失败")

        # 3. 交叉验证
        validated = self._cross_validate(valid_results)

        # 4. 加权融合
        fused = self._fuse(validated)

        return fused


class FarsideLLMScraper:
    """Farside LLM爬虫（主数据源）"""

    def __init__(self):
        self.firecrawl = FireCrawl(api_key=os.getenv("FIRECRAWL_API_KEY"))
        self.url = "https://farside.co.uk/bitcoin-etf-flow-all-data/"

    async def fetch(self) -> ETFFlowData:
        """使用FireCrawl爬取并用LLM解析"""
        result = self.firecrawl.scrape_url(
            self.url,
            formats=["markdown"]
        )

        # LLM提取结构化数据
        flow_data = await self._llm_extract(result.markdown)

        return ETFFlowData(
            date=flow_data["date"],
            total_flow=flow_data["total_flow"],
            total_flow_percent=flow_data["total_flow_percent"],
            etf_flows=[
                ETFFlow(ticker=t, flow=f, ...)
                for t, f in flow_data["etf_flows"].items()
            ],
            sources_used=["farside"],
            confidence=0.95,
            validation_notes=[],
            timestamp=datetime.now()
        )

    async def _llm_extract(self, content: str) -> Dict:
        """使用LLM从内容提取ETF数据"""
        prompt = """
        从以下比特币ETF资金流页面内容中提取今日数据。

        请提取：
        1. 日期
        2. 各ETF净流入（IBIT, FBTC, ARKB等）
        3. 总净流入

        输出JSON格式，包含date, total_flow, etf_flows字段。
        """

        response = openai.chat.completions.create(
            model="gpt-4o-mini",
            messages=[
                {"role": "system", "content": "你是一个专业的金融数据提取专家"},
                {"role": "user", "content": f"{prompt}\n\n内容：\n{content[:6000]}"}
            ],
            response_format={"type": "json_object"}
        )

        return json.loads(response.choices[0].message.content)


class SoSoValueNpmScraper:
    """SoSoValue npm包采集器"""

    async def fetch(self) -> ETFFlowData:
        """使用bitcoin-etf-data npm包"""
        cmd = """
        const { fetchEtfData } = require('bitcoin-etf-data');
        fetchEtfData().then(data => console.log(JSON.stringify(data[0])));
        """

        result = await asyncio.create_subprocess_exec(
            "node", "-e", cmd,
            stdout=asyncio.subprocess.PIPE,
            stderr=asyncio.subprocess.PIPE
        )

        stdout, _ = await result.communicate()
        data = json.loads(stdout.decode())

        return ETFFlowData(
            date=data["date"],
            total_flow=data["total_flow"],
            total_flow_percent=0,
            etf_flows=[],
            sources_used=["sosovalue_npm"],
            confidence=0.80,
            validation_notes=[],
            timestamp=datetime.now()
        )


class CoinGlassAPIScraper:
    """CoinGlass API采集器"""

    async def fetch(self) -> ETFFlowData:
        """CoinGlass REST API"""
        async with aiohttp.ClientSession() as session:
            async with session.get(
                "https://open-api-v4.coinglass.com/api/etf/bitcoin/flow-history",
                headers={"accept": "application/json"}
            ) as resp:
                data = await resp.json()

        latest = data["data"][0]

        return ETFFlowData(
            date=timestamp_to_date(latest["timestamp"]),
            total_flow=latest["total_flow_usd"] / 1_000_000,
            total_flow_percent=latest["total_flow_percent"],
            etf_flows=[
                ETFFlow(
                    ticker=e["etf_ticker"],
                    flow=e["flow_usd"] / 1_000_000,
                    ...
                )
                for e in latest["etf_flows"]
            ],
            sources_used=["coinglass"],
            confidence=0.90,
            validation_notes=[],
            timestamp=datetime.now()
        )
```

## 3. 交叉验证

```python
class ETFCrossValidator:
    """ETF数据交叉验证"""

    def __init__(self, diff_threshold_percent: float = 10.0):
        self.diff_threshold = diff_threshold_percent / 100

    def validate(self, results: List[SourceResult]) -> List[SourceResult]:
        """交叉验证多源数据"""
        if len(results) < 2:
            return results

        # 1. 计算总流入平均值
        total_flows = [r.data.total_flow for r in results]
        avg_flow = sum(total_flows) / len(total_flows)

        # 2. 检查每个源的偏差
        validated = []
        for r in results:
            deviation = abs(r.data.total_flow - avg_flow) / avg_flow

            if deviation > self.diff_threshold:
                # 偏差过大，降低置信度
                r.data.confidence *= 0.5
                r.data.validation_notes.append(
                    f"偏差{int(deviation*100)}%，超过{int(self.diff_threshold*100)}%阈值"
                )

            validated.append(r)

        # 3. 过滤置信度过低的源
        return [r for r in validated if r.data.confidence >= 0.3]
```

## 4. 加权融合

```python
def fuse(self, results: List[SourceResult]) -> ETFFlowData:
    """加权融合多源ETF数据"""

    # 1. 计算总权重
    total_weight = sum(r.weight for r in results)
    total_weight = total_weight if total_weight > 0 else 1.0

    # 2. 加权平均总流入
    fused_total_flow = sum(
        r.data.total_flow * r.weight
        for r in results
    ) / total_weight

    # 3. 融合ETF详情（按权重分配）
    fused_etf_flows = self._fuse_etf_details(results, total_weight)

    # 4. 计算置信度
    avg_confidence = sum(r.data.confidence for r in results) / len(results)

    # 5. 汇总验证备注
    all_notes = []
    for r in results:
        all_notes.extend(r.data.validation_notes)

    return ETFFlowData(
        date=results[0].data.date,
        total_flow=round(fused_total_flow, 2),
        total_flow_percent=results[0].data.total_flow_percent,
        etf_flows=fused_etf_flows,
        sources_used=[r.source for r in results],
        confidence=avg_confidence,
        validation_notes=all_notes,
        timestamp=datetime.now()
    )

def _fuse_etf_details(self, results, total_weight):
    """融合各ETF详情"""
    etf_dict = {}

    for r in results:
        for etf in r.data.etf_flows:
            if etf.ticker not in etf_dict:
                etf_dict[etf.ticker] = []
            etf_dict[etf.ticker].append((etf, r.weight))

    fused = []
    for ticker, items in etf_dict.items():
        avg_flow = sum(etf.flow * w for etf, w in items) / total_weight
        fused.append(ETFFlow(
            ticker=ticker,
            flow=round(avg_flow, 2),
            ...
        ))

    return fused
```

## 2. 数据缓存

```python
class ETFCache:
    """ETF数据缓存"""

    def __init__(self):
        self.cache = {}
        self.max_age = 25 * 3600  # 25小时

    def get_cached(self) -> Optional[ETFFlowData]:
        """获取缓存数据"""
        if not self.cache:
            return None

        age = time.time() - self.cache["timestamp"]
        if age > self.max_age:
            return None

        return self.cache["data"]

    def set_cached(self, data: ETFFlowData):
        """设置缓存"""
        self.cache = {
            "data": data,
            "timestamp": time.time()
        }
```
