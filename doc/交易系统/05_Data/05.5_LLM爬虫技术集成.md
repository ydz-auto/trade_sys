# 🧠 一、为什么需要 LLM 爬虫

## 传统爬虫的问题

```
1. 反爬机制：Cloudflare / CAPTCHA / UA检测
2. 动态内容：JavaScript 渲染
3. 非结构化：页面结构不标准
4. 维护成本：网站改版就失效
```

## LLM 爬虫的优势

```
1. 智能解析：理解页面语义，不依赖DOM结构
2. 绕过检测：模拟人类行为
3. 零维护：页面改版不影响
4. 多格式输出：JSON / Markdown / 结构化
```

# 🔧 三、集成方案

## 3.1 FireCrawl（推荐首选）

```python
import firecrawl

class FireCrawlSource:
    """FireCrawl 智能爬虫"""

    def __init__(self, api_key):
        self.client = firecrawl.Firecrawl(api_key=api_key)

    def scrape(self, url):
        """抓取任意网站"""
        result = self.client.scrape_url(
            url,
            formats=['markdown', 'html']
        )
        return {
            "markdown": result.markdown,
            "html": result.html,
            "metadata": result.metadata
        }

    def crawl(self, urls):
        """批量抓取"""
        result = self.client.crawl_url(urls)
        return result
```

## 3.2 Crawl4AI（LLM友好）

```python
from crawl4ai import AsyncWebCrawler

class Crawl4AISource:
    """Crawl4AI 专为LLM设计"""

    async def scrape(self, url):
        async with AsyncWebCrawler() as crawler:
            result = await crawler.araw(url)

            return {
                "content": result.markdown,
                "links": result.links,
                "media": result.media
            }
```

## 3.3 ScrapeGraphAI（复杂页面）

```python
from scrapegraphai import SmartScraper

class ScrapeGraphSource:
    """ScrapeGraphAI 智能解析"""

    def __init__(self, llm_service):
        self.llm = llm_service

    def scrape(self, url, prompt):
        """自然语言定义抓取目标"""
        graph_config = {
            "llm": self.llm,
            "prompt": prompt,  # "提取所有新闻标题、时间和摘要"
            "source": url
        }

        scraper = SmartScraper(**graph_config)
        result = scraper.run()

        return result
```

## 3.4 Crawlee（规模化）

```python
from crawlee.playwright import PlaywrightCrawler

class CrawleeSource:
    """Crawlee 规模化爬虫"""

    def __init__(self):
        self.crawler = PlaywrightCrawler(
            max_request_timeout=30000,
            maxConcurrency=5
        )

    async def crawl(self, urls):
        @self.crawler.router.default_handler
        async def request_handler(request, page):
            await page.wait_for_load_state("networkidle")

            title = await page.title()
            content = await page.content()

            await self.crawler.push_data({
                "url": request.url,
                "title": title,
                "content": content
            })

        await self.crawler.run(urls)
```

## 4.2 优先级策略

```python
SCRAPER_STRATEGY = {
    # 第一选择：传统API（最快）
    "api_available": {
        "priority": 1,
        "scraper": "exchange_api",
        "cost": "free"
    },

    # 第二选择：简单页面
    "static_page": {
        "priority": 2,
        "scraper": "requests + beautifulsoup",
        "cost": "free"
    },

    # 第三选择：FireCrawl
    "dynamic_or_complex": {
        "priority": 3,
        "scraper": "firecrawl",
        "cost": "api_call"
    },

    # 第四选择：Crawl4AI
    "llm_optimized": {
        "priority": 4,
        "scraper": "crawl4ai",
        "cost": "compute"
    }
}

def select_scraper(target_url):
    """智能选择爬虫"""
    if is_simple_api(target_url):
        return "api_available"
    elif is_static_page(target_url):
        return "static_page"
    elif requires_js_rendering(target_url):
        return "dynamic_or_complex"
    else:
        return "llm_optimized"
```

## 4.3 新闻采集示例

```python
class LLMNewsCollector:
    """LLM增强的新闻采集"""

    def __init__(self):
        self.firecrawl = FireCrawlSource(FIRECRAWL_API_KEY)
        self.llm = LLMService()
        self.rss = RSSCollector()

    async def collect_news(self, sources):
        results = []

        # 1. RSS 获取新闻列表
        rss_items = await self.rss.fetch(sources)

        # 2. 并行抓取详细内容
        tasks = []
        for item in rss_items:
            scraper = select_scraper(item["url"])
            if scraper == "firecrawl":
                tasks.append(self.firecrawl.scrape(item["url"]))
            else:
                tasks.append(self.traditional_scrape(item["url"]))

        contents = await asyncio.gather(*tasks)

        # 3. LLM 提取结构化信息
        for item, content in zip(rss_items, contents):
            structured = await self.llm.extract({
                "title": item["title"],
                "content": content.get("markdown", ""),
                "prompt": """
                提取：sentiment, event_type, impact, entities
                """
            })
            results.append({**item, **structured})

        return results
```
