# 🔧 五、TDP协议处理 `tdp/formatter.py`

```python
import time
import uuid
from typing import Dict, List, Any, Optional
from dataclasses import dataclass, asdict
from .validator import TDPValidator

@dataclass
class TDPMeta:
    """TDP Meta字段"""
    request_id: str
    source: str
    destination: str
    time_generated: int
    time_received: Optional[int] = None
    status: str = "OK"
    error: Optional[str] = None

@dataclass
class TDPMessage:
    """完整TDP消息格式"""
    version: str
    timestamp: int
    mode: str
    meta: TDPMeta
    market: Dict[str, Any]
    events: List[Dict]
    signals: Dict[str, Any]

class TDPFormatter:
    """TDP协议格式化器"""

    def __init__(self, version: str = "1.0", mode: str = "LIVE"):
        self.version = version
        self.mode = mode
        self.validator = TDPValidator()

    def create_message(
        self,
        source: str,
        destination: str,
        market_data: Dict = None,
        events: List[Dict] = None,
        signals: Dict = None,
        status: str = "OK",
        error: str = None
    ) -> Dict:
        """创建完整TDP消息"""
        message = {
            "version": self.version,
            "timestamp": int(time.time()),
            "mode": self.mode,
            "meta": asdict(TDPMeta(
                request_id=f"{source}_{uuid.uuid4().hex[:8]}",
                source=source,
                destination=destination,
                time_generated=int(time.time()),
                status=status,
                error=error
            )),
            "market": market_data or {},
            "events": events or [],
            "signals": signals or {}
        }

        self.validator.validate(message)
        return message

    def format_market_data(
        self,
        crypto: Dict = None,
        commodities: Dict = None,
        etf: Dict = None,
        funding_rate: Dict = None
    ) -> Dict:
        """格式化市场数据（TDP Market字段）"""
        market = {}

        if crypto:
            market["crypto"] = crypto

        if commodities:
            market["commodities"] = commodities

        if etf:
            market["etf"] = etf

        if funding_rate:
            market["funding_rate"] = funding_rate

        return market

    def format_crypto_price(
        self,
        symbol: str,
        price: float,
        open_: float = None,
        high: float = None,
        low: float = None,
        volume: float = None,
        change_24h: float = None,
        change_percent_24h: float = None,
        exchange: str = "binance",
        timeframe: str = "5m"
    ) -> Dict:
        """格式化单个加密货币价格"""
        data = {
            symbol: {
                "price": price,
                "exchange": exchange,
                "timeframe": timeframe,
                "timestamp": int(time.time())
            }
        }

        if open_ is not None:
            data[symbol]["open"] = open_
        if high is not None:
            data[symbol]["high"] = high
        if low is not None:
            data[symbol]["low"] = low
        if volume is not None:
            data[symbol]["volume"] = volume
        if change_24h is not None:
            data[symbol]["change_24h"] = change_24h
        if change_percent_24h is not None:
            data[symbol]["change_percent_24h"] = change_percent_24h

        return data

    def format_event(
        self,
        event_type: str,
        title: str,
        source: str,
        subtype: str = None,
        content: str = None,
        sentiment: str = None,
        sentiment_score: float = None,
        impact: str = None,
        related_assets: List[str] = None,
        lang: str = "en"
    ) -> Dict:
        """格式化事件数据"""
        event = {
            "id": f"evt_{uuid.uuid4().hex[:8]}",
            "type": event_type,
            "source": source,
            "title": title,
            "timestamp": int(time.time()),
            "lang": lang
        }

        if subtype:
            event["subtype"] = subtype
        if content:
            event["content"] = content
        if sentiment:
            event["sentiment"] = sentiment
        if sentiment_score is not None:
            event["sentiment_score"] = sentiment_score
        if impact:
            event["impact"] = impact
        if related_assets:
            event["related_assets"] = related_assets

        return event

    def format_news_with_llm(
        self,
        title: str,
        content: str,
        source: str,
        link: str,
        sentiment_score: float,
        sentiment_label: str,
        event_type: str,
        impact_level: int,
        entities: List[str],
        lang: str = "en"
    ) -> Dict:
        """格式化LLM分析后的新闻"""
        sentiment_map = {
            "positive": "bullish",
            "negative": "bearish",
            "neutral": "neutral"
        }

        impact_map = {
            1: "very_low",
            2: "low",
            3: "medium",
            4: "high",
            5: "very_high"
        }

        return self.format_event(
            event_type="news",
            title=title,
            source=source,
            subtype=event_type,
            content=content[:500] if content else None,
            sentiment=sentiment_map.get(sentiment_label, "neutral"),
            sentiment_score=sentiment_score,
            impact=impact_map.get(impact_level, "medium"),
            related_assets=entities,
            lang=lang
        )

    def format_social_with_llm(
        self,
        platform: str,
        kol_name: str = None,
        subreddit: str = None,
        sentiment_score: float = 0,
        sentiment_label: str = "neutral",
        dominant_topic: str = None,
        market_outlook: str = "neutral",
        influence_score: int = 5,
        content: str = None,
        lang: str = "en"
    ) -> Dict:
        """格式化LLM分析后的社交媒体"""
        sentiment_map = {
            "positive": "bullish",
            "negative": "bearish",
            "neutral": "neutral"
        }

        title = f"{platform.upper()}"
        if kol_name:
            title += f" - {kol_name}"
        if subreddit:
            title += f" - r/{subreddit}"

        return self.format_event(
            event_type="social",
            title=title,
            source=platform,
            subtype="sentiment_analysis",
            content=content[:500] if content else None,
            sentiment=sentiment_map.get(sentiment_label, "neutral"),
            sentiment_score=sentiment_score,
            impact="medium" if abs(sentiment_score) > 0.3 else "low",
            related_assets=[dominant_topic] if dominant_topic else None,
            lang=lang
        )

tdp_formatter = TDPFormatter()
```

# 🔧 六、TDP校验器 `tdp/validator.py`

```python
import logging

logger = logging.getLogger(__name__)

class TDPValidator:
    """TDP协议校验器"""

    REQUIRED_FIELDS = ["version", "timestamp", "mode", "meta"]
    META_REQUIRED = ["request_id", "source", "destination", "time_generated", "status"]
    FIELD_TYPES = {
        "version": str,
        "timestamp": int,
        "mode": str,
        "price": float,
        "volume": float,
        "sentiment_score": (int, float),
    }
    FIELD_RANGES = {
        "sentiment_score": (-1, 1),
        "change_percent_24h": (-1, 1),
        "leverage": (1, 125),
        "position_size": (0, 1)
    }
    VALID_MODES = ["LIVE", "BACKTEST", "SIMULATION"]
    VALID_STATUS = ["OK", "ERROR", "PARTIAL"]

    def validate(self, message: dict) -> bool:
        """校验完整消息"""
        try:
            self.validate_required_fields(message)
            self.validate_meta(message.get("meta", {}))
            self.validate_mode(message.get("mode"))
            self.validate_market(message.get("market", {}))
            self.validate_ranges(message)
            logger.debug(f"TDP消息校验通过: {message.get('meta', {}).get('request_id')}")
            return True
        except TDPValidationError as e:
            logger.error(f"TDP校验失败: {e}")
            raise

    def validate_required_fields(self, message: dict):
        """校验必填字段"""
        for field in self.REQUIRED_FIELDS:
            if field not in message:
                raise TDPValidationError(f"Missing required field: {field}")

    def validate_meta(self, meta: dict):
        """校验meta字段"""
        for field in self.META_REQUIRED:
            if field not in meta:
                raise TDPValidationError(f"Missing meta field: {field}")

        if meta.get("status") not in self.VALID_STATUS:
            raise TDPValidationError(f"Invalid status: {meta.get('status')}")

    def validate_mode(self, mode: str):
        """校验mode字段"""
        if mode not in self.VALID_MODES:
            raise TDPValidationError(f"Invalid mode: {mode}")

    def validate_market(self, market: dict):
        """校验market字段"""
        if "crypto" in market:
            for symbol, data in market["crypto"].items():
                if "price" not in data:
                    raise TDPValidationError(f"Missing price for {symbol}")

    def validate_ranges(self, message: dict):
        """校验数值范围"""
        def check_range(value, min_val, max_val, field_path):
            if not (min_val <= value <= max_val):
                raise TDPValidationError(
                    f"Value {value} out of range [{min_val}, {max_val}] for {field_path}"
                )

        if "signals" in message:
            signals = message["signals"]
            if "risk" in signals:
                risk = signals["risk"]
                if "risk_index" in risk:
                    check_range(risk["risk_index"], 0, 100, "signals.risk.risk_index")

        for event in message.get("events", []):
            if "sentiment_score" in event:
                check_range(event["sentiment_score"], -1, 1, "events.sentiment_score")

class TDPValidationError(Exception):
    pass
```

```python
import redis
import json
import logging
from typing import Dict, Optional
from .config import settings

logger = logging.getLogger(__name__)

class StateClient:
    """State服务客户端 - 管理采集状态"""

    STATE_KEYS = {
        "data_status": "state:data:status",
        "last_collection": "state:data:last_collection",
        "collection_stats": "state:data:stats",
        "source_health": "state:data:source_health",
    }

    def __init__(self):
        self.redis = redis.from_url(settings.REDIS_URL, decode_responses=True)

    def update_data_status(self, status: str, error: str = None):
        """更新数据采集状态"""
        data = {
            "status": status,
            "error": error,
            "timestamp": int(time.time())
        }
        self.redis.set(self.STATE_KEYS["data_status"], json.dumps(data))
        logger.debug(f"更新数据状态: {status}")

    def get_data_status(self) -> Dict:
        """获取数据采集状态"""
        data = self.redis.get(self.STATE_KEYS["data_status"])
        return json.loads(data) if data else {"status": "UNKNOWN"}

    def update_last_collection(self, source: str, count: int):
        """更新最后采集时间"""
        data = {
            "source": source,
            "count": count,
            "timestamp": int(time.time())
        }
        self.redis.set(self.STATE_KEYS["last_collection"], json.dumps(data))

    def get_last_collection(self) -> Dict:
        """获取最后采集信息"""
        data = self.redis.get(self.STATE_KEYS["last_collection"])
        return json.loads(data) if data else {}

    def update_source_health(self, source: str, success: bool, error: str = None):
        """更新数据源健康状态"""
        health_key = f"{self.STATE_KEYS['source_health']}:{source}"
        data = {
            "source": source,
            "success": success,
            "error": error,
            "timestamp": int(time.time())
        }
        self.redis.set(health_key, json.dumps(data))
        logger.debug(f"更新源健康状态: {source} - {'OK' if success else 'FAIL'}")

    def get_source_health(self, source: str) -> Dict:
        """获取数据源健康状态"""
        health_key = f"{self.STATE_KEYS['source_health']}:{source}"
        data = self.redis.get(health_key)
        return json.loads(data) if data else {"source": source, "success": False}

    def get_all_source_health(self) -> Dict:
        """获取所有数据源健康状态"""
        health = {}
        for key in self.redis.scan_iter(f"{self.STATE_KEYS['source_health']}:*"):
            source = key.split(":")[-1]
            health[source] = self.get_source_health(source)
        return health

    def record_collection_stats(self, source: str, collected: int, duration: float):
        """记录采集统计"""
        stats_key = f"{self.STATE_KEYS['collection_stats']}:{source}"
        existing = self.redis.get(stats_key)
        stats = json.loads(existing) if existing else {
            "total_count": 0,
            "total_duration": 0,
            "avg_duration": 0
        }

        stats["total_count"] += collected
        stats["total_duration"] += duration
        stats["avg_duration"] = stats["total_duration"] / max(1, stats["total_count"])

        self.redis.set(stats_key, json.dumps(stats))

    def get_collection_stats(self, source: str) -> Dict:
        """获取采集统计"""
        stats_key = f"{self.STATE_KEYS['collection_stats']}:{source}"
        data = self.redis.get(stats_key)
        return json.loads(data) if data else {}

state_client = StateClient()
```

```python
import requests
import logging
from typing import Dict, Optional
from .config import settings
import time

logger = logging.getLogger(__name__)

class ConfigClient:
    """Config服务客户端 - 获取采集配置"""

    def __init__(self):
        self.base_url = settings.CONFIG_SERVICE_URL
        self.cache = {}
        self.cache_ttl = settings.CONFIG_CACHE_TTL

    def get_config(self, category: str) -> Dict:
        """获取配置（带缓存）"""
        cache_key = f"config:{category}"

        if cache_key in self.cache:
            cached_data, cached_time = self.cache[cache_key]
            if time.time() - cached_time < self.cache_ttl:
                return cached_data

        try:
            response = requests.get(
                f"{self.base_url}/api/v1/config/{category}",
                timeout=5
            )
            if response.status_code == 200:
                data = response.json()
                self.cache[cache_key] = (data, time.time())
                return data
            else:
                logger.warning(f"Config服务返回: {response.status_code}")
                return self.get_fallback_config(category)
        except Exception as e:
            logger.error(f"Config服务请求失败: {e}")
            return self.get_fallback_config(category)

    def get_fallback_config(self, category: str) -> Dict:
        """获取默认配置"""
```

# 🔧 九、采集器基类 `collector/base.py`（导入 shared.tdp）

```python
import asyncio
import time
from abc import ABC, abstractmethod
from typing import Dict, Any, List
from dataclasses import dataclass
import logging

from shared.tdp import tdp_formatter      # 从共享库导入
from shared.state import state_client     # 从共享库导入
from shared.config import config_client    # 从共享库导入

logger = logging.getLogger(__name__)

@dataclass
class CollectResult:
    success: bool
    data: Any
    source: str
    timestamp: float
    tdp_message: Dict = None
    error: str = None

class BaseCollector(ABC):
    def __init__(self, name: str, interval: int, source_config: Dict = None):
        self.name = name
        self.interval = interval
        self.source_config = source_config or {}
        self.primary_source = self.source_config.get("primary", "")
        self.fallback_sources = self.source_config.get("fallback", [])
        self.last_run = 0
        self.failure_count = 0
        self.max_failures = 3
        self.current_source = self.primary_source

    def load_config(self):
        """从Config服务加载配置"""
        intervals = config_client.get_collection_intervals()
        if self.name in intervals:
            self.interval = intervals[self.name]
        sources = config_client.get_data_sources()
        if self.name in sources:
            self.source_config = sources[self.name]

    @abstractmethod
    async def fetch_from_source(self, source: str) -> List[Dict]:
        pass

    async def fetch(self) -> List[Dict]:
        all_results = []
        sources_to_try = [self.current_source] + self.fallback_sources

        for source in sources_to_try:
            if not source:
                continue
            try:
                start_time = time.time()
                results = await self.fetch_from_source(source)
                duration = time.time() - start_time

                if results:
                    all_results.extend(results)
                    self.current_source = source
                    self.failure_count = 0

                    state_client.update_source_health(source, True)
                    state_client.record_collection_stats(source, len(results), duration)
                    state_client.update_last_collection(source, len(results))

                    logger.info(f"{self.name} 从 {source} 获取成功: {len(results)} 条")
                    break

            except Exception as e:
                logger.warning(f"{self.name} {source} 获取失败: {e}")
                state_client.update_source_health(source, False, str(e))
                continue

        if not all_results:
            self.failure_count += 1
            state_client.update_data_status("ERROR", f"所有数据源均失败")
            raise Exception(f"所有数据源均失败")

        state_client.update_data_status("OK")
        return all_results

    @abstractmethod
    def parse(self, raw_data) -> Dict:
        pass

    async def run(self) -> CollectResult:
        self.load_config()

        try:
            raw = await self.fetch()
            parsed = self.parse(raw)
            self.last_run = time.time()

            tdp_message = tdp_formatter.create_message(
                source=f"{self.name}_service",
                destination="factor_engine",
                market_data=parsed.get("market"),
                events=parsed.get("events", []),
                signals=parsed.get("signals", {}),
                status="OK"
            )

            return CollectResult(
                success=True,
                data=parsed,
                source=self.current_source,
                timestamp=self.last_run,
                tdp_message=tdp_message
            )

        except Exception as e:
            self.failure_count += 1
            logger.error(f"{self.name} 采集失败: {e}")
            state_client.update_data_status("ERROR", str(e))

            tdp_message = tdp_formatter.create_message(
                source=f"{self.name}_service",
                destination="factor_engine",
                status="ERROR",
                error=str(e)
            )

            return CollectResult(
                success=False,
                data=None,
                source=self.current_source,
                timestamp=time.time(),
                tdp_message=tdp_message,
                error=str(e)
            )

    def is_available(self) -> bool:
        return self.failure_count < self.max_failures
```

```python
import asyncio
import ccxt
from typing import Dict, List
from .base import BaseCollector
import logging

logger = logging.getLogger(__name__)

class ExchangeCollector(BaseCollector):
    def __init__(self, symbols: List[str] = None):
        self.symbols = symbols or ["BTC/USDT", "ETH/USDT"]
        source_config = config_client.get_data_sources().get("market", MARKET_SOURCES)
        super().__init__("market", interval=60, source_config=source_config)
        self.exchanges = {}
        self._init_exchanges()

    def _init_exchanges(self):
        self.exchanges = {
            'binance': ccxt.binance({'enableRateLimit': True, 'options': {'defaultType': 'spot'}}),
            'okx': ccxt.okx({'enableRateLimit': True, 'options': {'defaultType': 'spot'}}),
            'huobi': ccxt.huobi({'enableRateLimit': True}),
            'gateio': ccxt.gateio({'enableRateLimit': True}),
            'coinbase': ccxt.coinbase({'enableRateLimit': True}),
        }

    async def fetch_from_source(self, source: str) -> List[Dict]:
        if source not in self.exchanges:
            raise Exception(f"未知交易所: {source}")

        exchange = self.exchanges[source]
        results = []

        for symbol in self.symbols:
            try:
                ticker = await asyncio.to_thread(exchange.fetch_ticker, symbol)
                results.append({
                    'symbol': symbol,
                    'exchange': source,
                    'price': ticker['last'],
                    'volume_24h': ticker['baseVolume'],
                    'high_24h': ticker['high'],
                    'low_24h': ticker['low'],
                    'timestamp': ticker['timestamp']
                })
            except Exception as e:
                logger.warning(f"{source} {symbol} 获取失败: {e}")

        return results

    def parse(self, raw_data: List[Dict]) -> Dict:
        crypto_data = {}

        for item in raw_data:
            symbol = item['symbol'].replace('/', '')
            if symbol not in crypto_data:
                crypto_data[symbol] = {
                    'price': float(item['price']),
                    'exchange': item['exchange'],
                    'volume': float(item['volume_24h']),
                    'high': float(item['high_24h']),
                    'low': float(item['low_24h']),
                    'timestamp': item['timestamp']
                }

        market = tdp_formatter.format_market_data(crypto=crypto_data)

        return {
            "market": market,
            "events": [],
            "signals": {}
        }
```

```python
import asyncio
import feedparser
from typing import Dict, List
from .base import BaseCollector
from shared.tdp import tdp_formatter     # 从共享库导入
from ..scraper.firecrawl_scraper import FireCrawlScraper
from ..scraper.cn_news_scraper import CnNewsScraper
from ..scraper.llm_service import llm_service
import logging

logger = logging.getLogger(__name__)

class NewsCollector(BaseCollector):
    RSS_FEEDS_EN = {
        'coindesk': 'https://www.coindesk.com/arc/outboundfeeds/rss/',
        'cointelegraph': 'https://cointelegraph.com/rss',
        'cryptopanic': 'https://cryptopanic.com/news/feed/',
    }

    COMPLEX_NEWS_SITES = {
        'theblock': 'https://www.theblock.co/rss.xml',
        'decrypt': 'https://decrypt.co/feed',
    }

    def __init__(self):
        source_config = config_client.get_data_sources().get("news", {})
        super().__init__("news", interval=300, source_config=source_config)
        self.firecrawl = FireCrawlScraper()
        self.cn_scraper = CnNewsScraper()

    async def fetch(self) -> List[Dict]:
        all_results = []

        rss_results = await self._fetch_rss_news()
        all_results.extend(rss_results)

        cn_results = await self._fetch_cn_news()
        all_results.extend(cn_results)

        complex_results = await self._fetch_complex_pages()
        all_results.extend(complex_results)

        return all_results

    async def _fetch_rss_news(self) -> List[Dict]:
        results = []
        for source, url in self.RSS_FEEDS_EN.items():
            try:
                feed = await asyncio.to_thread(feedparser.parse, url)
                for entry in feed.entries[:10]:
                    results.append({
                        'title': entry.get('title', ''),
                        'link': entry.get('link', ''),
                        'published': entry.get('published', ''),
                        'source': source,
                        'lang': 'en',
                        'summary': entry.get('summary', '')[:500],
                        'fetch_method': 'rss'
                    })
            except Exception as e:
                logger.warning(f"{source} RSS失败: {e}")
        return results

    async def _fetch_cn_news(self) -> List[Dict]:
        results = []
        cn_sources = self.source_config.get('cn', ['jinshi', 'odaily'])
        for source in cn_sources:
            try:
                cn_items = await self.cn_scraper.fetch(source)
                results.extend(cn_items)
            except Exception as e:
                logger.warning(f"{source} 中文新闻失败: {e}")
        return results

    async def _fetch_complex_pages(self) -> List[Dict]:
        results = []
        for source, url in self.COMPLEX_NEWS_SITES.items():
            try:
                analyzed = await self._scrape_with_llm(url, source)
                if analyzed:
                    results.append(analyzed)
            except Exception as e:
                logger.warning(f"{source} LLM抓取失败: {e}")
        return results

    async def _scrape_with_llm(self, url: str, source: str) -> Dict:
        scraped = await self.firecrawl.scrape(url)
        if not scraped.get("markdown"):
            return {}

        content = scraped["markdown"]
        structured = await llm_service.extract_news_structured(
            title=content[:200],
            content=content[:4000]
        )

        return tdp_formatter.format_news_with_llm(
            title=structured.get('summary', '')[:200] or content[:100],
            content=content[:500],
            source=source,
            link=url,
            sentiment_score=structured.get('sentiment_score', 0),
            sentiment_label=structured.get('sentiment_label', 'neutral'),
            event_type=structured.get('event_type', 'unknown'),
            impact_level=structured.get('impact_level', 3),
            entities=structured.get('entities', []),
            lang='en'
        )

    def parse(self, raw_data: List[Dict]) -> Dict:
        events = []

        for item in raw_data:
            if item.get('fetch_method') == 'llm_scrape':
                events.append(item)
            else:
                event = tdp_formatter.format_event(
                    event_type="news",
                    title=item.get('title', ''),
                    source=item.get('source', ''),
                    content=item.get('summary', ''),
                    lang=item.get('lang', 'en')
                )
                events.append(event)

        return {
            "market": {},
            "events": events,
            "signals": {}
        }
```

```python
import os
import json
import logging
from typing import Dict, List

logger = logging.getLogger(__name__)

class LLMService:
    def __init__(self):
        self.api_key = os.getenv("OPENAI_API_KEY", "")
        self.model = "gpt-4o-mini"

    async def extract_structured(self, content: str, prompt: str) -> Dict:
        if not self.api_key:
            logger.warning("OPENAI_API_KEY 未设置")
            return {}

        try:
            import openai
            client = openai.OpenAI(api_key=self.api_key)

            response = client.chat.completions.create(
                model=self.model,
                messages=[
                    {"role": "system", "content": "你是一个信息提取专家"},
                    {"role": "user", "content": f"内容：\n{content[:4000]}\n\n{prompt}"}
                ],
                response_format={"type": "json_object"}
            )

            return json.loads(response.choices[0].message.content)

        except Exception as e:
            logger.error(f"LLM提取失败: {e}")
            return {}

    async def analyze_sentiment(self, text: str) -> Dict:
        prompt = f"""
        分析以下文本的情绪：
        文本：{text[:2000]}

        输出JSON：
        {{
            "score": 情绪分数(-1到1),
            "label": "positive/negative/neutral",
            "event_type": "政策/技术/市场/其他",
            "topics": ["话题1", "话题2"],
            "entities": ["实体1", "实体2"]
        }}
        """
        return await self.extract_structured(text, prompt)

    async def extract_news_structured(self, title: str, content: str) -> Dict:

            "entities": ["BTC", "ETH", "Binance", "SEC"],

        }}
        """
        return await self.extract_structured(content, prompt)
```

```python
import json

from typing import Dict
from aiokafka import AIOKafkaProducer
from .config import settings

logger = logging.getLogger(__name__)

class KafkaProducer:
    def __init__(self):
        self.producer = None

    async def start(self):
        self.producer = AIOKafkaProducer(
            bootstrap_servers=settings.KAFKA_BOOTSTRAP_SERVERS,
            value_serializer=lambda v: json.dumps(v).encode('utf-8')
        )
        await self.producer.start()
        logger.info("Kafka Producer 已启动")

    async def stop(self):
        if self.producer:
            await self.producer.stop()

    async def send(self, topic: str, tdp_message: Dict):
        if not self.producer:
            await self.start()

        try:
            await self.producer.send_and_wait(topic, tdp_message)
            logger.debug(f"TDP消息已发送至 {topic}: {tdp_message.get('meta', {}).get('request_id')}")
        except Exception as e:
            logger.error(f"Kafka 发送失败: {e}")
```

```python
from fastapi import WebSocket
from typing import List
import logging

logger = logging.getLogger(__name__)

class WebSocketManager:
    def __init__(self):
        self.active_connections: List[WebSocket] = []

    async def connect(self, websocket: WebSocket):
        await websocket.accept()
        self.active_connections.append(websocket)
        logger.info(f"WebSocket 连接: {len(self.active_connections)}")

    def disconnect(self, websocket: WebSocket):
        if websocket in self.active_connections:
            self.active_connections.remove(websocket)

    async def broadcast(self, message: dict):
        disconnected = []
        for connection in self.active_connections:
            try:
                await connection.send_json(message)
            except Exception:
                disconnected.append(connection)

        for conn in disconnected:
            self.disconnect(conn)

ws_manager = WebSocketManager()
```

```python
    try:
        while True:
            await websocket.receive_text()
    except Exception:
        ws_manager.disconnect(websocket)

    return {
        "status": "ok",

    collectors = [
        ExchangeCollector(),
        NewsCollector(),
        MacroCollector(),
        ETFCollector(),

                        settings.KAFKA_TOPIC_RAW_DATA,
                        result.tdp_message
                    )
```
