# 📊 交易员数据 vs 其他数据的差异

| 特点 | 交易员数据 | 新闻资讯 | 社交媒体 |
|------|------------|----------|----------|
| **数据结构** | 🔴 80%非结构化 | 🟡 50%非结构化 | 🔴 90%非结构化 |
| **LLM依赖** | 🔴 **极高**（观点提取） | 🟡 高（情绪分析） | 🟢 低（关键词过滤） |
| **链上关联** | ✅ 有（持仓追踪） | ❌ 无 | ❌ 无 |
| **数据源** | 混合（链上+链下） | 主要链下 | 主要链下 |
| **时效** | 🟡 分钟级 | 🟡 分钟级 | 🔴 实时 |

## 1. 数据模型

```python
from dataclasses import dataclass, field
from typing import List, Optional, Dict, Any
from datetime import datetime
from enum import Enum

class DataStructuredness(Enum):
    STRUCTURED = "structured"
    SEMI_STRUCTURED = "semi_structured"
    UNSTRUCTURED = "unstructured"

class SentimentScore(Enum):
    BULLISH = "bullish"
    BEARISH = "bearish"
    NEUTRAL = "neutral"

@dataclass
class TraderProfile:
    """交易员画像"""
    trader_id: str
    name: str
    aliases: List[str]
    platform: List[str]
    followers_count: int
    credibility_score: float
    historical_accuracy: float
    typical_holding_size: str
    known_for: List[str]
    wallet_addresses: List[str] = field(default_factory=list)

@dataclass
class TraderStatement:
    """交易员言论（由LLM提取）"""
    statement_id: str
    trader_id: str
    platform: str
    original_text: str
    parsed观点: str
    sentiment: SentimentScore
    sentiment_confidence: float
    mentioned_assets: List[str]
    time_horizon: str
    is_historical_record: bool
    source_url: str
    timestamp: datetime
    llm_model: str
    llm_confidence: float

@dataclass
class OnChainPosition:
    """链上持仓数据"""
    trader_id: str
    wallet_address: str
    blockchain: str
    asset: str
    balance: float
    balance_usd: float
    last_updated: datetime
    source: str

@dataclass
class PositionChange:
    """持仓变动（关联链上与言论）"""
    change_id: str
    trader_id: str
    asset: str
    change_type: str
    change_amount: float
    change_percent: float
    before_balance: float
    after_balance: float
    timestamp: datetime
    correlated_statements: List[str]
    correlation_confidence: float

@dataclass
class TraderData:
    """交易员综合数据"""
    trader_id: str
    profile: TraderProfile
    statements: List[TraderStatement]
    positions: List[OnChainPosition]
    position_changes: List[PositionChange]
    composite_sentiment: SentimentScore
    sentiment_confidence: float
    last_updated: datetime
```

# 🔄 采集流程

## 1. 采集Pipeline

```python
class TraderDataCollector:
    """交易员数据采集器"""

    def __init__(self, config: Dict):
        self.llm_collector = FireCrawlCollector()
        self.twitter_api = TwitterAPICollector()
        self.dune_api = DuneRESTCollector()
        self.nansen_api = NansenRESTCollector()
        self.exchange_api = ExchangeRESTCollector()
        self.trader_profiles = self._load_trader_profiles()

    async def collect_all(self) -> List[TraderData]:
        """全量采集"""
        results = await asyncio.gather(
            self._collect_twitter_statements(),
            self._collect_telegram_signals(),
            self._collect_on_chain_positions(),
            self._collect_exchange_wallets(),
            return_exceptions=True
        )

        twitter, telegram, on_chain, exchange = results

        traders = self._merge_trader_data(twitter, telegram, on_chain, exchange)

        for trader in traders:
            trader = await self._llm_analysis(trader)
            trader = await self._correlate_on_chain_off_chain(trader)

        return traders

    async def _collect_twitter_statements(self) -> Dict[str, List[TraderStatement]]:
        """采集Twitter KOL言论"""
        traders = self._get_target_traders()

        tasks = [self._scrape_twitter(trader) for trader in traders]
        results = await asyncio.gather(*tasks, return_exceptions=True)

        return {t.trader_id: t for t in results if isinstance(t, TraderStatement)}

    async def _scrape_twitter(self, trader: TraderProfile) -> TraderStatement:
        """抓取单个交易员Twitter"""
        url = f"https://twitter.com/{trader.aliases[0]}"
        content = await self.llm_collector.scrape(url)

        prompt = TRADER_LLM_PROMPTS["statement_extraction"].format(text=content)
        parsed = await self._llm_extract(prompt)

        return TraderStatement(
            statement_id=generate_id(),
            trader_id=trader.trader_id,
            platform="twitter",
            original_text=content[:1000],
            parsed观点=parsed["观点"],
            sentiment=SentimentScore(parsed["情绪"]),
            sentiment_confidence=parsed["情绪置信度"],
            mentioned_assets=parsed["资产"],
            time_horizon=parsed["时间预期"],
            is_historical_record=False,
            source_url=url,
            timestamp=datetime.now(),
            llm_model="gpt-4o-mini",
            llm_confidence=0.85
        )

    async def _collect_on_chain_positions(self) -> Dict[str, List[OnChainPosition]]:
        """采集链上持仓"""
        traders = self._get_target_traders()

        positions = {}
        for trader in traders:
            if trader.wallet_addresses:
                tasks = [self._fetch_wallet_position(addr) for addr in trader.wallet_addresses]
                trader_positions = await asyncio.gather(*tasks, return_exceptions=True)
                positions[trader.trader_id] = [p for p in trader_positions if isinstance(p, OnChainPosition)]

        return positions

    async def _llm_analysis(self, trader: TraderData) -> TraderData:
        """对交易员数据进行LLM综合分析"""
        all_statements = "\n".join([
            f"- [{s.timestamp}] {s.original_text[:200]}"
            for s in trader.statements
        ])

        prompt = TRADER_LLM_PROMPTS["sentiment_scoring"].format(
            statement=all_statements[:2000],
            historical_accuracy=trader.profile.historical_accuracy,
            followers_count=trader.profile.followers_count
        )

        result = await self._llm_extract(prompt)

        trader.composite_sentiment = SentimentScore(result["adjusted_sentiment"])
        trader.sentiment_confidence = result["adjusted_confidence"]

        return trader

    async def _correlate_on_chain_off_chain(self, trader: TraderData) -> TraderData:
        """链上链下数据关联"""
        for change in trader.position_changes:
            statements_text = "\n".join([
                f"- {s.original_text[:200]}" for s in trader.statements
                if s.timestamp < change.timestamp
                and any(a in s.mentioned_assets for a in [change.asset])
            ])

            prompt = TRADER_LLM_PROMPTS["correlation_analysis"].format(
                wallet_address=change.wallet_address,
                asset=change.asset,
                amount=change.change_amount,
                percent=change.change_percent,
                timestamp=change.timestamp,
                statements=statements_text[:1000]
            )

            result = await self._llm_extract(prompt)

            change.correlation_confidence = result["correlation_confidence"]
            change.correlated_statements = [
                s.statement_id for s in trader.statements
                if s.mentioned_assets and change.asset in s.mentioned_assets
            ]

        return trader
```

## 2. KOL权重计算

```python
def calculate_kol_weight(trader: TraderProfile) -> float:
    """计算KOL综合权重"""
    follower_score = min(trader.followers_count / 1000000, 1.0) * 0.3
    credibility_score = trader.credibility_score * 0.4
    accuracy_score = trader.historical_accuracy * 0.3

    return follower_score + credibility_score + accuracy_score

def adjust_sentiment_by_kol(sentiment: SentimentScore, confidence: float, trader: TraderProfile) -> Tuple[SentimentScore, float]:
    """根据KOL权重调整情绪置信度"""
    kol_weight = calculate_kol_weight(trader)

    adjusted_confidence = confidence * (0.5 + 0.5 * kol_weight)

    return sentiment, adjusted_confidence
```
