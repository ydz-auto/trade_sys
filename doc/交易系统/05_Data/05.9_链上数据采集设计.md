```python
from abc import ABC, abstractmethod
from typing import List, Dict, Optional, Any
import asyncio

class OnChainDataProvider(ABC):
    """链上数据Provider抽象"""

    def __init__(self, api_key: str = None):

    @abstractmethod
    async def get_metric(
        self,
        chain: Chain,
        metric_name: str,
        start_time: datetime,
        end_time: datetime,
        interval: str = "1h"
    ) -> List[OnChainMetric]:

        pass

    @abstractmethod
    async def get_whale_activity(
        self,
        chain: Chain,
        min_amount_usd: float = 100000,
        start_time: datetime = None
    ) -> List[WhaleActivity]:
        """获取Whale活动"""
        pass

    @abstractmethod
    async def get_holder_distribution(
        self,
        chain: Chain,
        timestamp: datetime = None
    ) -> HolderDistribution:
        """获取持仓分布"""
        pass

    @abstractmethod
    async def get_exchange_flow(
        self,
        exchange: str,
        chain: Chain,
        start_time: datetime,
        end_time: datetime
    ) -> List[ExchangeFlow]:
        """获取交易所流量"""
        pass


class OnChainAggregator:
    """链上数据聚合器 - 统一多个免费数据源"""

    def __init__(self):
        self.providers: Dict[str, OnChainDataProvider] = {}
        self.cache = RedisCache()
        self.fallback_chain = ["dune", "coingecko", "etherscan"]

    def register_provider(self, name: str, provider: OnChainDataProvider):
        """注册数据Provider"""
        self.providers[name] = provider

    async def get_metric_with_fallback(
        self,
        chain: Chain,
        metric_name: str,
        start_time: datetime,
        end_time: datetime,
        preferred_provider: str = None
    ) -> List[OnChainMetric]:
        """
        带fallback的指标获取

        优先使用首选Provider，失败时自动切换到备用Provider
        """

        providers_to_try = []
        if preferred_provider and preferred_provider in self.providers:
            providers_to_try.append(preferred_provider)

        # 添加fallback chain
        for provider_name in self.fallback_chain:
            if provider_name not in providers_to_try and provider_name in self.providers:
                providers_to_try.append(provider_name)

        last_error = None
        for provider_name in providers_to_try:
            try:
                # 检查缓存
                cache_key = f"metric:{provider_name}:{chain.value}:{metric_name}:{start_time}:{end_time}"
                cached = await self.cache.get(cache_key)
                if cached:
                    return cached

                # 获取数据
                provider = self.providers[provider_name]
                result = await provider.get_metric(
                    chain=chain,
                    metric_name=metric_name,
                    start_time=start_time,
                    end_time=end_time
                )

                # 缓存结果
                await self.cache.set(cache_key, result, ttl=3600)

                return result

            except Exception as e:
                last_error = e
                logger.warning(f"{provider_name}获取{metric_name}失败: {e}")
                continue

        raise AllProvidersFailedError(
            f"所有Provider均失败: {last_error}"
        )
```

## 1. Dune客户端

```python
import aiohttp
import pandas as pd
from dune_client.client import DuneClient
from dune_client.models import DuneExecution

class DuneAnalyticsProvider(OnChainDataProvider):
    """Dune Analytics数据Provider"""

    BASE_URL = "https://api.dune.com/api/v1"

    def __init__(self, api_key: str):
        super().__init__(api_key)
        self.client = DuneClient(api_key)

    async def get_metric(
        self,
        chain: Chain,
        metric_name: str,
        start_time: datetime,
        end_time: datetime,
        interval: str = "1h"
    ) -> List[OnChainMetric]:
        """通过Dune SQL查询获取链上指标"""

        # 预定义SQL模板
        sql_template = self._get_sql_template(chain, metric_name)

        if not sql_template:
            raise NotSupportedError(f"Dune不支持的指标: {metric_name}")

        # 构建SQL
        sql = sql_template.format(
            start_time=start_time.strftime("%Y-%m-%d %H:00:00"),
            end_time=end_time.strftime("%Y-%m-%d %H:00:00"),
            interval=interval
        )

        # 执行查询
        results = await self._execute_query(sql)

        # 转换为OnChainMetric
        return self._parse_results(chain, metric_name, results, interval)

    async def _execute_query(self, sql: str) -> pd.DataFrame:
        """执行Dune查询"""
        # 创建执行任务
        execution = self.client.create_execution(sql)

        # 轮询结果
        max_wait = 300  # 5分钟超时
        waited = 0

        while waited < max_wait:
            status = self.client.get_execution_status(execution.job_id)

            if status.state == "SUCCESSFUL":
                return self.client.get_execution_results(execution.job_id)

            elif status.state in ["FAILED", "CANCELLED"]:
                raise Exception(f"查询失败: {status.state}")

            await asyncio.sleep(5)
            waited += 5

        raise TimeoutError("查询超时")

    def _get_sql_template(self, chain: Chain, metric_name: str) -> str:
        """获取SQL模板"""
        templates = {
            "bitcoin": {
                "active_addresses": """
                    SELECT
                        time,
                        COUNT(DISTINCT from_address) as active_addresses
                    FROM bitcoin.transactions
                    WHERE block_time >= '{start_time}'
                      AND block_time < '{end_time}'
                    GROUP BY time
                    ORDER BY time
                """,
                "transaction_count": """
                    SELECT
                        time,
                        COUNT(*) as tx_count
                    FROM bitcoin.transactions
                    WHERE block_time >= '{start_time}'
                      AND block_time < '{end_time}'
                    GROUP BY time
                """,
                "exchange_inflow": """
                    SELECT
                        time,
                        SUM(CASE WHEN to_address IN (SELECT address FROM exchange_addresses)
                            THEN amount ELSE 0 END) as inflow
                    FROM bitcoin.transfers
                    WHERE block_time >= '{start_time}'
                      AND block_time < '{end_time}'
                    GROUP BY time
                """
            },
            "ethereum": {
                "active_addresses": """
                    SELECT
                        date_trunc('{interval}', block_time) as time,
                        COUNT(DISTINCT from_address) as active_addresses
                    FROM ethereum.transactions
                    WHERE block_time >= '{start_time}'
                      AND block_time < '{end_time}'
                    GROUP BY 1
                """,
                "gas_used": """
                    SELECT
                        date_trunc('{interval}', block_time) as time,
                        AVG(gas_used) as avg_gas_used,
                        AVG(gas_price) as avg_gas_price
                    FROM ethereum.transactions
                    WHERE block_time >= '{start_time}'
                      AND block_time < '{end_time}'
                    GROUP BY 1
                """
            }
        }

        chain_templates = templates.get(chain.value, {})
        return chain_templates.get(metric_name, "")

    def _parse_results(
        self,
        chain: Chain,
        metric_name: str,
        results: pd.DataFrame,
        interval: str
    ) -> List[OnChainMetric]:
        """解析查询结果"""
        metrics = []

        for _, row in results.iterrows():
            metric = OnChainMetric(
                metric_id=f"dune_{chain.value}_{metric_name}_{row['time']}",
                chain=chain,
                category=self._get_category(metric_name),
                name=metric_name,
                timestamp=row["time"],
                window=interval,
                value=row.get(list(row.keys())[-1], 0),
                unit=self._get_unit(metric_name),
                source="dune"
            )
            metrics.append(metric)

        return metrics

    def _get_category(self, metric_name: str) -> MetricCategory:
        """获取指标分类"""
        categories = {
            "active_addresses": MetricCategory.HOLDERS,
            "transaction_count": MetricCategory.TRANSACTIONS,
            "exchange_inflow": MetricCategory.EXCHANGE_FLOW,
            "gas_used": MetricCategory.GAS,
        }
        return categories.get(metric_name, MetricCategory.NETWORK)

    def _get_unit(self, metric_name: str) -> str:
        """获取指标单位"""
        units = {
            "active_addresses": "count",
            "transaction_count": "count",
            "exchange_inflow": "BTC",
            "gas_used": "gas",
        }
        return units.get(metric_name, "value")
```

```python
class DuneQueryManager:
    """Dune查询管理器 - 优化API使用成本"""

    def __init__(self, client: DuneAnalyticsProvider):
        self.client = client
        self.query_cache = {}  # 查询ID缓存
        self.results_cache = RedisCache(ttl=3600)

    async def execute_cached_query(
        self,
        query_name: str,
        sql: str,
        max_age_seconds: int = 3600
    ) -> pd.DataFrame:
        """
        执行带缓存的查询

        1. 检查是否有相同查询正在执行
        2. 检查缓存是否有新鲜结果
        3. 执行新查询
        """

        cache_key = f"dune_query:{hashlib.md5(sql.encode()).hexdigest()}"

        # 检查缓存
        cached = await self.results_cache.get(cache_key)
        if cached is not None:
            return cached

        # 检查是否有正在执行的相同查询
        if query_name in self.query_cache:
            existing_job_id = self.query_cache[query_name]
            try:
                # 等待现有查询完成
                result = await self._wait_for_query(existing_job_id)
                await self.results_cache.set(cache_key, result)
                return result
            except:
                pass  # 查询失败，继续执行新查询

        # 执行新查询
        execution = self.client.client.create_execution(sql)
        self.query_cache[query_name] = execution.job_id

        try:
            result = await self._wait_for_query(execution.job_id)
            await self.results_cache.set(cache_key, result)
            return result
        finally:
            del self.query_cache[query_name]

    async def _wait_for_query(self, job_id: str, timeout: int = 300) -> pd.DataFrame:
        """等待查询完成"""
        waited = 0
        while waited < timeout:
            status = self.client.client.get_execution_status(job_id)

            if status.state == "SUCCESSFUL":
                return self.client.client.get_execution_results(job_id)

            elif status.state in ["FAILED", "CANCELLED"]:
                raise Exception(f"查询失败: {status.state}")

            await asyncio.sleep(5)
            waited += 5

        raise TimeoutError("查询超时")
```

```python
class OnChainFactorCalculator:
    """链上因子计算器"""

    def calculate_whale_ratio(self, holder_dist: HolderDistribution) -> float:
        """
        计算Whale占比

        Whale地址持币量 / 总流通量
        """
        whale_tiers = ["1000+", "100-1000"]
        whale_balance = sum(
            holder_dist.distribution.get(tier, 0)
            for tier in whale_tiers
        )

        total_balance = sum(holder_dist.distribution.values())

        return whale_balance / total_balance if total_balance > 0 else 0

    def calculate_exchange_pressure(self, flows: List[ExchangeFlow]) -> Dict:
        """
        计算交易所压力

        综合流入流出趋势判断市场压力
        """
        if not flows:
            return {"pressure": "neutral", "score": 0}

        # 计算最近24h vs 最近7天
        recent_24h = flows[-1] if flows else None
        avg_7d = sum(f.net_flow_24h for f in flows[-7:]) / min(len(flows), 7)

        if recent_24h:
            diff = recent_24h.net_flow_24h - avg_7d
            if diff > 0:
                pressure = "selling" if recent_24h.net_flow_24h > 0 else "buying"
                score = min(abs(diff) / avg_7d, 1) if avg_7d != 0 else 0
            else:
                pressure = "buying" if recent_24h.net_flow_24h < 0 else "selling"
                score = min(abs(diff) / abs(avg_7d), 1) if avg_7d != 0 else 0
        else:
            pressure = "neutral"
            score = 0

        return {
            "pressure": pressure,
            "score": score,
            "net_flow_24h": recent_24h.net_flow_24h if recent_24h else 0,
            "avg_7d": avg_7d
        }

    def calculate_market_sentiment_from_onchain(
        self,
        active_addresses_trend: List[float],
        exchange_flow_trend: List[float],
        whale_activity: List[WhaleActivity]
    ) -> Dict:
        """
        从链上数据计算市场情绪

        综合地址活跃度、交易所流量、Whale活动
        """

        # 1. 地址活跃度评分
        if len(active_addresses_trend) >= 7:
            recent_avg = sum(active_addresses_trend[-3:]) / 3
            older_avg = sum(active_addresses_trend[-7:-3]) / 4
            addr_score = (recent_avg - older_avg) / older_avg if older_avg != 0 else 0
        else:
            addr_score = 0

        # 2. 交易所流量评分
        if exchange_flow_trend:
            flow_score = -sum(exchange_flow_trend[-3:]) / 3  # 正流出=负分
        else:
            flow_score = 0

        # 3. Whale活动评分
        whale_score = 0
        for activity in whale_activity[-24:]:  # 最近24小时
            if activity.activity_type == "transfer_out":
                whale_score -= 0.1
            elif activity.activity_type == "transfer_in":
                whale_score += 0.1

        # 综合评分
        total_score = (addr_score * 0.3 + flow_score * 0.4 + whale_score * 0.3)

        # 转换为情绪
        if total_score > 0.2:
            sentiment = "very_bullish"
        elif total_score > 0.05:
            sentiment = "bullish"
        elif total_score < -0.2:
            sentiment = "very_bearish"
        elif total_score < -0.05:
            sentiment = "bearish"
        else:
            sentiment = "neutral"

        return {
            "sentiment": sentiment,
            "score": total_score,
            "breakdown": {
                "address_activity": addr_score,
                "exchange_flow": flow_score,
                "whale_activity": whale_score
            }
        }
```
