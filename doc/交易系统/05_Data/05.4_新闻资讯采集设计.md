# 🏗️ 架构设计

## 1. 数据模型

```python
from dataclasses import dataclass
from datetime import datetime
from typing import List, Optional, Dict
from enum import Enum

class Sentiment(Enum):
    BULLISH = "bullish"
    BEARISH = "bearish"
    NEUTRAL = "neutral"

@dataclass
class NewsArticle:
    """单条新闻"""
    title: str                 # 标题
    content: str              # 内容摘要
    url: str                 # 原文链接
    source: str              # 来源
    published_at: datetime    # 发布时间
    collected_at: datetime   # 采集时间
    sentiment: Optional[Sentiment] = None  # 情绪
    sentiment_confidence: float = 0.0      # 情绪置信度
    topics: List[str] = None              # 话题标签
    related_symbols: List[str] = None     # 关联币种

@dataclass
class NewsEvent:
    """新闻事件（去重后）"""
    event_id: str                      # 事件ID（哈希）
    title: str                         # 代表性标题
    articles: List[NewsArticle]        # 来源文章列表
    primary_sentiment: Sentiment       # 主要情绪
    sentiment_score: float             # 情绪分数 (-1 to 1)
    sentiment_confidence: float        # 置信度
    sources_count: int                # 来源数量
    first_published: datetime         # 首次发布时间
    last_updated: datetime            # 最后更新时间
    topics: List[str]                 # 话题

@dataclass
class AggregatedNews:
    """聚合后的新闻数据"""
    events: List[NewsEvent]           # 新闻事件列表
    overall_sentiment: Sentiment      # 整体情绪
    overall_score: float             # 整体情绪分数
    sentiment_by_topic: Dict[str, float]  # 各话题情绪
    timestamp: datetime               # 聚合时间
```

## 2. 采集器设计

```python
class NewsCollector:
    """新闻采集器（多源）"""

    def __init__(self):
        self.sources = {
            "coindesk_rss": CoinDeskRSSCollector(),
            "cointelegraph_llm": CoinTeleLLMCollector(),
            "jinshi_llm": JinShiLLMCollector(),
            "cryptopanic": CryptoPanicCollector(),
        }
        self.dedup_threshold_hours = 2  # 2小时内相同标题视为同一事件

    async def collect(self) -> AggregatedNews:
        """多源并行采集 + 去重 + 聚合"""

        # 1. 并行采集所有数据源
        tasks = [collector.fetch() for collector in self.sources.values()]
        results = await asyncio.gather(*tasks, return_exceptions=True)

        # 2. 合并所有新闻
        all_articles = []
        for source_name, result in zip(self.sources.keys(), results):
            if isinstance(result, list):
                for article in result:
                    article.source = source_name
                    all_articles.append(article)

        # 3. 去重
        deduped_events = self._deduplicate(all_articles)

        # 4. 情绪分析
        events_with_sentiment = await self._analyze_sentiment(deduped_events)

        # 5. 情绪聚合
        aggregated = self._aggregate_sentiment(events_with_sentiment)

        return aggregated

    def _deduplicate(self, articles: List[NewsArticle]) -> List[NewsEvent]:
        """基于标题相似度的去重"""
        events = []
        processed = set()

        for article in articles:
            # 生成事件ID（标题+时间窗口）
            event_id = self._generate_event_id(article)

            if event_id not in processed:
                # 新事件
                event = NewsEvent(
                    event_id=event_id,
                    title=article.title,
                    articles=[article],
                    primary_sentiment=article.sentiment,
                    sentiment_score=0,
                    sentiment_confidence=0,
                    sources_count=1,
                    first_published=article.published_at,
                    last_updated=article.collected_at,
                    topics=article.topics or []
                )
                events.append(event)
                processed.add(event_id)
            else:
                # 已存在事件，添加文章
                for event in events:
                    if event.event_id == event_id:
                        event.articles.append(article)
                        event.sources_count = len(event.articles)
                        # 更新时间
                        if article.published_at > event.last_updated:
                            event.last_updated = article.published_at
                        break

        return events

    def _generate_event_id(self, article: NewsArticle) -> str:
        """生成事件ID"""
        # 标准化标题（去标点、小写、截取前50字符）
        normalized_title = re.sub(r'[^\w\s]', '', article.title.lower())[:50]
        # 按小时取整
        hour_bucket = article.published_at.replace(minute=0, second=0)
        return f"{normalized_title}_{hour_bucket}"

    def _calculate_title_similarity(self, title1: str, title2: str) -> float:
        """计算标题相似度（简单版）"""
        words1 = set(title1.lower().split())
        words2 = set(title2.lower().split())
        if not words1 or not words2:
            return 0
        return len(words1 & words2) / len(words1 | words2)
```

## 2. 新闻去重

> **无需LLM，使用传统文本相似度算法**

```python
class NewsDeduplicator:
    """新闻去重器（传统算法，无需LLM）"""

    def __init__(self, similarity_threshold: float = 0.8):
        self.similarity_threshold = similarity_threshold

    def deduplicate(self, articles: List[NewsArticle]) -> List[NewsEvent]:
        """去重并生成事件列表"""
        events = []
        event_map = {}

        for article in articles:
            event_id = self._generate_event_id(article)

            if event_id not in event_map:
                event = NewsEvent(
                    event_id=event_id,
                    title=article.title,
                    articles=[article],
                    sources_count=1,
                    first_published=article.published_at,
                    last_updated=article.published_at,
                    topics=article.topics or []
                )
                events.append(event)
                event_map[event_id] = event
            else:
                event_map[event_id].articles.append(article)
                event_map[event_id].sources_count += 1

        return events

    def _generate_event_id(self, article: NewsArticle) -> str:
        """生成事件ID（基于标准化标题+时间桶）"""
        normalized = re.sub(r'[^\w\s]', '', article.title.lower())
        words = set(normalized.split())
        signature = "_".join(sorted(words)[:5])
        hour_bucket = article.published_at.replace(minute=0, second=0)
        return f"{signature}_{hour_bucket}"

    def _jaccard_similarity(self, title1: str, title2: str) -> float:
        """Jaccard相似度"""
        words1 = set(title1.lower().split())
        words2 = set(title2.lower().split())
        if not words1 or not words2:
            return 0
        return len(words1 & words2) / len(words1 | words2)

    def _cosine_similarity(self, title1: str, title2: str) -> float:
        """TF-IDF余弦相似度（可选，更精确但更慢）"""
        vectorizer = TfidfVectorizer()
        tfidf = vectorizer.fit_transform([title1, title2])
        return cosine_similarity(tfidf[0:1], tfidf[1:2])[0][0]
```

## 3. 情绪分析

> **混合方案：长文本用LLM，短文本用词典 + 降级策略降低成本**

### 3.1 LLM情绪分析器

```python
class LLMSentimentAnalyzer:
    """LLM情绪分析器（高精度）"""

    def __init__(self, model: str = "gpt-4o-mini"):
        self.model = model
        self.prompt_template = """
        分析以下加密货币新闻的情绪倾向。

        新闻标题: {title}
        新闻内容: {content}

        请判断:
        1. 情绪: bullish(看涨), bearish(看跌), neutral(中性)
        2. 置信度: 0.0-1.0
        3. 情绪分数: -1.0(极度看跌) 到 1.0(极度看涨)

        以JSON格式输出: {{"sentiment": "...", "confidence": 0.0-1.0, "score": -1.0到1.0}}
        """

    async def analyze(self, article: NewsArticle) -> NewsArticle:
        """分析单条新闻情绪"""
        response = openai.chat.completions.create(
            model=self.model,
            messages=[
                {"role": "system", "content": "你是一个专业的加密货币市场情绪分析师"},
                {"role": "user", "content": self.prompt_template.format(
                    title=article.title,
                    content=article.content[:2000]
                )}
            ],
            response_format={"type": "json_object"}
        )

        result = json.loads(response.choices[0].message.content)

        article.sentiment = Sentiment(result["sentiment"])
        article.sentiment_confidence = result["confidence"]
        if article.sentiment == Sentiment.BULLISH:
            article.sentiment_score = result["score"]
        elif article.sentiment == Sentiment.BEARISH:
            article.sentiment_score = -abs(result["score"])
        else:
            article.sentiment_score = 0

        return article
```

### 3.2 词典情绪分析器（低成本备选）

```python
class LexiconSentimentAnalyzer:
    """基于词典的情绪分析器（免费但精度较低）"""

    BULLISH_WORDS = {
        "bullish": 1.0, "up": 0.8, "gain": 0.7, "rise": 0.7,
        "soar": 1.0, "surge": 0.9, "rally": 0.8, "high": 0.5,
        "all-time": 0.9, "record": 0.7, "breakout": 0.8, "pump": 0.8,
        "bull": 0.7, "moon": 0.9, "hodl": 0.5, "buy": 0.6
    }

    BEARISH_WORDS = {
        "bearish": -1.0, "down": -0.8, "loss": -0.7, "fall": -0.7,
        "crash": -1.0, "plunge": -0.9, "drop": -0.8, "low": -0.5,
        "sell": -0.6, "warning": -0.5, "risk": -0.4, "dump": -0.8,
        "bear": -0.7, "scam": -0.8, "hack": -0.7, "regulation": -0.3
    }

    def analyze(self, text: str) -> tuple:
        """返回 (sentiment, confidence, score)"""
        words = text.lower().split()
        score = 0
        count = 0

        for word in words:
            if word in self.BULLISH_WORDS:
                score += self.BULLISH_WORDS[word]
                count += 1
            elif word in self.BEARISH_WORDS:
                score += self.BEARISH_WORDS[word]
                count += 1

        if count == 0:
            return (Sentiment.NEUTRAL, 0.3, 0)

        avg_score = score / count
        sentiment = Sentiment.BULLISH if avg_score > 0.2 else Sentiment.BEARISH if avg_score < -0.2 else Sentiment.NEUTRAL

        return (sentiment, 0.5, avg_score)
```

### 3.3 混合情绪分析管道

```python
class HybridSentimentAnalyzer:
    """混合情绪分析管道：低成本高效果"""

    def __init__(self, llm_analyzer: LLMSentimentAnalyzer = None,
                 lexicon_analyzer: LexiconSentimentAnalyzer = None):
        self.llm_analyzer = llm_analyzer or LLMSentimentAnalyzer()
        self.lexicon_analyzer = lexicon_analyzer or LexiconSentimentAnalyzer()
        self.llm_threshold = 200  # 超过200字符用LLM

    async def analyze(self, article: NewsArticle) -> NewsArticle:
        """混合分析"""
        content_length = len(article.content)

        if content_length >= self.llm_threshold:
            return await self.llm_analyzer.analyze(article)
        else:
            return self._analyze_fast(article)

    def _analyze_fast(self, article: NewsArticle) -> NewsArticle:
        """快速分析（词典+降级）"""
        text = article.title + " " + article.content
        sentiment, confidence, score = self.lexicon_analyzer.analyze(text)

        article.sentiment = sentiment
        article.sentiment_confidence = confidence * 0.7  # 降级系数
        article.sentiment_score = score

        return article

    async def batch_analyze(self, articles: List[NewsArticle]) -> List[NewsArticle]:
        """批量分析（优先LLM批量降低成本）"""
        llm_articles = [a for a in articles if len(a.content) >= self.llm_threshold]
        fast_articles = [a for a in articles if len(a.content) < self.llm_threshold]

        for article in fast_articles:
            self._analyze_fast(article)

        if llm_articles:
            tasks = [self.llm_analyzer.analyze(a) for a in llm_articles]
            await asyncio.gather(*tasks, return_exceptions=True)

        return articles
```

## 4. 情绪聚合

> **无需LLM，使用加权平均和投票算法**

```python
class SentimentAggregator:
    """情绪聚合器（传统算法，无需LLM）"""

    def aggregate(self, events: List[NewsEvent]) -> AggregatedNews:
        """聚合多事件情绪"""

        if not events:
            return AggregatedNews(
                events=[],
                overall_sentiment=Sentiment.NEUTRAL,
                overall_score=0,
                sentiment_by_topic={},
                timestamp=datetime.now()
            )

        total_weight = sum(e.sentiment_confidence for e in events)
        if total_weight > 0:
            overall_score = sum(
                e.sentiment_score * e.sentiment_confidence
                for e in events
            ) / total_weight
        else:
            overall_score = 0

        # 2. 确定整体情绪
        if overall_score > 0.2:
            overall_sentiment = Sentiment.BULLISH
        elif overall_score < -0.2:
            overall_sentiment = Sentiment.BEARISH
        else:
            overall_sentiment = Sentiment.NEUTRAL

        # 3. 按话题分组情绪
        sentiment_by_topic = self._aggregate_by_topic(events)

        return AggregatedNews(
            events=events,
            overall_sentiment=overall_sentiment,
            overall_score=round(overall_score, 3),
            sentiment_by_topic=sentiment_by_topic,
            timestamp=datetime.now()
        )

    def _aggregate_by_topic(self, events: List[NewsEvent]) -> Dict[str, float]:
        """按话题聚合情绪"""
        topic_scores = {}

        for event in events:
            if event.topics:
                for topic in event.topics:
                    if topic not in topic_scores:
                        topic_scores[topic] = []
                    topic_scores[topic].append(event.sentiment_score)

        return {
            topic: sum(scores) / len(scores)
            for topic, scores in topic_scores.items()
        }
```

class NewsWebhookReceiver:

            user_id=user_id,
            keywords=keywords,
            platforms=platforms,
            subscribed_at=datetime.now()
        )

    async def unsubscribe(self, user_id: int):
        """取消订阅"""
        if user_id in self.subscribers:
            del self.subscribers[user_id]

    async def broadcast(self, article: NewsArticle):
        """广播给订阅用户"""
        for user_id, sub in self.subscribers.items():
            if self._match_subscription(article, sub):
                message = self._format_message(article)
                await self.telegram_pusher.send_message(user_id, message)

    def _match_subscription(self, article: NewsArticle, sub: NewsSubscription) -> bool:
        """匹配订阅条件"""
        if article.source not in sub.platforms:
            return False

        article_text = (article.title + " " + article.content).lower()
        return any(kw.lower() in article_text for kw in sub.keywords)

    def _format_message(self, article: NewsArticle) -> str:
        """格式化推送消息"""
        emoji = self._get_sentiment_emoji(article.sentiment)
        return (
            f"{emoji} <b>{article.source.upper()}</b>\n"
            f"<a href='{article.url}'>{article.title}</a>\n"
            f"⏰ {article.published_at.strftime('%H:%M')}"
        )

    def _get_sentiment_emoji(self, sentiment: Sentiment) -> str:
        if sentiment == Sentiment.BULLISH:
            return "🟢"

## 3. 规则-based快速筛选器

```python
class RuleBasedEventScreener:
    """规则-based事件快速筛选器"""

    def __init__(self):
        self.critical_keywords = [
            "sec", "cfte", "regulation", "hack", "exploit",
            "crash", "collapse", "ban", "war", "sanction",
            "emergency", "urgent", "breaking"
        ]

        self.authority_sources = {
            "reuters": 1.0,
            "bloomberg": 0.95,
            "coindesk": 0.9,
            "cointelegraph": 0.85,
            "decrypt": 0.8,
            "twitter": 0.3,
            "reddit": 0.2
        }

        self.min_sources_for_urgent = 2
        self.time_window_minutes = 30

    def should_escalate_to_llm(
        self,
        article: NewsArticle,
        recent_events: List[BlackSwanAnalysis] = None
    ) -> tuple:
        """
        判断是否需要升级到LLM深度分析
        返回: (should_escalate, reason, priority_score)
        """

        priority_score = 0
        reasons = []

        # 1. 关键词命中检查
        keyword_hit = self._check_critical_keywords(article)
        if keyword_hit:
            priority_score += 0.4
            reasons.append(f"关键词:{keyword_hit}")

        # 2. 来源权威性检查
        authority = self.authority_sources.get(article.source.lower(), 0.5)
        if authority >= 0.8:
            priority_score += 0.2
            reasons.append(f"权威来源:{article.source}")

        # 3. 多源确认检查
        if recent_events:
            similar_events = self._find_similar_events(article, recent_events)
            if similar_events:
                priority_score += 0.3
                reasons.append(f"多源确认:{len(similar_events)+1}个源")

        # 4. 时间敏感性检查 (近期多次更新)
        if recent_events:
            recent_count = self._count_recent_events(article, recent_events)
            if recent_count >= 3:
                priority_score += 0.1
                reasons.append(f"高频更新:{recent_count}次")

        # 决策阈值
        should_escalate = priority_score >= 0.5

        return should_escalate, "; ".join(reasons), priority_score

    def _check_critical_keywords(self, article: NewsArticle) -> Optional[str]:
        """检查关键关键词"""
        text = (article.title + " " + article.content).lower()

        for keyword in self.critical_keywords:
            if keyword in text:
                return keyword

        return None

    def _find_similar_events(
        self,
        article: NewsArticle,
        recent_events: List[BlackSwanAnalysis]
    ) -> List[BlackSwanAnalysis]:
        """查找相似事件"""
        similar = []
        article_words = set(article.title.lower().split())

        for event in recent_events:
            if event.first_detected:
                time_diff = datetime.now() - event.first_detected
                if time_diff.total_seconds() > self.time_window_minutes * 60:
                    continue

            event_words = set(event.event_summary.lower().split())
            overlap = len(article_words & event_words)

            if overlap >= 3:
                similar.append(event)

        return similar

    def _count_recent_events(
        self,
        article: NewsArticle,
        recent_events: List[BlackSwanAnalysis]
    ) -> int:
        """统计近期相似事件数量"""
        similar = self._find_similar_events(article, recent_events)
        return len(similar)
```

class ComprehensiveBLACKSWANDetector:

            return existing_event

        # 5. 保存新事件
        self.recent_events.append(llm_result)
        self._cleanup_old_events()

        return llm_result

    async def analyze_batch(
        self,
        articles: List[NewsArticle]
    ) -> List[BlackSwanAnalysis]:
        """批量分析"""

        # 1. 先规则筛选
        candidate_articles = []
        for article in articles:
            should, reason, priority = self.rule_screener.should_escalate_to_llm(
                article, self.recent_events
            )
            if should:
                candidate_articles.append((article, priority))

        # 2. 按优先级排序
        candidate_articles.sort(key=lambda x: x[1], reverse=True)

        # 3. 批量LLM分析(限制并发)
        results = []
        for article, _ in candidate_articles[:10]:  # 最多10条
            result = await self.analyze_article(article)
            if result:
                results.append(result)

        return results

    def _count_related_sources(self, article: NewsArticle) -> int:
        """统计相关来源数量"""
        count = 1
        article_words = set(article.title.lower().split())

        for event in self.recent_events:
            if event.first_detected:
                time_diff = datetime.now() - event.first_detected
                if time_diff.total_seconds() > 2 * 3600:  # 2小时内
                    continue

            event_words = set(event.event_summary.lower().split())
            overlap = len(article_words & event_words)

            if overlap >= 3:
                count += 1

        return count

    def _find_existing_event(
        self,
        new_result: BlackSwanAnalysis
    ) -> Optional[BlackSwanAnalysis]:
        """查找已存在的事件"""
        for event in self.recent_events:
            if event.event_type == new_result.event_type:
                time_diff = datetime.now() - event.first_detected
                if time_diff.total_seconds() < self.event_ttl_hours * 3600:
                    if self._events_overlap(event, new_result):
                        return event
        return None

    def _events_overlap(
        self,
        event1: BlackSwanAnalysis,
        event2: BlackSwanAnalysis
    ) -> bool:
        """判断两个事件是否重叠"""
        words1 = set(event1.event_summary.lower().split())
        words2 = set(event2.event_summary.lower().split())
        overlap = len(words1 & words2)
        return overlap >= 5

    def _merge_events(
        self,
        existing: BlackSwanAnalysis,
        new_result: BlackSwanAnalysis
    ):
        """合并事件"""
        existing.source_articles.extend(new_result.source_articles)
        existing.source_count = len(existing.source_articles)
        existing.last_updated = datetime.now()

        if new_result.black_swan_score > existing.black_swan_score:
            existing.black_swan_score = new_result.black_swan_score

    def _cleanup_old_events(self):
        """清理过期事件"""
        cutoff = datetime.now() - timedelta(hours=self.event_ttl_hours)
        self.recent_events = [
            e for e in self.recent_events
            if e.first_detected and e.first_detected > cutoff
        ]
