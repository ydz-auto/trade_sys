# 📊 为什么不是主备切换

| 观点 | 说明 |
|------|------|
| ❌ 主备切换 | 主挂了才用备，只用1个价格，丢失其他信息 |
| ✅ 多源并行 | 同时获取所有交易所价格，不丢失任何信息 |
| ✅ 保留价差 | 可用于套利分析、价差监控 |
| ✅ 故障无关 | 某交易所故障不影响其他交易所数据 |

# 🏗️ 架构设计

## 1. 数据模型

```python
from dataclasses import dataclass
from datetime import datetime
from typing import Dict, Optional

@dataclass
class ExchangePrice:
    """单个交易所的价格数据"""
    exchange: str              # "binance", "okx", "coinbase", "gate"
    symbol: str               # "BTC/USDT", "ETH/USDT"
    price: float             # 最新成交价
    bid: float               # 买一价
    ask: float               # 卖一价
    spread: float            # 买卖价差 (ask - bid)
    spread_percent: float   # 价差百分比
    volume_24h: float       # 24小时成交量
    timestamp: datetime      # 数据时间戳
    status: str              # "ok", "stale", "error"

@dataclass
class MultiExchangePrices:
    """多交易所价格（分别保存，不融合）"""
    symbol: str
    prices: Dict[str, ExchangePrice]  # 各交易所原始价格
    timestamp: datetime                # 采集时间

    @property
    def all_ok(self) -> bool:
        """所有交易所是否正常"""
        return all(p.status == "ok" for p in self.prices.values())

    @property
    def available_exchanges(self) -> list:
        """可用的交易所列表"""
        return [name for name, p in self.prices.items() if p.status == "ok"]

    @property
    def max_price_info(self) -> tuple:
        """最高价交易所"""
        valid = [(n, p) for n, p in self.prices.items() if p.status == "ok"]
        return max(valid, key=lambda x: x[1].price) if valid else (None, None)

    @property
    def min_price_info(self) -> tuple:
        """最低价交易所"""
        valid = [(n, p) for n, p in self.prices.items() if p.status == "ok"]
        return min(valid, key=lambda x: x[1].price) if valid else (None, None)

    @property
    def arbitrage_opportunity(self) -> Optional[Dict]:
        """套利机会检测"""
        min_name, min_price = self.min_price_info
        max_name, max_price = self.max_price_info

        if min_name is None or max_name is None:
            return None

        diff = max_price.price - min_price.price
        diff_percent = diff / min_price.price * 100

        if diff_percent > 0.005:  # 超过0.005%视为套利机会
            return {
                "buy_exchange": min_name,
                "sell_exchange": max_name,
                "buy_price": min_price.price,
                "sell_price": max_price.price,
                "diff": diff,
                "diff_percent": diff_percent
            }
        return None
```

## 2. 采集器设计

```python
import asyncio
from typing import Dict, List

class PriceCollector:
    """价格采集器（多源并行，分别保存）"""

    def __init__(self, exchanges: List[str] = None):
        self.exchanges = exchanges or ["binance", "okx", "coinbase", "gate"]
        self.collectors = self._init_collectors()

    def _init_collectors(self) -> Dict:
        return {
            "binance": BinancePriceCollector(),
            "okx": OKXPriceCollector(),
            "coinbase": CoinbasePriceCollector(),
            "gate": GatePriceCollector(),
        }

    async def collect(self, symbol: str) -> MultiExchangePrices:
        """多源并行采集所有交易所价格"""
        tasks = []
        for exchange in self.exchanges:
            collector = self.collectors.get(exchange)
            if collector:
                tasks.append(self._collect_one(exchange, collector, symbol))

        results = await asyncio.gather(*tasks, return_exceptions=True)

        prices = {}
        for exchange, result in zip(self.exchanges, results):
            if isinstance(result, ExchangePrice):
                prices[exchange] = result
            elif isinstance(result, Exception):
                prices[exchange] = ExchangePrice(
                    exchange=exchange,
                    symbol=symbol,
                    price=0, bid=0, ask=0,
                    spread=0, spread_percent=0,
                    volume_24h=0,
                    timestamp=datetime.now(),
                    status="error"
                )
                await self._log_error(exchange, result)

        return MultiExchangePrices(
            symbol=symbol,
            prices=prices,
            timestamp=datetime.now()
        )

    async def _collect_one(self, exchange: str, collector, symbol: str) -> ExchangePrice:
        """从单个交易所获取价格"""
        return await collector.fetch(symbol)

    async def _log_error(self, exchange: str, error: Exception):
        """记录错误日志"""
        logger.error(f"{exchange} 价格采集失败: {error}")
```

## 3. 交易时选择价格

```python
class TradingPriceSelector:
    """交易时按交易所选择价格"""

    def __init__(self, collector: PriceCollector):
        self.collector = collector
        self.price_cache: Dict[str, MultiExchangePrices] = {}

    async def get_price_for_trading(
        self,
        symbol: str,
        target_exchange: str
    ) -> ExchangePrice:
        """
        获取交易用的价格

        Args:
            symbol: 交易对，如 "BTC/USDT"
            target_exchange: 目标交易所，如 "binance"

        Returns:
            该交易所的价格数据
        """
        # 1. 获取最新价格
        if symbol not in self.price_cache:
            self.price_cache[symbol] = await self.collector.collect(symbol)

        prices = self.price_cache[symbol]

        # 2. 直接返回目标交易所价格
        if target_exchange in prices.prices:
            price = prices.prices[target_exchange]
            if price.status == "ok":
                return price
            else:
                raise ExchangeError(f"{target_exchange} 价格状态异常")

        raise ExchangeNotSupportedError(f"不支持的交易所: {target_exchange}")

    async def execute_trade(
        self,
        symbol: str,
        exchange: str,
        side: str,  # "buy" or "sell"
        amount: float
    ):
        """执行交易"""
        price = await self.get_price_for_trading(symbol, exchange)

        if side == "buy":
            execution_price = price.ask  # 买入用卖一价
        else:
            execution_price = price.bid   # 卖出用买一价

        return await self._place_order(
            exchange=exchange,
            symbol=symbol,
            side=side,
            price=execution_price,
            amount=amount
        )
```

# 🛡️ 质量保障

## 1. 健康检查

```python
class PriceHealthChecker:
    """价格健康检查"""

    def check_staleness(self, price: ExchangePrice, max_age_seconds: int = 5) -> bool:
        """检查数据是否过期"""
        age = (datetime.now() - price.timestamp).total_seconds()
        return age > max_age_seconds

    def check_spread(self, price: ExchangePrice, max_spread_percent: float = 0.1) -> bool:
        """检查买卖价差是否过大"""
        return price.spread_percent > max_spread_percent

    def check_volume(self, price: ExchangePrice, min_volume: float = 1000) -> bool:
        """检查成交量是否异常低"""
        return price.volume_24h < min_volume

    def get_health_status(self, price: ExchangePrice) -> str:
        """获取健康状态"""
        if price.status == "error":
            return "error"
        if self.check_staleness(price):
            return "stale"
        if self.check_spread(price):
            return "high_spread"
        return "ok"
```
