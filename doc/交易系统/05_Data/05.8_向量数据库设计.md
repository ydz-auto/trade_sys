from dataclasses import dataclass, field
from datetime import datetime
from typing import List, Optional, Dict, Any
import numpy as np

@dataclass
class HistoricalFactor:
    """Historical因子基础模型"""
    factor_id: str                          # 因子唯一ID
    symbol: str                            # 交易对: "BTC/USDT"
    exchange: str                          # 交易所: "binance"
    factor_type: str                       # 因子类型: "price", "onchain", "macro"

    # 时间信息
    timestamp: datetime                    # 数据时间点
    window_start: datetime                 # 窗口起始
    window_end: datetime                   # 窗口结束

    # 原始数值
    raw_values: Dict[str, float]           # 原始因子值

    # 向量化表示
    vector: np.ndarray                     # 向量化表示
    vector_dim: int                        # 向量维度

    # 统计特征
    stats: Dict[str, float] = field(default_factory=dict)
    # {
    #     "mean": 0.05,
    #     "std": 0.02,
    #     "min": -0.1,
    #     "max": 0.2,
    #     "skewness": 0.5,
    #     "kurtosis": 3.0
    # }

    # 模式编码
    pattern_code: str = ""                 # 二值化模式字符串

    # 元数据
    metadata: Dict[str, Any] = field(default_factory=dict)
    # {
    #     "source": "binance",
    #     "confidence": 0.95,
    #     "is_estimated": False,
    #     "gaps": []
    # }


@dataclass
class FactorTimeSeries:
    """因子时间序列"""
    symbol: str
    factor_type: str
    start_time: datetime
    end_time: datetime
    interval: str                          # "1h", "4h", "1d"
    factors: List[HistoricalFactor]

    @property
    def price_change(self) -> float:
        """区间价格变化"""
        if len(self.factors) < 2:
            return 0.0
        return (self.factors[-1].raw_values.get("close", 0) /
                self.factors[0].raw_values.get("close", 1) - 1)

    @property
    def volatility(self) -> float:
        """区间波动率"""
        return self.stats.get("volatility", 0.0) if self.factors else 0.0


@dataclass
class SimilarPattern:
    """相似模式结果"""
    pattern_id: str
    similarity: float                      # 0-1, 越高越相似
    time_range: str                        # "2024-01-01 ~ 2024-01-07"
    price_change: float                    # 当时的价格变化
    outcome: str                            # "bull", "bear", "sideways"
    confidence: float                      # 置信度
    matching_factors: List[str]            # 匹配的因子列表

# 🔧 向量化方法

## 1. 时序因子向量化

```python
import numpy as np
from typing import List, Dict
from dataclasses import dataclass

class TimeSeriesVectorizer:
    """时序因子向量化器"""

    def __init__(self, window_size: int = 24, vector_dim: int = 128):
        self.window_size = window_size          # 时间窗口 (小时)
        self.vector_dim = vector_dim

    def vectorize(self, factor_series: FactorTimeSeries) -> np.ndarray:
        """
        将时序因子转换为向量

        向量构成 (128维):
        - 前32维: 收益率统计 (均值, 标准差, 偏度, 峰度, 最大, 最小, ...分位数)
        - 中间64维: 模式编码 (归一化后的价格路径)
        - 后32维: 关系特征 (与BTC相关性, 与市场相关性, 波动率特征)
        """
        vectors = []

        # 1. 收益率统计特征 (32维)
        stats_vec = self._compute_stats_features(factor_series)
        vectors.append(stats_vec)

        # 2. 模式编码 (64维)
        pattern_vec = self._compute_pattern_features(factor_series)
        vectors.append(pattern_vec)

        # 3. 关系特征 (32维)
        relation_vec = self._compute_relation_features(factor_series)
        vectors.append(relation_vec)

        return np.concatenate(vectors)

    def _compute_stats_features(self, series: FactorTimeSeries) -> np.ndarray:
        """计算统计特征"""
        prices = [f.raw_values.get("close", 0) for f in series.factors]
        returns = np.diff(prices) / prices[:-1] if len(prices) > 1 else [0]

        if len(returns) == 0:
            return np.zeros(32)

        returns = np.array(returns)

        features = [
            np.mean(returns),                    # 均值
            np.std(returns),                     # 标准差
            np.mean(returns[:len(returns)//3]), # 前1/3均值
            np.mean(returns[len(returns)*2//3:]),# 后1/3均值
            np.max(returns),                     # 最大收益
            np.min(returns),                     # 最小收益
            np.sum(returns > 0) / len(returns), # 正收益比例
            np.sum(returns < 0) / len(returns),  # 负收益比例
            np.percentile(returns, 25),          # 25分位
            np.percentile(returns, 50),          # 50分位
            np.percentile(returns, 75),          # 75分位
        ]

        # 计算更多统计量填满32维
        additional = []
        for i in range(21):
            if i < len(returns):
                additional.append(returns[i] if i < len(returns) else 0)
            else:
                additional.append(0)

        return np.array(features + additional[:21])

    def _compute_pattern_features(self, series: FactorTimeSeries) -> np.ndarray:
        """
        计算模式编码特征 (64维)
        归一化价格路径，分段编码
        """
        prices = [f.raw_values.get("close", 0) for f in series.factors]

        if len(prices) < 2:
            return np.zeros(64)

        # 归一化到 [0, 1]
        min_p, max_p = min(prices), max(prices)
        if max_p - min_p > 0:
            normalized = [(p - min_p) / (max_p - min_p) for p in prices]
        else:
            normalized = [0.5] * len(prices)

        # 分段编码 (64段)
        pattern = np.zeros(64)
        segment_size = len(normalized) / 64

        for i in range(64):
            start_idx = int(i * segment_size)
            end_idx = int((i + 1) * segment_size)
            if start_idx < len(normalized):
                pattern[i] = np.mean(normalized[start_idx:min(end_idx, len(normalized))])

        return pattern

    def _compute_relation_features(self, series: FactorTimeSeries) -> np.ndarray:
        """计算关系特征 (32维)"""
        # 这里应该关联其他相关币种的数据计算相关性
        # 简化版: 使用波动率和成交量关系
        features = []

        volumes = [f.raw_values.get("volume", 0) for f in series.factors]
        prices = [f.raw_values.get("close", 0) for f in series.factors]

        if len(volumes) > 1 and len(prices) > 1:
            vol_changes = np.diff(volumes) / np.array(volumes[:-1]) if volumes[0] > 0 else [0]
            price_changes = np.diff(prices) / np.array(prices[:-1]) if prices[0] > 0 else [0]

            features.extend([
                np.corrcoef(vol_changes, price_changes)[0, 1] if len(vol_changes) > 1 else 0,
                np.std(vol_changes),
                np.mean(vol_changes),
            ])
        else:
            features = [0, 0, 0]

        # 填充到32维
        while len(features) < 32:
            features.append(0)

        return np.array(features[:32])


class OnChainVectorizer:
    """链上因子向量化器"""

    def vectorize(self, onchain_data: Dict) -> np.ndarray:
        """
        链上因子向量 (128维)
        - 持仓分布 (32维)
        - Gas使用 (32维)
        - 活跃度 (32维)
        - 链上行为 (32维)
        """
        vectors = []

        # 1. 持仓分布 (32维)
        holder_dist = self._encode_holder_distribution(onchain_data.get("holder_distribution", {}))
        vectors.append(holder_dist)

        # 2. Gas使用 (32维)
        gas_features = self._encode_gas_usage(onchain_data.get("gas_usage", {}))
        vectors.append(gas_features)

        # 3. 活跃度 (32维)
        activity = self._encode_activity(onchain_data.get("activity", {}))
        vectors.append(activity)

        # 4. 链上行为 (32维)
        behavior = self._encode_behavior(onchain_data.get("behavior", {}))
        vectors.append(behavior)

        return np.concatenate(vectors)

    def _encode_holder_distribution(self, holder_data: Dict) -> np.ndarray:
        """持仓分布编码"""
        # 归一化各层级持仓比例
        tiers = holder_data.get("tiers", [0] * 10)
        return np.array(tiers[:32] + [0] * max(0, 32 - len(tiers)))

    def _encode_gas_usage(self, gas_data: Dict) -> np.ndarray:
        """Gas使用编码"""
        features = [
            gas_data.get("avg_gas_price", 0),
            gas_data.get("max_gas_price", 0),
            gas_data.get("min_gas_price", 0),
            gas_data.get("gas_used", 0),
        ]
        return np.array(features[:32] + [0] * max(0, 32 - len(features)))

    def _encode_activity(self, activity_data: Dict) -> np.ndarray:
        """活跃度编码"""
        features = [
            activity_data.get("active_addresses", 0),
            activity_data.get("new_addresses", 0),
            activity_data.get("tx_count", 0),
            activity_data.get("volume", 0),
        ]
        return np.array(features[:32] + [0] * max(0, 32 - len(features)))

    def _encode_behavior(self, behavior_data: Dict) -> np.ndarray:
        """链上行为编码"""
        features = [
            behavior_data.get("exchange_inflow", 0),
            behavior_data.get("exchange_outflow", 0),
            behavior_data.get(" Whale_tx_count", 0),
            behavior_data.get("defi_volume", 0),
        ]
        return np.array(features[:32] + [0] * max(0, 32 - len(features)))
```

## 2. 模式编码 (二值化/分桶)

```python
class PatternEncoder:
    """模式编码器 - 将连续值转换为离散模式"""

    # 预定义模式类型
    PATTERN_TYPES = {
        "price": {
            "buckets": [-np.inf, -0.05, -0.02, 0.02, 0.05, np.inf],
            "labels": ["crash", "drop", "stable", "rise", "surge"]
        },
        "volatility": {
            "buckets": [0, 0.01, 0.03, 0.05, np.inf],
            "labels": ["very_low", "low", "medium", "high"]
        },
        "volume": {
            "buckets": [0, 0.5, 1.0, 2.0, np.inf],
            "labels": ["very_low", "low", "normal", "high"]
        },
        "sentiment": {
            "buckets": [-1, -0.5, -0.1, 0.1, 0.5, 1],
            "labels": ["very_bearish", "bearish", "neutral", "bullish", "very_bullish"]
        }
    }

    def encode(self, factor_type: str, value: float) -> str:
        """单值编码"""
        if factor_type not in self.PATTERN_TYPES:
            return "unknown"

        bucket_config = self.PATTERN_TYPES[factor_type]
        buckets = bucket_config["buckets"]
        labels = bucket_config["labels"]

        for i in range(len(buckets) - 1):
            if buckets[i] <= value < buckets[i + 1]:
                return labels[i]

        return labels[-1]

    def encode_series(self, factor_type: str, values: List[float]) -> str:
        """序列编码 - 返回模式字符串"""
        pattern_codes = [self.encode(factor_type, v) for v in values]
        return "".join([c[0].upper() for c in pattern_codes])

    def compute_pattern_similarity(self, pattern1: str, pattern2: str) -> float:
        """计算两个模式的相似度 (编辑距离)"""
        if len(pattern1) != len(pattern2):
            return 0.0

        matches = sum(1 for a, b in zip(pattern1, pattern2) if a == b)
        return matches / len(pattern1)
```

## 1. 相似模式检索

```python
from qdrant_client import QdrantClient
from qdrant_client.models import Filter, FieldCondition, MatchAny, Range

class HistoricalFactorRetriever:
    """Historical因子检索器"""

    def __init__(self, client: QdrantClient, collection_name: str = "historical_factors"):
        self.client = client
        self.collection_name = collection_name

    async def find_similar_patterns(
        self,
        query_vector: np.ndarray,
        symbol: str,
        limit: int = 10,
        time_range_days: int = 365,
        filters: Dict = None
    ) -> List[SimilarPattern]:
        """
        查找相似历史模式

        Args:
            query_vector: 查询向量
            symbol: 交易对
            limit: 返回数量
            time_range_days: 时间范围 (天)
            filters: 额外过滤条件
        """

        # 构建过滤条件
        must_conditions = [
            FieldCondition(key="symbol", match=MatchAny(any=[symbol])),
            FieldCondition(
                key="timestamp",
                range=Range(
                    gte=datetime.now() - timedelta(days=time_range_days),
                    lte=datetime.now()
                )
            )
        ]

        # 添加额外过滤
        if filters:
            for key, value in filters.items():
                if isinstance(value, list):
                    must_conditions.append(
                        FieldCondition(key=key, match=MatchAny(any=value))
                    )
                else:
                    must_conditions.append(
                        FieldCondition(key=key, match=value)
                    )

        search_result = self.client.search(
            collection_name=self.collection_name,
            query_vector=query_vector.tolist(),
            query_filter=Filter(must=must_conditions),
            limit=limit,
            with_payload=True
        )

        # 转换为SimilarPattern
        results = []
        for hit in search_result:
            payload = hit.payload

            results.append(SimilarPattern(
                pattern_id=payload["factor_id"],
                similarity=hit.score,
                time_range=f"{payload['window_start']} ~ {payload['window_end']}",
                price_change=payload.get("price_change", 0),
                outcome=self._classify_outcome(payload.get("price_change", 0)),
                confidence=payload.get("confidence", 1.0),
                matching_factors=payload.get("factor_types", [])
            ))

        return results

    async def find_similar_with_outcome(
        self,
        query_vector: np.ndarray,
        symbol: str,
        target_outcome: str,               # "bull", "bear"
        min_change: float = 0.05,          # 最小变化幅度
        limit: int = 5
    ) -> List[SimilarPattern]:
        """查找具有特定结果的历史相似模式"""

        filters = {
            "symbol": [symbol],
            "price_change": min_change if target_outcome == "bull" else -min_change
        }

        all_similar = await self.find_similar_patterns(
            query_vector=query_vector,
            symbol=symbol,
            limit=100,
            filters=filters
        )

        # 过滤符合outcome的
        filtered = [p for p in all_similar if p.outcome == target_outcome][:limit]

        return filtered

    def _classify_outcome(self, price_change: float) -> str:
        """分类结果"""
        if price_change > 0.05:
            return "bull"
        elif price_change < -0.05:
            return "bear"
        else:
            return "sideways"

    async def analyze_pattern_success_rate(
        self,
        query_vector: np.ndarray,
        symbol: str,
        top_k: int = 20
    ) -> Dict:
        """分析相似模式的成功率"""

        similar_patterns = await self.find_similar_patterns(
            query_vector=query_vector,
            symbol=symbol,
            limit=top_k
        )

        if not similar_patterns:
            return {"sample_size": 0, "bull_rate": 0, "bear_rate": 0}

        bull_count = sum(1 for p in similar_patterns if p.outcome == "bull")
        bear_count = sum(1 for p in similar_patterns if p.outcome == "bear")
        sideways_count = sum(1 for p in similar_patterns if p.outcome == "sideways")

        # 加权计算 (根据相似度)
        total_weight = sum(p.similarity for p in similar_patterns)
        bull_weight = sum(p.similarity for p in similar_patterns if p.outcome == "bull")
        bear_weight = sum(p.similarity for p in similar_patterns if p.outcome == "bear")

        return {
            "sample_size": len(similar_patterns),
            "bull_count": bull_count,
            "bear_count": bear_count,
            "sideways_count": sideways_count,
            "bull_rate": bull_count / len(similar_patterns),
            "bear_rate": bear_count / len(similar_patterns),
            "weighted_bull_rate": bull_weight / total_weight if total_weight > 0 else 0,
            "weighted_bear_rate": bear_weight / total_weight if total_weight > 0 else 0,
            "patterns": similar_patterns
        }
```

            "market_regime": "bull",
            "has_etf_event": True,
            "price_change_range": (-0.1, 0.1),
            "volatility_range": (0.01, 0.05),
            "time_range": (datetime(2024,1,1), datetime(2024,12,31))
        }
        """

        must_conditions = []

        for key, value in conditions.items():
            if key == "time_range":
                must_conditions.append(
                    FieldCondition(
                        key="timestamp",
                        range=Range(gte=value[0], lte=value[1])
                    )
                )
            elif key == "price_change_range":
                must_conditions.append(
                    FieldCondition(
                        key="price_change",
                        range=Range(gte=value[0], lte=value[1])
                    )
                )
            elif key == "volatility_range":
                must_conditions.append(
                    FieldCondition(
                        key="volatility",
                        range=Range(gte=value[0], lte=value[1])
                    )
                )
            elif isinstance(value, list):
                must_conditions.append(
                    FieldCondition(key=key, match=MatchAny(any=value))
                )
            elif isinstance(value, bool):
                must_conditions.append(
                    FieldCondition(key=key, match=value)
                )
            else:
                must_conditions.append(
                    FieldCondition(key=key, match=value)
                )

        results = self.client.search(
            collection_name=self.collection_name,
            query_vector=query_vector.tolist(),
            query_filter=Filter(must=must_conditions),
            limit=limit,
            with_payload=True
        )

        return [hit.payload for hit in results]

    async def get_market_regime_factors(
        self,
        symbol: str,
        regime: str,
        limit: int = 100
    ) -> List[Dict]:
        """获取特定市场状态的历史因子"""
        return await self.search_with_conditions(
            query_vector=np.zeros(128),  # 不需要向量检索
            conditions={
                "symbol": symbol,
                "market_regime": regime
            },
            limit=limit
        )

class FactorDataLifecycleManager:
    """因子数据生命周期管理器"""

    LIFECYCLE_CONFIG = {
        "hot": {
            "storage": "qdrant",
            "retention_days": 7,
            "index_type": "HNSW"
        },
        "warm": {
            "storage": "postgresql",
            "retention_days": 90,
            "compression": "pg_compression"
        },
        "cold": {
            "storage": "s3",
            "retention_days": 365 * 2,
            "format": "parquet"
        }
    }

    async def migrate_data(self):
        """数据迁移: 热 -> 温 -> 冷"""

        # 7天前: Qdrant -> PostgreSQL
        await self._hot_to_warm()

        # 90天前: PostgreSQL -> S3
        await self._warm_to_cold()

    async def _hot_to_warm(self):
        """热数据转温数据"""
        cutoff = datetime.now() - timedelta(days=7)

        # 导出向量
        vectors = self.qdrant.export_range(
            collection="historical_factors",
            filter={"timestamp": {"lt": cutoff.isoformat()}}
        )

        # 存入PostgreSQL
        await self.pg.bulk_insert("factor_vectors_warm", vectors)

        # 删除Qdrant中的旧数据
        self.qdrant.delete(
            collection="historical_factors",
            filter={"timestamp": {"lt": cutoff.isoformat()}}
        )

    async def _warm_to_cold(self):
        """温数据转冷数据"""
        cutoff = datetime.now() - timedelta(days=90)

        # 导出
        vectors = await self.pg.export_range(
            table="factor_vectors_warm",
            filter={"timestamp": {"lt": cutoff.isoformat()}}
        )

        # 转Parquet存S3
        df = pd.DataFrame(vectors)
        s3_path = f"s3://bucket/factors/{cutoff.year}/{cutoff.month}/"
        await self.s3.write_parquet(df, s3_path)

        # 删除PG中的旧数据
        await self.pg.delete(
            table="factor_vectors_warm",
            filter={"timestamp": {"lt": cutoff.isoformat()}}
        )

class VectorFactorStore:
    """向量因子存储器 - 与因子系统对接"""

    def __init__(self, retriever: HistoricalFactorRetriever):
        self.retriever = retriever
        self.vectorizer = TimeSeriesVectorizer()

    async def store_factor(
        self,
        factor_data: Dict,
        factor_type: str
    ) -> str:
        """存储因子到向量数据库"""

        # 1. 创建HistoricalFactor对象
        factor = HistoricalFactor(
            factor_id=f"{factor_data['symbol']}_{factor_data['timestamp']}_{factor_type}",
            symbol=factor_data["symbol"],
            exchange=factor_data.get("exchange", "binance"),
            factor_type=factor_type,
            timestamp=factor_data["timestamp"],
            window_start=factor_data.get("window_start", factor_data["timestamp"]),
            window_end=factor_data.get("window_end", factor_data["timestamp"]),
            raw_values=factor_data["values"],
            stats=self._compute_stats(factor_data["values"]),
            metadata=factor_data.get("metadata", {})
        )

        # 2. 向量化
        if factor_type == "price":
            series = self._to_time_series(factor)
            factor.vector = self.vectorizer.vectorize(series)
        else:
            factor.vector = np.random.randn(128)  # 占位

        factor.vector_dim = len(factor.vector)

        # 3. 存储
        await self._store_to_qdrant(factor)

        return factor.factor_id

    async def retrieve_similar(
        self,
        current_factors: Dict,
        symbol: str,
        outcome_filter: str = None
    ) -> Dict:
        """检索相似历史"""

        # 向量化当前因子
        series = self._to_time_series(current_factors)
        query_vector = self.vectorizer.vectorize(series)

        # 检索相似模式
        if outcome_filter:
            similar = await self.retriever.find_similar_with_outcome(
                query_vector=query_vector,
                symbol=symbol,
                target_outcome=outcome_filter
            )
        else:
            similar = await self.retriever.find_similar_patterns(
                query_vector=query_vector,
                symbol=symbol,
                limit=10
            )

        # 分析成功率
        analysis = await self.retriever.analyze_pattern_success_rate(
            query_vector=query_vector,
            symbol=symbol
        )

        return {
            "similar_patterns": similar,
            "analysis": analysis,
            "query_vector": query_vector.tolist()
        }
