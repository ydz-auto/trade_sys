# 6.2 错误日志记录

```python
import traceback
import sys

def log_error(error: Exception, context: dict = None):
    exc_type, exc_value, exc_tb = sys.exc_info()

    log_entry = {
        "timestamp": datetime.utcnow().isoformat(),
        "level": "ERROR",
        "logger": context.get("module", "unknown") if context else "unknown",
        "message": str(error),
        "error_id": f"err_{uuid.uuid4().hex[:12]}",
        "error_type": exc_type.__name__ if exc_type else "UnknownError",
        "stack_trace": traceback.format_exception(exc_type, exc_value, exc_tb),
        "module": context.get("module") if context else None,
        "function": context.get("function") if context else None,
        "request_id": context.get("request_id") if context else None
    }
    logger.error("error", log_entry)
```

# 9.2 数据库存储

```sql
CREATE TABLE logs (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    log_type VARCHAR(20) NOT NULL,
    timestamp TIMESTAMP NOT NULL,
    level VARCHAR(10) NOT NULL,
    message TEXT,
    payload JSON,
    INDEX idx_log_type_timestamp (log_type, timestamp),
    INDEX idx_level (level),
    INDEX idx_timestamp (timestamp)
);
```

# 🔧 10. 日志收集与处理

# 10.1 日志收集器

```python
class LogCollector:
    def __init__(self):
        self.handlers = {
            "file": FileHandler(),
            "database": DatabaseHandler(),
            "elasticsearch": ESHandler()
        }

    def collect(self, log_type: str, log_entry: dict):
        for handler in self.handlers.values():
            handler.send(log_type, log_entry)
```

# 10.2 日志轮转

```python
from logging.handlers import TimedRotatingFileHandler

file_handler = TimedRotatingFileHandler(

    when="midnight",
    interval=1,
```
