## 1.2 数据库选型依据

| 数据库 | 场景 | 原因 |
|--------|------|------|
| **PostgreSQL** | 主数据库 | 事务可靠、结构复杂数据 |
| **PostgreSQL** | 交易数据 | 高并发写入、简单查询 |
| **ClickHouse** | 时序数据 | 高效OLAP、压缩存储 |
| **Redis** | 缓存 | 内存存储、低延迟 |
| **InfluxDB** | 监控指标 | 时序数据、聚合查询 |
| **MinIO** | 文件存储 | S3兼容、分布式 |

# 🗃️ 二、PostgreSQL 表设计

## 2.1 用户相关表

```sql
-- 用户表
CREATE TABLE users (
    id              BIGSERIAL PRIMARY KEY,
    username        VARCHAR(50) UNIQUE NOT NULL,
    email           VARCHAR(100) UNIQUE NOT NULL,
    password_hash   VARCHAR(255) NOT NULL,
    status          VARCHAR(20) DEFAULT 'active',
    risk_level      VARCHAR(20) DEFAULT 'normal',
    max_position_pct DECIMAL(5,2) DEFAULT 10.00,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_users_username ON users(username);
CREATE INDEX idx_users_email ON users(email);
CREATE INDEX idx_users_status ON users(status);

-- 用户角色表
CREATE TABLE roles (
    id          BIGSERIAL PRIMARY KEY,
    name        VARCHAR(50) UNIQUE NOT NULL,
    description TEXT,
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 用户角色关联表
CREATE TABLE user_roles (
    user_id     BIGINT REFERENCES users(id),
    role_id     BIGINT REFERENCES roles(id),
    PRIMARY KEY (user_id, role_id)
);

-- API Key表
CREATE TABLE api_keys (
    id          BIGSERIAL PRIMARY KEY,
    user_id     BIGINT REFERENCES users(id),
    api_key     VARCHAR(64) UNIQUE NOT NULL,
    api_secret  VARCHAR(128) NOT NULL,
    permissions JSONB DEFAULT '{}',
    status      VARCHAR(20) DEFAULT 'active',
    last_used   TIMESTAMP,
    expires_at  TIMESTAMP,
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_api_keys_key ON api_keys(api_key);
```

# 💰 三、PostgreSQL 表设计

## 3.1 订单相关表

```sql
-- 订单表
CREATE TABLE orders (
    id              BIGINT PRIMARY KEY AUTO_INCREMENT,
    order_id        VARCHAR(64) UNIQUE NOT NULL,
    user_id         BIGINT NOT NULL,
    symbol          VARCHAR(20) NOT NULL,
    side            ENUM('buy', 'sell') NOT NULL,
    order_type      ENUM('market', 'limit', 'stop') NOT NULL,
    price           DECIMAL(20, 8),
    quantity         DECIMAL(20, 8),
    filled_quantity  DECIMAL(20, 8) DEFAULT 0,
    avg_fill_price   DECIMAL(20, 8),
    status           ENUM('pending', 'filled', 'partial', 'cancelled') NOT NULL,
    exchange         VARCHAR(20),
    created_at       DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at       DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_orders_user (user_id),
    INDEX idx_orders_symbol (symbol),
    INDEX idx_orders_status (status),
    INDEX idx_orders_created (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 持仓表
CREATE TABLE positions (
    id              BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id         BIGINT NOT NULL,
    symbol          VARCHAR(20) NOT NULL,
    quantity         DECIMAL(20, 8) NOT NULL,
    avg_entry_price  DECIMAL(20, 8) NOT NULL,
    unrealized_pnl  DECIMAL(20, 8) DEFAULT 0,
    realized_pnl    DECIMAL(20, 8) DEFAULT 0,
    exchange        VARCHAR(20),
    updated_at       DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    UNIQUE KEY uk_user_symbol (user_id, symbol),
    INDEX idx_positions_user (user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 成交记录表
CREATE TABLE trades (
    id              BIGINT PRIMARY KEY AUTO_INCREMENT,
    trade_id        VARCHAR(64) UNIQUE NOT NULL,
    order_id        VARCHAR(64) NOT NULL,
    user_id         BIGINT NOT NULL,
    symbol          VARCHAR(20) NOT NULL,
    side            ENUM('buy', 'sell') NOT NULL,
    price           DECIMAL(20, 8) NOT NULL,
    quantity         DECIMAL(20, 8) NOT NULL,
    fee             DECIMAL(20, 8) DEFAULT 0,
    fee_currency    VARCHAR(10),
    exchange        VARCHAR(20),
    created_at       DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_trades_order (order_id),
    INDEX idx_trades_user (user_id),
    INDEX idx_trades_symbol (symbol),
    INDEX idx_trades_time (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 资金流水表
CREATE TABLE fund_flows (
    id              BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id         BIGINT NOT NULL,
    type            ENUM('deposit', 'withdraw', 'trade', 'fee', 'pnl') NOT NULL,
    amount          DECIMAL(20, 8) NOT NULL,
    balance_after   DECIMAL(20, 8) NOT NULL,
    currency        VARCHAR(10) DEFAULT 'USDT',
    reference_id    VARCHAR(64),
    description     VARCHAR(255),
    created_at       DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_fund_flows_user (user_id),
    INDEX idx_fund_flows_type (type),
    INDEX idx_fund_flows_time (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

# 📊 四、ClickHouse 表设计

## 4.1 行情数据表

```sql
-- K线数据表
CREATE TABLE klines (
    symbol       String,
    timeframe    String,
    open_time    DateTime,
    close_time   DateTime,
    open         Decimal(20, 8),
    high         Decimal(20, 8),
    low          Decimal(20, 8),
    close        Decimal(20, 8),
    volume       Decimal(20, 8),
    quote_volume Decimal(20, 8),
    trades       UInt32,
    INDEX idx_symbol (symbol) TYPE bloom_filter,
    INDEX idx_time (open_time) TYPE minmax
) ENGINE = MergeTree()
PARTITION BY toYYYYMM(open_time)
ORDER BY (symbol, timeframe, open_time)
TTL open_time + INTERVAL 90 DAY;

-- 深度数据表
CREATE TABLE orderbook (
    symbol       String,
    exchange     String,
    timestamp    DateTime,
    bids         String,
    asks         String,
    INDEX idx_symbol_exchange (symbol, exchange) TYPE bloom_filter
) ENGINE = MergeTree()
ORDER BY (symbol, exchange, timestamp)
TTL timestamp + INTERVAL 7 DAY;

-- 价格快照表
CREATE TABLE price_snapshots (
    symbol       String,
    exchange     String,
    price        Decimal(20, 8),
    bid          Decimal(20, 8),
    ask          Decimal(20, 8),
    volume       Decimal(20, 8),
    timestamp    DateTime,
    INDEX idx_symbol (symbol) TYPE bloom_filter
) ENGINE = MergeTree()
ORDER BY (symbol, exchange, timestamp)
TTL timestamp + INTERVAL 30 DAY;
```

## 4.2 因子数据表

```sql
-- 特征数据表
CREATE TABLE features (
    id            UInt64,
    symbol        String,
    timeframe     String,
    timestamp     DateTime,
    returns_1m    Float64,
    returns_5m    Float64,
    returns_1h    Float64,
    volatility    Float64,
    atr           Float64,
    momentum      Float64,
    rsi           Float64,
    macd          Float64,
    ma_5          Float64,
    ma_20         Float64,
    ma_alignment  Int8,
    trend_strength Float64,
    INDEX idx_symbol (symbol) TYPE bloom_filter,
    INDEX idx_time (timestamp) TYPE minmax
) ENGINE = MergeTree()
PARTITION BY toYYYYMM(timestamp)
ORDER BY (symbol, timeframe, timestamp)
TTL timestamp + INTERVAL 90 DAY;

-- 因子数据表
CREATE TABLE factors (
    id              UInt64,
    symbol          String,
    timestamp       DateTime,
    factor_trend    Float64,
    factor_flow     Float64,
    factor_sentiment Float64,
    factor_macro    Float64,
    factor_behavioral Float64,
    factor_historical Float64,
    composite_score Float64,
    regime          String,
    confidence      Float64,
    INDEX idx_symbol (symbol) TYPE bloom_filter,
    INDEX idx_regime (regime) TYPE bloom_filter
) ENGINE = MergeTree()
ORDER BY (symbol, timestamp)
TTL timestamp + INTERVAL 180 DAY;

-- Regime历史表
CREATE TABLE regimes (
    id          UInt64,
    symbol      String,
    timestamp   DateTime,
    regime      String,
    confidence  Float64,
    risk_level  UInt8,
    drivers     String,
    duration    UInt32,
    INDEX idx_symbol (symbol) TYPE bloom_filter
) ENGINE = MergeTree()
ORDER BY (symbol, timestamp)
TTL timestamp + INTERVAL 365 DAY;
```

# 🔧 六、索引设计

## 6.1 索引策略

| 表类型 | 索引类型 | 说明 |
|--------|----------|------|
| 用户表 | B-tree | 等值查询 |
| 订单表 | B-tree + Hash | 复合索引 |
| K线表 | MergeTree | 主键索引 |
| 因子表 | Bloom Filter | 高效去重 |
| 监控表 | Time Series | 时间索引 |

## 6.2 常用查询优化

```sql
-- 优化1：获取用户最近订单
CREATE INDEX idx_orders_user_time ON orders(user_id, created_at DESC);

-- 优化2：获取标的价格历史
CREATE INDEX idx_klines_symbol_time ON klines(symbol, timeframe, open_time DESC);

-- 优化3：获取Regime历史
CREATE INDEX idx_regimes_symbol_time ON regimes(symbol, timestamp DESC);

-- 优化4：获取因子最新值
CREATE INDEX idx_factors_symbol_time ON factors(symbol, timestamp DESC);
```

# 🛡️ 七、数据安全

## 7.1 备份策略

| 数据库 | 备份频率 | 保留时间 | 方式 |
|--------|----------|----------|------|
| PostgreSQL | 每日全量 | 30天 | pg_dump |
| PostgreSQL | 每小时增量 | 7天 | binlog |
| ClickHouse | 每日增量 | 90天 | ALTER TABLE...TTL |
| Redis | 每5分钟 | 1天 | RDB快照 |

## 7.2 权限设计

```sql
-- 应用用户（读写）
CREATE USER 'tradeagent_app'@'%' IDENTIFIED BY 'app_password';
GRANT SELECT, INSERT, UPDATE, DELETE ON tradeagent.* TO 'tradeagent_app'@'%';

-- 报表用户（只读）
CREATE USER 'tradeagent_read'@'%' IDENTIFIED BY 'read_password';
GRANT SELECT ON tradeagent.* TO 'tradeagent_read'@'%';

-- 备份用户（管理员）
CREATE USER 'tradeagent_backup'@'%' IDENTIFIED BY 'backup_password';
GRANT ALL PRIVILEGES ON tradeagent.* TO 'tradeagent_backup'@'%';
```
