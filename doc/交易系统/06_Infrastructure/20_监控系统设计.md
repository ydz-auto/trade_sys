```python
class AlertSender:
    def __init__(self):
        self.channels = {
            "dashboard": DashboardChannel(),
            "telegram": TelegramChannel(),
            "sms": SMSChannel(),
            "email": EmailChannel()
        }

    def send(self, level: str, message: str, context: dict = None):
        channels = ALERT_LEVELS[level]["channels"]
        for channel in channels:
            self.channels[channel].send(level, message, context)
```

```python
class TelegramChannel:
    def __init__(self):
        self.bot_token = config["telegram"]["bot_token"]
        self.chat_id = config["telegram"]["chat_id"]

    def send(self, level: str, message: str, context: dict = None):
        text = f"[{level}] {message}"
        if context:
            text += f"\n\n{json.dumps(context, indent=2)}"

        url = f"https://api.telegram.org/bot{self.bot_token}/sendMessage"
        requests.post(url, json={
            "chat_id": self.chat_id,
            "text": text,
            "parse_mode": "HTML"
        })
```

# 4.2 邮件通知

```python
class EmailChannel:
    def __init__(self):
        self.smtp_server = config["email"]["smtp_server"]
        self.smtp_port = config["email"]["smtp_port"]
        self.sender = config["email"]["sender"]
        self.recipients = config["email"]["recipients"]

    def send(self, level: str, message: str, context: dict = None):
        msg = MIMEMultipart()
        msg["From"] = self.sender
        msg["To"] = ", ".join(self.recipients)
        msg["Subject"] = f"[TradeAgent Alert] {level}: {message[:50]}"

        body = f"{message}\n\nContext:\n{json.dumps(context, indent=2)}"
        msg.attach(MIMEText(body, "plain"))

        with smtplib.SMTP(self.smtp_server, self.smtp_port) as server:
            server.starttls()
            server.login(self.sender, config["email"]["password"])
            server.send_message(msg)
```

# 📈 5. 实时监控

# 5.1 健康检查

```python
class HealthChecker:
    def __init__(self):
        self.services = {
            "data": DataServiceHealthCheck(),
            "factor": FactorServiceHealthCheck(),
            "risk": RiskServiceHealthCheck(),
            "execution": ExecutionServiceHealthCheck()
        }

    def check_all(self) -> dict:
        results = {}
        for name, checker in self.services.items():
            try:
                status = checker.check()
                results[name] = {"status": "OK", "latency_ms": status["latency"]}
            except Exception as e:
                results[name] = {"status": "DOWN", "error": str(e)}

        overall_status = "OK" if all(r["status"] == "OK" for r in results.values()) else "DEGRADED"

        return {
            "overall": overall_status,
            "services": results,
            "timestamp": time.time()
        }
```

# 5.2 性能监控

```python
class PerformanceMonitor:
    def record_latency(self, service: str, operation: str, latency_ms: float):
        metrics.record("latency", latency_ms, tags={
            "service": service,
            "operation": operation
        })

        if latency_ms > self.get_sla_threshold(service, operation):
            alert_sender.send("WARNING", f"High latency: {service}.{operation} = {latency_ms}ms")

    def record_error(self, service: str, error: Exception):
        metrics.increment("errors", tags={
            "service": service,
            "error_type": type(error).__name__
        })
```

# 🎯 6. 交易监控

# 6.1 交易异常检测

```python
class TradingAnomalyDetector:
    def detect_unusual_activity(self, symbol: str) -> bool:
        recent_trades = self.get_recent_trades(symbol, window_minutes=30)

        if len(recent_trades) > self.thresholds["max_trades_per_half_hour"]:
            return True

        total_size = sum(t["size"] for t in recent_trades)
        if total_size > self.thresholds["max_size_per_half_hour"]:
            return True

        return False

    def detect_rapid_signal_reversal(self, symbol: str) -> bool:
        recent_signals = self.get_recent_signals(symbol, window_minutes=15)

        if len(recent_signals) >= 2:
            if recent_signals[-1]["action"] != recent_signals[-2]["action"]:
                return True

        return False
```

# 6.2 PnL监控

```python
class PnLMonitor:
    def check_pnl_limits(self) -> dict:
        current_pnl = state_service.get_current_pnl()
        daily_pnl = state_service.get_daily_pnl()
        max_drawdown = state_service.get_max_drawdown()

        alerts = []

        if daily_pnl < -config["risk"]["limits"]["max_daily_loss"]:
            alerts.append({
                "level": "ERROR",
                "message": f"Daily loss limit exceeded: {daily_pnl:.2%}"
            })

        if max_drawdown > config["risk"]["limits"]["max_drawdown"]:
            alerts.append({
                "level": "ERROR",
                "message": f"Max drawdown exceeded: {max_drawdown:.2%}"
            })

        return alerts
```

# 8.1 监控日志表

```sql
CREATE TABLE monitoring_metrics (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    metric_name VARCHAR(100) NOT NULL,
    metric_type VARCHAR(20) NOT NULL,
    value DECIMAL(20, 10) NOT NULL,
    tags JSON,
    timestamp TIMESTAMP NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_metric_timestamp (metric_name, timestamp),
    INDEX idx_tags (tags)
);

CREATE TABLE alerts (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    alert_id VARCHAR(50) NOT NULL UNIQUE,
    level VARCHAR(20) NOT NULL,
    message TEXT NOT NULL,
    context JSON,
    status VARCHAR(20) DEFAULT 'ACTIVE',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    resolved_at TIMESTAMP,
    INDEX idx_level_status (level, status),
    INDEX idx_created_at (created_at)
);
```
