);

# 🔐 3. 认证机制

# 3.1 密码认证

```python
from argon2 import PasswordHasher

ph = PasswordHasher()

def hash_password(password: str) -> str:
    return ph.hash(password)

def verify_password(password: str, password_hash: str) -> bool:
    try:
        return ph.verify(password_hash, password)
    except:
        return False
```

# 3.3 JWT认证

```python
import jwt
from datetime import datetime, timedelta

SECRET_KEY = "your-secret-key"

def create_token(user_id: str, role: str, expires_hours: int = 24) -> str:
    payload = {
        "user_id": user_id,
        "role": role,
        "exp": datetime.utcnow() + timedelta(hours=expires_hours),
        "iat": datetime.utcnow()
    }
    return jwt.encode(payload, SECRET_KEY, algorithm="HS256")

def verify_token(token: str) -> dict:
    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=["HS256"])
        return payload
    except jwt.ExpiredSignatureError:
        raise AuthError("Token expired")
    except jwt.InvalidTokenError:
        raise AuthError("Invalid token")
```

# ✅ 4. 权限检查

# 4.1 权限检查器

```python
import fnmatch

class PermissionChecker:
    def __init__(self):
        self.role_permissions = ROLES

    def has_permission(self, user_role: str, method: str, path: str) -> bool:
        permissions = self.role_permissions.get(user_role, {}).get("permissions", [])

        for perm in permissions:
            if perm == "*":
                return True

            perm_pattern = perm.replace("*", "*")
            if fnmatch.fnmatch(f"{method} {path}", perm_pattern):
                return True

        return False

    def check(self, user_role: str, method: str, path: str):
        if not self.has_permission(user_role, method, path):
            raise PermissionDeniedError(f"Permission denied: {method} {path}")
```

# 🛡️ 6. 风控权限控制

# 6.1 交易权限

```python
TRADING_PERMISSIONS = {
    "can_trade": True,
    "max_position_size": 0.3,
    "max_leverage": 3,
    "allowed_symbols": ["BTC", "ETH"],
    "allow_high_risk": False
}

def check_trading_permission(user: dict, order: dict) -> bool:
    if not TRADING_PERMISSIONS["can_trade"]:
        return False

    if order["symbol"] not in TRADING_PERMISSIONS["allowed_symbols"]:
        return False

    if order["leverage"] > TRADING_PERMISSIONS["max_leverage"]:
        return False

    if order["size"] > TRADING_PERMISSIONS["max_position_size"]:
        return False

    return True
```
