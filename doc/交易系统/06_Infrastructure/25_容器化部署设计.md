# 🐳 二、Dockerfile 设计

## 2.1 基础镜像

```dockerfile
# Python 基础镜像
FROM python:3.11-slim

# Node 基础镜像
FROM node:20-alpine
```

## 2.2 服务 Dockerfile 模板

```dockerfile
# data-service/Dockerfile
FROM python:3.11-slim

WORKDIR /app

# 安装依赖
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# 复制代码
COPY . .

# 共享库
COPY shared /app/shared

# 环境变量
ENV PYTHONUNBUFFERED=1
ENV SERVICE_NAME=data-service

# 健康检查
HEALTHCHECK --interval=30s --timeout=10s --start-period=5s --retries=3 \
    CMD python -c "import requests; requests.get('http://localhost:8002/health')"

# 启动命令
CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8002"]
```

```yaml
  CLICKHOUSE_PORT: "9000"
```

## 3.3 Deployment 模板

```yaml
# data-service-deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: data-service
  namespace: tradeagent
spec:
  replicas: 2
  selector:
    matchLabels:
      app: data-service
  template:
    metadata:
      labels:
        app: data-service
        version: v1
    spec:
      containers:
        - name: data-service
          image: tradeagent/data-service:latest
          ports:
            - containerPort: 8002
          envFrom:
            - configMapRef:
                name: tradeagent-config
          resources:
            requests:
              memory: "512Mi"
              cpu: "250m"
            limits:
              memory: "1Gi"
              cpu: "500m"
          livenessProbe:
            httpGet:
              path: /health
              port: 8002
            initialDelaySeconds: 10
            periodSeconds: 30
          readinessProbe:
            httpGet:
              path: /health
              port: 8002
            initialDelaySeconds: 5
            periodSeconds: 10
      affinity:
        podAntiAffinity:
          preferredDuringSchedulingIgnoredDuringExecution:
            - weight: 100
              podAffinityTerm:
                labelSelector:
                  matchExpressions:
                    - key: app
                      operator: In
                      values:
                        - data-service
                topologyKey: kubernetes.io/hostname
```

## 4.3 部署命令

```bash
# 安装

# 回滚
helm rollback tradeagent 1 -n tradeagent

# 卸载
helm uninstall tradeagent -n tradeagent
```

# 🔄 五、CI/CD 流程

## 5.1 GitLab CI 配置

```yaml
# .gitlab-ci.yml
stages:
  - build
  - test
  - push
  - deploy

variables:
  IMAGE_REGISTRY: registry.example.com
  DOCKERFILE_PATH: services/${SERVICE_NAME}/Dockerfile

build:
  stage: build
  script:
    - docker build -t ${IMAGE_REGISTRY}/${SERVICE_NAME}:${CI_COMMIT_SHORT_SHA} -f ${DOCKERFILE_PATH} .
    - docker build -t ${IMAGE_REGISTRY}/${SERVICE_NAME}:latest -f ${DOCKERFILE_PATH} .
  only:
    - main
  parallel:
    matrix:
      - SERVICE_NAME: data-service
      - SERVICE_NAME: feature-service
      - SERVICE_NAME: factor-service
      - SERVICE_NAME: regime-service
      - SERVICE_NAME: risk-service
      - SERVICE_NAME: decision-service

test:
  stage: test
  script:
    - docker run ${IMAGE_REGISTRY}/${SERVICE_NAME}:${CI_COMMIT_SHORT_SHA} pytest
  only:
    - main

push:
  stage: push
  script:
    - docker push ${IMAGE_REGISTRY}/${SERVICE_NAME}:${CI_COMMIT_SHORT_SHA}
    - docker push ${IMAGE_REGISTRY}/${SERVICE_NAME}:latest
  only:
    - main

deploy-staging:
  stage: deploy
  script:
    - helm upgrade --install tradeagent ./charts/tradeagent -f ./charts/values-staging.yaml -n tradeagent-staging
  environment:
    name: staging
  only:
    - main

deploy-prod:
  stage: deploy
  script:
    - helm upgrade --install tradeagent ./charts/tradeagent -f ./charts/values-prod.yaml -n tradeagent-prod
  environment:
    name: production
  when: manual
  only:
    - main
```

# 🛡️ 七、运维配置

## 7.1 健康检查

```yaml
# livenessProbe: 判断容器是否需要重启
livenessProbe:
  httpGet:
    path: /health
    port: 8002
  initialDelaySeconds: 30
  periodSeconds: 10
  failureThreshold: 3

# readinessProbe: 判断容器是否接收流量
readinessProbe:
  httpGet:
    path: /health
    port: 8002
  initialDelaySeconds: 5
  periodSeconds: 5
  failureThreshold: 3

# startupProbe: 判断容器是否启动完成
startupProbe:
  httpGet:
    path: /health
    port: 8002
  failureThreshold: 30
  periodSeconds: 10
```

## 7.2 资源限制

```yaml
resources:
  requests:
    memory: "512Mi"    # Guaranteed
    cpu: "250m"       # 0.25 CPU
  limits:
    memory: "1Gi"     # Maximum
    cpu: "500m"       # 0.5 CPU

# 推荐配置
resources:
  data-service:
    requests: { memory: "512Mi", cpu: "250m" }
    limits: { memory: "1Gi", cpu: "500m" }
  feature-service:
    requests: { memory: "512Mi", cpu: "250m" }
    limits: { memory: "1Gi", cpu: "500m" }
  factor-service:
    requests: { memory: "1Gi", cpu: "500m" }
    limits: { memory: "2Gi", cpu: "1000m" }  # 因子计算需要更多资源
  regime-service:
    requests: { memory: "512Mi", cpu: "250m" }
    limits: { memory: "1Gi", cpu: "500m" }
```
