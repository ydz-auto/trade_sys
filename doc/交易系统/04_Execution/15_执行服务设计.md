# 🧠 1. 模块定位

# 1.1 核心作用

将**交易信号**转化为**真实订单**发送到交易所，并管理订单生命周期。

它是**仓位引擎到交易所之间的桥梁**。

# 1.2 在系统中的位置

```
Position Engine → Execution Service → Exchange API → 交易所
```

# ⚙️ 4. 核心功能

# 4.1 订单发送

```python
class ExecutionService:
    def __init__(self, exchange: ExchangeInterface):
        self.exchange = exchange
        self.order_cache = {}

    def submit_order(self, order_request: dict) -> dict:
        if system_config["execution"]["mode"] == "DRY_RUN":
            return self.simulate_order(order_request)

        validated = self.validate_order(order_request)
        order_id = self.exchange.send_order(validated)
        self.order_cache[order_id] = validated
        return {"order_id": order_id, "status": "SUBMITTED"}

    def validate_order(self, order: dict) -> dict:
        if order["size"] <= 0:
            raise OrderValidationError("Invalid order size")
        if order["action"] not in ["OPEN_LONG", "OPEN_SHORT", "CLOSE_LONG", "CLOSE_SHORT"]:
            raise OrderValidationError("Invalid action")
        return order
```

# 4.2 订单跟踪

```python
class OrderTracker:
    def __init__(self):
        self.pending_orders = {}
        self.filled_orders = {}

    def track_order(self, order_id: str):
        while True:
            status = self.exchange.get_order_status(order_id)
            if status["status"] in ["FILLED", "CANCELLED", "REJECTED"]:
                self.handle_order_completed(order_id, status)
                break
            time.sleep(1)

    def handle_order_completed(self, order_id: str, status: dict):
        if status["status"] == "FILLED":
            self.filled_orders[order_id] = status
            event_bus.emit("order_filled", status)
        else:
            event_bus.emit("order_failed", {"order_id": order_id, "reason": status.get("reason")})
```

# 4.3 重试机制

```python
class RetryHandler:
    def __init__(self, max_retries: int = 3, retry_interval: int = 5):
        self.max_retries = max_retries
        self.retry_interval = retry_interval

    def execute_with_retry(self, func, *args, **kwargs):
        for attempt in range(self.max_retries):
            try:
                return func(*args, **kwargs)
            except ExchangeError as e:
                if attempt == self.max_retries - 1:
                    raise
                time.sleep(self.retry_interval)
                logger.warning(f"Retry {attempt + 1}/{self.max_retries}: {e}")
```

# 4.4 订单状态同步

```python
def sync_orders(self):
    remote_orders = self.exchange.get_open_orders()
    local_orders = self.order_tracker.get_local_orders()

    for remote in remote_orders:
        if remote["order_id"] not in local_orders:
            self.order_tracker.add(remote)

    for local_id in local_orders:
        if local_id not in [r["order_id"] for r in remote_orders]:
            status = self.exchange.get_order_status(local_id)
            self.order_tracker.update(local_id, status)
```

# 🛡️ 5. 风控拦截

# 5.1 执行前检查

```python
def pre_execution_check(self, order_request: dict) -> bool:
    risk_state = state_service.get_risk_state()
    if not risk_state["allow_trading"]:
        raise ExecutionBlockedError("Trading not allowed due to risk")

    position_state = state_service.get_position_state()
    new_exposure = position_state["total_exposure"] + order_request["size"]
    if new_exposure > risk_state["max_position"]:
        raise ExecutionBlockedError("Position limit exceeded")

    if order_request["leverage"] > risk_state["max_leverage"]:
        raise ExecutionBlockedError("Leverage limit exceeded")

    return True
```

# 5.2 仓位校验

```python
def validate_position_for_close(self, symbol: str, size: float, side: str) -> bool:
    position = state_service.get_position(symbol)
    if not position:
        raise ExecutionError(f"No position found for {symbol}")

    if side == "CLOSE_LONG" and position["direction"] != "LONG":
        raise ExecutionError(f"Cannot close short position as long")
    if side == "CLOSE_SHORT" and position["direction"] != "SHORT":
        raise ExecutionError(f"Cannot close long position as short")

    if size > position["size"]:
        raise ExecutionError(f"Close size {size} exceeds position size {position['size']}")

    return True
```

# 6.2 Binance实现

```python
class BinanceExchange(ExchangeInterface):
    def __init__(self, api_key: str, secret_key: str, testnet: bool = False):
        self.client = BinanceClient(api_key, secret_key, testnet)

    def send_order(self, order: dict) -> str:
        response = self.client.futures_create_order(
            symbol=order["symbol"],
            side=order["side"],
            type=order["type"],
            quantity=order["quantity"],
            price=order.get("price"),
            stopPrice=order.get("stopPrice")
        )
        return str(response["orderId"])

    def get_positions(self) -> list:
        positions = self.client.futures_position_information()
        return [p for p in positions if float(p["positionAmt"]) != 0]
```

# 6.3 OKX实现

```python
class OKXExchange(ExchangeInterface):
    def __init__(self, api_key: str, secret_key: str, passphrase: str, testnet: bool = False):
        self.client = OKXClient(api_key, secret_key, passphrase, testnet)

    def send_order(self, order: dict) -> str:
        inst_id = f"{order['symbol']}-USDT-SWAP"
        response = self.client.place_order(
            instId=inst_id,
            tdMode="cross",
            side=order["side"],
            ordType=order["type"],
            sz=order["quantity"],
            px=order.get("price"),
            slTriggerPx=order.get("stopPrice")
        )
        return response["ordId"]
```
