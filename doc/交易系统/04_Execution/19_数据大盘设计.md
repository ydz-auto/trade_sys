# 数据大盘设计文档（Data Dashboard Spec v1）

---

# 🧠 一、模块定位

## 1.1 核心作用

数据大盘是**交易系统的"眼睛"**，让用户实时感知市场全貌，包括：
- 实时行情数据
- ETF资金流向
- 宏观市场数据
- 新闻资讯流
- 社交媒体动态
- 市场情绪指数

## 1.2 在系统中的位置

```
Data Service → llm_service → WebSocket Server → 前端大盘
                                              ↓
                                      React + Redux/WebSocket
```

## 1.3 设计目标

```
✅ 实时数据展示（WebSocket推送，延迟 < 1秒）
✅ 新闻/社交媒体流式展示
✅ 情绪指数可视化
✅ 数据采集状态监控
✅ 多数据源质量一目了然
```

## 4.2 核心Hook设计

### 4.2.1 useWebSocket

```typescript
// hooks/useWebSocket.ts
import { useEffect, useRef, useCallback } from 'react';

interface WebSocketMessage {
  topic: string;
  timestamp: number;
  data: any;
}

interface UseWebSocketOptions {
  url: string;
  topics: string[];
  onMessage: (message: WebSocketMessage) => void;
  onConnect?: () => void;
  onDisconnect?: () => void;
  reconnectInterval?: number;
}

export function useWebSocket({
  url,
  topics,
  onMessage,
  onConnect,
  onDisconnect,
  reconnectInterval = 3000
}: UseWebSocketOptions) {
  const wsRef = useRef<WebSocket | null>(null);
  const reconnectTimerRef = useRef<NodeJS.Timeout>();

  const connect = useCallback(() => {
    const ws = new WebSocket(url);
    
    ws.onopen = () => {
      console.log('WebSocket connected');
      ws.send(JSON.stringify({ action: 'subscribe', topics }));
      onConnect?.();
    };

    ws.onmessage = (event) => {
      try {
        const message = JSON.parse(event.data);
        onMessage(message);
      } catch (e) {
        console.error('Failed to parse message:', e);
      }
    };

    ws.onclose = () => {
      console.log('WebSocket disconnected');
      onDisconnect?.();
      reconnectTimerRef.current = setTimeout(connect, reconnectInterval);
    };

    ws.onerror = (error) => {
      console.error('WebSocket error:', error);
    };

    wsRef.current = ws;
  }, [url, topics, onMessage, onConnect, onDisconnect, reconnectInterval]);

  useEffect(() => {
    connect();
    return () => {
      if (reconnectTimerRef.current) {
        clearTimeout(reconnectTimerRef.current);
      }
      wsRef.current?.close();
    };
  }, [connect]);

  const send = useCallback((message: any) => {
    wsRef.current?.send(JSON.stringify(message));
  }, []);

  return { send };
}
```

### 4.2.2 useNews

```typescript
// hooks/useNews.ts
import { useState, useEffect, useCallback } from 'react';
import { useWebSocket } from './useWebSocket';

interface NewsItem {
  id: string;
  source: string;
  title: string;
  sentiment: number;
  sentiment_label: string;
  importance: number;
  published_at: number;
}

interface UseNewsOptions {
  limit?: number;
  language?: 'cn' | 'en' | 'all';
}

export function useNews(options: UseNewsOptions = {}) {
  const [news, setNews] = useState<NewsItem[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const handleMessage = useCallback((message: any) => {
    if (message.topic === 'news.latest') {
      setNews((prev) => {
        const newItems = message.data.items.filter(
          (item: NewsItem) => !prev.some((n) => n.id === item.id)
        );
        return [...newItems, ...prev].slice(0, options.limit || 50);
      });
      setLoading(false);
    }
  }, [options.limit]);

  const { send } = useWebSocket({
    url: 'ws://api.tradeagent.com/ws/data',
    topics: ['news.latest'],
    onMessage: handleMessage,
    onConnect: () => setLoading(false),
  });

  const fetchMore = useCallback(() => {
    send({ action: 'fetch_more', topic: 'news.latest', limit: options.limit });
  }, [send, options.limit]);

  return { news, loading, error, fetchMore };
}
```

  if (value >= -0.6) return '#f97316'; // 偏消极 - 橙色
  return '#ef4444'; // 消极 - 红色
};

// 恐慌贪婪指数仪表
const gaugeStyle = {
  background: `conic-gradient(
    ${colors.very_fear} 0deg 36deg,
    ${colors.fear} 36deg 72deg,
    ${colors.neutral} 72deg 108deg,
    ${colors.greed} 108deg 144deg,
    ${colors.very_greed} 144deg 180deg
  )`
};
