# 🔧 四、TDP消息处理 `feature/base.py`

```python
import time
from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import Dict, Any, List, Optional
import logging

from shared.tdp import tdp_formatter
from shared.state import state_client

logger = logging.getLogger(__name__)

@dataclass
class FeatureResult:
    success: bool
    symbol: str
    features: Dict[str, float]
    timestamp: int
    timeframe: str
    error: Optional[str] = None

class BaseFeatureCalculator(ABC):
    def __init__(self, name: str, symbols: List[str], timeframe: str = "5m"):
        self.name = name
        self.symbols = symbols
        self.timeframe = timeframe
        self.price_cache: Dict[str, List[float]] = {}
        self.feature_cache: Dict[str, Any] = {}

    def update_price(self, symbol: str, price: float, timestamp: int):
        if symbol not in self.price_cache:
            self.price_cache[symbol] = []

        self.price_cache[symbol].append({
            "price": price,
            "timestamp": timestamp
        })

        max_points = 100
        if len(self.price_cache[symbol]) > max_points:
            self.price_cache[symbol] = self.price_cache[symbol][-max_points:]

    def get_prices(self, symbol: str) -> List[float]:
        return [p["price"] for p in self.price_cache.get(symbol, [])]

    def get_timestamps(self, symbol: str) -> List[int]:
        return [p["timestamp"] for p in self.price_cache.get(symbol, [])]

    @abstractmethod
    def calculate(self, symbol: str) -> Dict[str, float]:
        pass

    def compute_for_all(self) -> List[FeatureResult]:
        results = []
        for symbol in self.symbols:
            try:
                features = self.calculate(symbol)
                results.append(FeatureResult(
                    success=True,
                    symbol=symbol,
                    features=features,
                    timestamp=int(time.time()),
                    timeframe=self.timeframe
                ))
            except Exception as e:
                logger.error(f"{self.name} 计算失败 {symbol}: {e}")
                results.append(FeatureResult(
                    success=False,
                    symbol=symbol,
                    features={},
                    timestamp=int(time.time()),
                    timeframe=self.timeframe,
                    error=str(e)
                ))
        return results
```

# 🔧 五、收益率计算 `feature/returns.py`

```python
import numpy as np
from typing import Dict, List
from .base import BaseFeatureCalculator
import logging

logger = logging.getLogger(__name__)

class ReturnsCalculator(BaseFeatureCalculator):
    def __init__(self, symbols: List[str], timeframe: str = "5m"):
        super().__init__("returns", symbols, timeframe)

    def calculate(self, symbol: str) -> Dict[str, float]:
        prices = self.get_prices(symbol)

        if len(prices) < 2:
            return {"returns_1m": 0.0, "returns_5m": 0.0, "returns_1h": 0.0}

        returns_1m = (prices[-1] - prices[-2]) / prices[-2] if len(prices) >= 2 else 0.0
        returns_5m = (prices[-1] - prices[-min(5, len(prices))]) / prices[-min(5, len(prices))] if len(prices) >= 5 else returns_1m
        returns_1h = (prices[-1] - prices[-min(12, len(prices))]) / prices[-min(12, len(prices))] if len(prices) >= 12 else returns_5m

        return {
            "returns_1m": float(returns_1m),
            "returns_5m": float(returns_5m),
            "returns_1h": float(returns_1h),
            "returns_24h": float(self._calculate_24h_return(prices))
        }

    def _calculate_24h_return(self, prices: List[float]) -> float:
        if len(prices) < 24:
            if len(prices) >= 2:
                return (prices[-1] - prices[0]) / prices[0]
            return 0.0

        current = prices[-1]
        past_24 = prices[-24]
        return (current - past_24) / past_24
```

# 🔧 六、波动率计算 `feature/volatility.py`

```python
import numpy as np
from typing import Dict
from .base import BaseFeatureCalculator
from shared.config import settings
import logging

logger = logging.getLogger(__name__)

class VolatilityCalculator(BaseFeatureCalculator):
    def __init__(self, symbols: list, timeframe: str = "5m"):
        super().__init__("volatility", symbols, timeframe)
        self.window = settings.VOLATILITY_WINDOW

    def calculate(self, symbol: str) -> Dict[str, float]:
        prices = self.get_prices(symbol)

        if len(prices) < self.window:
            return self._empty_features()

        price_array = np.array(prices[-self.window:])

        returns = np.diff(price_array) / price_array[:-1]

        volatility = float(np.std(returns))

        atr = self._calculate_atr(symbol)

        return {
            "volatility": volatility,
            "volatility_annual": volatility * np.sqrt(288 * 365),
            "atr": atr,
            "atr_percent": atr / prices[-1] if prices[-1] > 0 else 0.0,
            "volatility_rank": self._calculate_volatility_rank(symbol)
        }

    def _calculate_atr(self, symbol: str) -> float:
        prices = self.get_prices(symbol)
        if len(prices) < settings.ATR_PERIOD + 1:
            return 0.0

        tr_list = []
        for i in range(1, min(settings.ATR_PERIOD + 1, len(prices))):
            high = prices[i] * 1.001
            low = prices[i] * 0.999
            prev_close = prices[i - 1]
            tr = max(high - low, abs(high - prev_close), abs(low - prev_close))
            tr_list.append(tr)

        return np.mean(tr_list) if tr_list else 0.0

    def _calculate_volatility_rank(self, symbol: str) -> float:
        current_vol = self.calculate(symbol).get("volatility", 0)
        historical_vols = list(self.feature_cache.get(f"{symbol}_hist", []))
        if not historical_vols:
            return 0.5

        rank = sum(1 for v in historical_vols if v < current_vol) / len(historical_vols)
        return float(rank)

    def _empty_features(self) -> Dict[str, float]:
        return {
            "volatility": 0.0,
            "volatility_annual": 0.0,
            "atr": 0.0,
            "atr_percent": 0.0,
            "volatility_rank": 0.5
        }
```

# 🔧 七、动量计算 `feature/momentum.py`

```python
import numpy as np
from typing import Dict
from .base import BaseFeatureCalculator
from shared.config import settings
import logging

logger = logging.getLogger(__name__)

class MomentumCalculator(BaseFeatureCalculator):
    def __init__(self, symbols: list, timeframe: str = "5m"):
        super().__init__("momentum", symbols, timeframe)
        self.window = settings.MOMENTUM_WINDOW

    def calculate(self, symbol: str) -> Dict[str, float]:
        prices = self.get_prices(symbol)

        if len(prices) < self.window:
            return self._empty_features()

        current_price = prices[-1]
        ma = np.mean(prices[-self.window:])

        momentum = (current_price - prices[-self.window]) / prices[-self.window]

        rsi = self._calculate_rsi(prices)

        macd = self._calculate_macd(prices)

        return {
            "momentum": float(momentum),
            "momentum_ma_ratio": float(current_price / ma - 1) if ma > 0 else 0.0,
            "rsi": float(rsi),
            "macd": float(macd["macd"]),
            "macd_signal": float(macd["signal"]),
            "macd_histogram": float(macd["histogram"])
        }

    def _calculate_rsi(self, prices: list, period: int = 14) -> float:
        if len(prices) < period + 1:
            return 50.0

        deltas = np.diff(prices[-period - 1:])
        gains = np.where(deltas > 0, deltas, 0)
        losses = np.where(deltas < 0, -deltas, 0)

        avg_gain = np.mean(gains)
        avg_loss = np.mean(losses)

        if avg_loss == 0:
            return 100.0

        rs = avg_gain / avg_loss
        rsi = 100 - (100 / (1 + rs))
        return float(rsi)

    def _calculate_macd(self, prices: list, fast: int = 12, slow: int = 26, signal: int = 9) -> Dict[str, float]:
        if len(prices) < slow + signal:
            return {"macd": 0.0, "signal": 0.0, "histogram": 0.0}

        prices_array = np.array(prices)

        ema_fast = self._ema(prices_array, fast)
        ema_slow = self._ema(prices_array, slow)

        macd_line = ema_fast - ema_slow
        signal_line = self._ema(np.array([macd_line] * len(prices_array)).flatten(), signal)

        macd = float(macd_line)
        sig = float(signal_line[-1]) if len(signal_line) > 0 else 0.0

        return {
            "macd": macd,
            "signal": sig,
            "histogram": macd - sig
        }

    def _ema(self, data: np.ndarray, period: int) -> np.ndarray:
        ema = np.zeros_like(data)
        ema[0] = data[0]
        multiplier = 2 / (period + 1)

        for i in range(1, len(data)):
            ema[i] = (data[i] - ema[i - 1]) * multiplier + ema[i - 1]

        return ema

    def _empty_features(self) -> Dict[str, float]:
        return {
            "momentum": 0.0,
            "momentum_ma_ratio": 0.0,
            "rsi": 50.0,
            "macd": 0.0,
            "macd_signal": 0.0,
            "macd_histogram": 0.0
        }
```

# 🔧 八、趋势特征 `feature/trend.py`

```python
import numpy as np
from typing import Dict
from .base import BaseFeatureCalculator
from shared.config import settings
import logging

logger = logging.getLogger(__name__)

class TrendCalculator(BaseFeatureCalculator):
    def __init__(self, symbols: list, timeframe: str = "5m"):
        super().__init__("trend", symbols, timeframe)
        self.ma_short = settings.MA_SHORT
        self.ma_long = settings.MA_LONG

    def calculate(self, symbol: str) -> Dict[str, float]:
        prices = self.get_prices(symbol)

        if len(prices) < self.ma_long:
            return self._empty_features()

        prices_array = np.array(prices)

        ma_5 = np.mean(prices_array[-self.ma_short:])
        ma_20 = np.mean(prices_array[-self.ma_long:])

        ma_slope = self._calculate_slope(prices_array[-self.ma_long:])

        ma_alignment = 1 if ma_5 > ma_20 else -1

        trend_strength = abs(ma_5 / ma_20 - 1) if ma_20 > 0 else 0.0

        return {
            "ma_5": float(ma_5),
            "ma_20": float(ma_20),
            "ma_alignment": float(ma_alignment),
            "trend_strength": float(trend_strength),
            "ma_slope": float(ma_slope),
            "price_ma5_ratio": float(prices[-1] / ma_5 - 1) if ma_5 > 0 else 0.0,
            "price_ma20_ratio": float(prices[-1] / ma_20 - 1) if ma_20 > 0 else 0.0
        }

    def _calculate_slope(self, prices: np.ndarray) -> float:
        if len(prices) < 2:
            return 0.0

        x = np.arange(len(prices))
        slope, _ = np.polyfit(x, prices, 1)

        normalized_slope = slope / np.mean(prices) if np.mean(prices) > 0 else 0.0
        return float(normalized_slope)

    def _empty_features(self) -> Dict[str, float]:
        return {
            "ma_5": 0.0,
            "ma_20": 0.0,
            "ma_alignment": 0.0,
            "trend_strength": 0.0,
            "ma_slope": 0.0,
            "price_ma5_ratio": 0.0,
            "price_ma20_ratio": 0.0
        }
```

# 🔧 九、资金流特征 `feature/flow.py`

```python
from typing import Dict, Optional
from .base import BaseFeatureCalculator
import logging

logger = logging.getLogger(__name__)

class FlowFeatureProcessor:
    def __init__(self):
        self.etf_cache = {}
        self.prev_net_flow = {}

    def process_etf_flow(self, symbol: str, inflow: float, outflow: float) -> Dict[str, float]:
        net_flow = inflow - outflow
        acceleration = 0.0

        if symbol in self.prev_net_flow:
            change = net_flow - self.prev_net_flow[symbol]
            acceleration = change / abs(self.prev_net_flow[symbol]) if self.prev_net_flow[symbol] != 0 else 0.0

        self.prev_net_flow[symbol] = net_flow

        direction = 1 if net_flow > 0 else -1

        return {
            f"{symbol.lower()}_net_flow": float(net_flow),
            f"{symbol.lower()}_flow_direction": float(direction),
            f"{symbol.lower()}_flow_acceleration": float(acceleration),
            f"{symbol.lower()}_inflow": float(inflow),
            f"{symbol.lower()}_outflow": float(outflow)
        }

    def aggregate_flow_features(self, flows: Dict[str, Dict]) -> Dict[str, float]:
        features = {}

        btc_flow = flows.get("BTC", {})
        eth_flow = flows.get("ETH", {})

        if btc_flow:
            features.update(self.process_etf_flow("BTC", btc_flow.get("inflow", 0), btc_flow.get("outflow", 0)))

        if eth_flow:
            features.update(self.process_etf_flow("ETH", eth_flow.get("inflow", 0), eth_flow.get("outflow", 0)))

        total_inflow = btc_flow.get("inflow", 0) + eth_flow.get("inflow", 0)
        total_outflow = btc_flow.get("outflow", 0) + eth_flow.get("outflow", 0)
        features["total_net_flow"] = float(total_inflow - total_outflow)
        features["flow_dominance"] = float(btc_flow.get("inflow", 0) / total_inflow) if total_inflow > 0 else 0.5

        return features
```

# 🔧 十、宏观特征 `feature/macro.py`

```python
from typing import Dict, Optional
from .base import BaseFeatureCalculator
import logging

logger = logging.getLogger(__name__)

class MacroFeatureProcessor:
    def __init__(self):
        self.gold_cache = []
        self.oil_cache = []
        self.max_cache_size = 100

    def process_macro(self, gold_price: Optional[float], oil_price: Optional[float], prev_gold: Optional[float] = None, prev_oil: Optional[float] = None) -> Dict[str, float]:
        features = {}

        if gold_price is not None and prev_gold is not None:
            gold_return = (gold_price - prev_gold) / prev_gold
            features["gold_return"] = float(gold_return)
            features["gold_price"] = float(gold_price)
            self.gold_cache.append(gold_return)
            if len(self.gold_cache) > self.max_cache_size:
                self.gold_cache.pop(0)

        if oil_price is not None and prev_oil is not None:
            oil_return = (oil_price - prev_oil) / prev_oil
            features["oil_return"] = float(oil_return)
            features["oil_price"] = float(oil_price)
            self.oil_cache.append(oil_return)
            if len(self.oil_cache) > self.max_cache_size:
                self.oil_cache.pop(0)

        features["macro_signal"] = self._calculate_macro_signal(features)

        return features

    def _calculate_macro_signal(self, features: Dict[str, float]) -> float:
        gold_return = features.get("gold_return", 0)
        oil_return = features.get("oil_return", 0)

        macro_signal = -(0.6 * gold_return + 0.4 * oil_return)

        return float(macro_signal)

    def get_historical_macro_features(self, lookback: int = 7) -> Dict[str, float]:
        features = {}

        if self.gold_cache:
            features["gold_return_ma"] = sum(self.gold_cache[-lookback:]) / min(lookback, len(self.gold_cache))
        else:
            features["gold_return_ma"] = 0.0

        if self.oil_cache:
            features["oil_return_ma"] = sum(self.oil_cache[-lookback:]) / min(lookback, len(self.oil_cache))
        else:
            features["oil_return_ma"] = 0.0

        return features
```

```python
import asyncio
from typing import Dict, List
from ..feature.returns import ReturnsCalculator
from ..feature.volatility import VolatilityCalculator
from ..feature.momentum import MomentumCalculator
from ..feature.trend import TrendCalculator
from .base import BaseFeatureCalculator
import logging

logger = logging.getLogger(__name__)

class PriceFeatureCalculator:
    def __init__(self, symbols: List[str], timeframe: str = "5m"):
        self.symbols = symbols
        self.timeframe = timeframe

        self.returns_calc = ReturnsCalculator(symbols, timeframe)
        self.volatility_calc = VolatilityCalculator(symbols, timeframe)
        self.momentum_calc = MomentumCalculator(symbols, timeframe)
        self.trend_calc = TrendCalculator(symbols, timeframe)

        self.calculators = [
            self.returns_calc,
            self.volatility_calc,
            self.momentum_calc,
            self.trend_calc
        ]

    def update_price(self, symbol: str, price: float, timestamp: int):
        for calc in self.calculators:
            if hasattr(calc, "update_price"):
                calc.update_price(symbol, price, timestamp)
            elif hasattr(calc, "price_cache"):
                if symbol not in calc.price_cache:
                    calc.price_cache[symbol] = []
                calc.price_cache[symbol].append({"price": price, "timestamp": timestamp})
                if len(calc.price_cache[symbol]) > 100:
                    calc.price_cache[symbol] = calc.price_cache[symbol][-100:]

    async def calculate_all(self) -> Dict[str, Dict[str, float]]:
        features = {}

        for symbol in self.symbols:
            symbol_features = {}

            returns = self.returns_calc.calculate(symbol)
            symbol_features.update(returns)

            volatility = self.volatility_calc.calculate(symbol)
            symbol_features.update(volatility)

            momentum = self.momentum_calc.calculate(symbol)
            symbol_features.update(momentum)

            trend = self.trend_calc.calculate(symbol)
            symbol_features.update(trend)

            features[symbol] = symbol_features

        return features
```

```python
import asyncio
from typing import Dict, List
from .price_calculator import PriceFeatureCalculator
from ..feature.flow import FlowFeatureProcessor
from ..feature.macro import MacroFeatureProcessor
import logging

logger = logging.getLogger(__name__)

class FeatureAggregator:
    def __init__(self, symbols: List[str]):
        self.symbols = symbols
        self.price_calculator = PriceFeatureCalculator(symbols)
        self.flow_processor = FlowFeatureProcessor()
        self.macro_processor = MacroFeatureProcessor()
        self.prev_prices = {}
        self.prev_gold = None
        self.prev_oil = None

    def update_market_data(self, crypto_data: Dict, etf_data: Dict = None, macro_data: Dict = None):
        if "crypto" in crypto_data:
            for symbol, data in crypto_data["crypto"].items():
                price = data.get("price")
                timestamp = data.get("timestamp", 0)
                if price:
                    self.price_calculator.update_price(symbol, price, timestamp)
                    self.prev_prices[symbol] = price

        if etf_data:
            pass

    async def compute_features(self) -> Dict:
        price_features = await self.price_calculator.calculate_all()

        flow_features = {}
        if hasattr(self, "etf_data"):
            flow_features = self.flow_processor.aggregate_flow_features(self.etf_data)

        macro_features = self.macro_processor.process_macro(
            gold_price=self.macro_data.get("gold"),
            oil_price=self.macro_data.get("oil"),
            prev_gold=self.prev_gold,
            prev_oil=self.prev_oil
        )

        all_features = {
            "price_features": price_features,
            "flow_features": flow_features,
            "macro_features": macro_features,
            "timestamp": int(asyncio.get_event_loop().time())
        }

        return all_features
```

```python
            value_deserializer=lambda m: json.loads(m.decode('utf-8')),
            auto_offset_reset='latest'
        )
        await self.consumer.start()
        logger.info("Feature Consumer 已启动")

    async def stop(self):
        if self.consumer:
            await self.consumer.stop()

    async def consume(self):
        async for message in self.consumer:
            try:
                tdp_message = message.value

                if tdp_message.get("status") != "OK":
                    logger.warning(f"收到异常TDP消息: {tdp_message.get('meta', {}).get('status')}")
                    continue

                market = tdp_message.get("market", {})

                crypto = market.get("crypto", {})
                if crypto:
                    self.aggregator.update_market_data({"crypto": crypto})

                etf = market.get("etf", {})
                if etf:
                    self.aggregator.etf_data = etf

                macro = market.get("commodities", {})
                if macro:
                    self.aggregator.macro_data = {
                        "gold": macro.get("gold", {}).get("price"),
                        "oil": macro.get("oil", {}).get("price")
                    }

            except Exception as e:
                logger.error(f"消费消息失败: {e}")
```

```python
from clickhouse_driver import Client
from typing import Dict, List
import logging
from ..config import settings

logger = logging.getLogger(__name__)

class ClickHouseWriter:
    def __init__(self):
        self.client = None
        self.table = settings.CLICKHOUSE_TABLE_FEATURES

    def connect(self):
        self.client = Client(
            host=settings.CLICKHOUSE_HOST,
            port=settings.CLICKHOUSE_PORT,
            database=settings.CLICKHOUSE_DATABASE
        )
        logger.info("ClickHouse 连接成功")

    def write_features(self, symbol: str, features: Dict, timestamp: int, timeframe: str = "5m"):
        if not self.client:
            self.connect()

        feature_values = []
        for key, value in features.items():
            if isinstance(value, (int, float)):
                feature_values.append(f"'{key}', {value}")

        if not feature_values:
            return

        sql = f"""
        INSERT INTO {self.table} (symbol, timestamp, timeframe, {', '.join([f.split(',')[0] for f in feature_values])})
        VALUES ('{symbol}', {timestamp}, '{timeframe}', {', '.join([f.split(',')[1] for f in feature_values])})
        """

        try:
            self.client.execute(sql)
            logger.debug(f"特征已写入 ClickHouse: {symbol}")
        except Exception as e:
            logger.error(f"写入 ClickHouse 失败: {e}")

    def create_table(self):
        if not self.client:
            self.connect()

        sql = f"""
        CREATE TABLE IF NOT EXISTS {self.table} (
            symbol String,
            timestamp Int64,
            timeframe String,
            returns_1m Float64,
            returns_5m Float64,
            returns_1h Float64,
            returns_24h Float64,
            volatility Float64,
            volatility_annual Float64,
            atr Float64,
            atr_percent Float64,
            momentum Float64,
            rsi Float64,
            macd Float64,
            macd_signal Float64,
            ma_5 Float64,
            ma_20 Float64,
            ma_alignment Int8,
            trend_strength Float64,
            net_flow Float64,
            flow_direction Int8,
            gold_return Float64,
            oil_return Float64,
            macro_signal Float64
        ) ENGINE = MergeTree()
        ORDER BY (symbol, timestamp)
        """
        try:
            self.client.execute(sql)
            logger.info(f"表 {self.table} 创建成功")
        except Exception as e:
            logger.error(f"创建表失败: {e}")
```

# 🔧 十六、FastAPI主入口 `main.py`

```python
from fastapi import FastAPI, WebSocket
from contextlib import asynccontextmanager
import asyncio
import logging

from .queue.consumer import FeatureConsumer
from .queue.producer import feature_producer
from .storage.clickhouse_writer import clickhouse_writer
from .websocket.manager import ws_manager
from shared.state import state_client
from shared.config import settings

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

SYMBOLS = ["BTC", "ETH", "SOL", "DOGE"]
consumer = FeatureConsumer(SYMBOLS)

@asynccontextmanager
async def lifespan(app: FastAPI):
    await feature_producer.start()
    clickhouse_writer.connect()
    asyncio.create_task(run_feature_calculation())
    asyncio.create_task(consumer.consume())
    state_client.update_data_status("RUNNING")
    yield
    await consumer.stop()
    await feature_producer.stop()
    state_client.update_data_status("STOPPED")

app = FastAPI(
    title="TradeAgent Feature Service",
    version="1.0.0",
    lifespan=lifespan
)

@app.websocket("/ws/features")
async def websocket_endpoint(websocket: WebSocket):
    await ws_manager.connect(websocket)
    try:
        while True:
            await websocket.receive_text()
    except Exception:
        ws_manager.disconnect(websocket)

@app.get("/health")
async def health():
    return {
        "status": "ok",
        "service": settings.SERVICE_NAME
    }

@app.get("/api/v1/features/{symbol}")
async def get_features(symbol: str):
    return consumer.aggregator.price_calculator.price_cache.get(symbol, {})

@app.get("/api/v1/features/latest")
async def get_all_features():
    features = await consumer.aggregator.compute_features()
    return features

async def run_feature_calculation():
    while True:
        try:
            features = await consumer.aggregator.compute_features()

            await feature_producer.send_features(features)

            for symbol, symbol_features in features.get("price_features", {}).items():
                clickhouse_writer.write_features(
                    symbol=symbol,
                    features=symbol_features,
                    timestamp=features.get("timestamp", 0)
                )

            await ws_manager.broadcast({
                "type": "features",
                "data": features
            })

        except Exception as e:
            logger.error(f"特征计算失败: {e}")

        await asyncio.sleep(60)
```
