# Backtest Engine（回测系统）设计文档

# 🧠 1. 模块定位

# 1.1 核心作用

使用**历史数据**模拟交易策略，验证策略盈利能力、风险水平，为实盘提供参考。

# 1.2 回测目标

```
✅ 验证策略是否盈利
✅ 评估风险水平（最大回撤、夏普比率）
✅ 优化策略参数
✅ 识别策略缺陷
✅ 对比不同策略表现
```

# 🔄 3. 回测引擎核心

# 3.1 回测器主循环

```python
class BacktestEngine:
    def __init__(self, config: dict):
        self.config = config
        self.state_manager = BacktestStateManager(config["initial_balance"])
        self.data_feeder = DataFeeder(config)
        self.strategy = StrategyLoader.load(config["strategy"])

    def run(self):
        self.logger.info("Starting backtest...")

        for timestamp in self.data_feeder:
            self.on_bar(timestamp)

        return self.generate_report()

    def on_bar(self, bar_data: dict):
        features = self.feature_service.calculate(bar_data)
        signals = self.strategy.generate_signals(features)

        for signal in signals:
            if self.risk_service.check(signal):
                self.execute_signal(signal)

        self.state_manager.update_state(bar_data)
```

# 3.2 数据供给器

```python
class DataFeeder:
    def __init__(self, config: dict):
        self.start_date = config["start_date"]
        self.end_date = config["end_date"]
        self.symbols = config["symbols"]
        self.timeframe = config["timeframe"]

    def __iter__(self):
        for timestamp in self.get_timestamps():
            yield self.get_bar_data(timestamp)
```

# 3.3 信号执行

```python
def execute_signal(self, signal: dict):
    if signal["action"] == "BUY":
        order = self.create_order("OPEN_LONG", signal)
    elif signal["action"] == "SELL":
        order = self.create_order("CLOSE_LONG", signal)
    elif signal["action"] == "SHORT":
        order = self.create_order("OPEN_SHORT", signal)

    simulated_fill = self.simulate_fill(order)
    self.state_manager.apply_fill(simulated_fill)
    self.trade_logger.log_trade(simulated_fill)
```

# 4. 模拟成交

# 4.1 市价单成交

```python
def simulate_market_fill(self, order: dict, bar: dict) -> dict:
    slippage = self.config["execution"]["slippage"]
    price = bar["close"]

    if order["action"] in ["OPEN_LONG", "CLOSE_SHORT"]:
        fill_price = price * (1 + slippage)
    else:
        fill_price = price * (1 - slippage)

    return {
        "order_id": f"bt_{uuid.uuid4().hex[:12]}",
        "symbol": order["symbol"],
        "action": order["action"],
        "size": order["size"],
        "price": price,
        "fill_price": fill_price,
        "commission": fill_price * order["size"] * self.config["execution"]["commission"],
        "timestamp": bar["timestamp"]
    }
```

# 4.2 限价单成交

```python
def simulate_limit_fill(self, order: dict, bar: dict) -> dict | None:
    if order["action"] == "BUY" and bar["low"] <= order["limit_price"]:
        fill_price = min(bar["open"], order["limit_price"])
    elif order["action"] == "SELL" and bar["high"] >= order["limit_price"]:
        fill_price = max(bar["open"], order["limit_price"])
    else:
        return None

    return self.create_fill(order, fill_price, bar["timestamp"])
```

```python
class PerformanceCalculator:
    def calculate(self, trades: list, balance: float) -> dict:
        equity_curve = self.build_equity_curve(trades, balance)
        returns = self.calculate_returns(equity_curve)

        return {
            "total_return": self.total_return(equity_curve),
            "total_return_percent": self.total_return_percent(equity_curve),
            "annualized_return": self.annualized_return(returns),
            "max_drawdown": self.max_drawdown(equity_curve),
            "max_drawdown_percent": self.max_drawdown_percent(equity_curve),
            "sharpe_ratio": self.sharpe_ratio(returns),
            "sortino_ratio": self.sortino_ratio(returns),
            "win_rate": self.win_rate(trades),
            "profit_factor": self.profit_factor(trades),
            "total_trades": len(trades),
            "avg_trade_return": self.avg_trade_return(trades)
        }

    def sharpe_ratio(self, returns: list, risk_free_rate: float = 0.02) -> float:
        if not returns:
            return 0
        excess_returns = np.array(returns) - risk_free_rate / 252
        return np.mean(excess_returns) / np.std(excess_returns) * np.sqrt(252)
```

# ⚠️ 9. 常见错误与防范

# 9.1 常见错误

```
❌ 前视偏差（look-ahead bias）
❌ 未来函数
❌ 过拟合
❌ 生存者偏差
❌ 流动性忽略
❌ 滑点假设不合理
```

# 9.2 防范措施

```python
def prevent_look_ahead(bar_data: dict):
    bar_data["close"] = None
    bar_data["high"] = None
    bar_data["low"] = None

    for indicator in bar_data["indicators"]:
        indicator["value"] = indicator.get("previous_value")

def add_realistic_slippage(order: dict, bar: dict):
    slippage = calculate_realistic_slippage(order["size"], bar["volume"])
    return slippage
```
